001package org.gbif.portal.action.species;
002
003import org.gbif.api.model.checklistbank.NameUsage;
004import org.gbif.api.model.common.paging.PagingRequest;
005import org.gbif.api.model.common.paging.PagingResponse;
006
007public class SynonymsAction extends UsageBaseAction {
008
009  private PagingResponse<NameUsage> page;
010  private long offset = 0;
011
012  @Override
013  public String execute() {
014    loadUsage();
015
016    PagingRequest p = new PagingRequest(offset, 25);
017    page = usageService.listSynonyms(id, getLocale(), p);
018
019    return SUCCESS;
020  }
021
022  public void setOffset(long offset) {
023    if (offset >= 0) {
024      this.offset = offset;
025    }
026  }
027
028  public PagingResponse<NameUsage> getPage() {
029    return page;
030  }
031}

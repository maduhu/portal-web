031/**
032 * Class that encapsulates the basic functionality of free text search and paginated navigation.
033 * The class expects a {@link SearchRequest} at creation time and delegates the parsing of search parameters to the
034 * concrete subclass that needs to implement {@link #readFilterParams}.
035 * 
036 * @param <T> the content type of the results
037 * @param <P> the search parameter enum
038 * @param <R> the request type
039 */
040@SuppressWarnings("serial")
041public abstract class BaseSearchAction<T, P extends Enum<?> & SearchParameter, R extends SearchRequest<P>>
042  extends BaseAction {
043
044  private static final Logger LOG = LoggerFactory.getLogger(BaseSearchAction.class);
045  /**
046   * Maximum # of characters shown in a highlighted field.
047   */
048  private static final String abrevMarker = "â€¦";
049  public static final String HL_PRE = "<em class=\"gbifHl\">";
050  public static final String HL_POST = "</em>";
051  public static final Pattern HL_MATCHER = Pattern.compile(HL_PRE + "([^<>]*)" + HL_POST, Pattern.CASE_INSENSITIVE);
052  private static final int HL_MARKER_LENGTH = HL_PRE.length() + HL_POST.length();
053
054  protected final Class<P> searchType;
055  protected final SearchService<T, P, R> searchService;
056  protected final R searchRequest;
057  protected SearchResponse<T, P> searchResponse;
058  protected String q;
059
060  /**
061   * Default constructor
062   */
063  public BaseSearchAction(SearchService<T, P, R> searchService, Class<P> searchType, R searchRequest) {
064    this.searchService = searchService;
065    this.searchType = searchType;
066    this.searchRequest = searchRequest;
067  }

069  /**
070   * Takes a highlighted text and trimmed it to show the first highlighted term.
071   * The text is found using the HL_PRE and HL_POST tags.
072   * Ensure that at least the whole term is shown or else MAX_LONG_HL_FIELD are displayed.
073   * 
074   * @param text highlighted text to be trimmed.
075   * @param maxLength maximum length of resulting string, ignoring the highlighting tags
076   * @return a trimmed version of the highlighted text
077   */
078  public static String getHighlightedText(String text, final int maxLength) {
079    final int firstHlBeginTag = text.indexOf(HL_PRE);
080    final int firstHlEndTag = text.indexOf(HL_POST) + HL_POST.length();
081    final int hlTextSize = firstHlEndTag - firstHlBeginTag - HL_MARKER_LENGTH;
082    // already smaller
083    if (text.length() <= maxLength) {
084      return text;
085    }
086    // no highlighted text, return first bit
087    if (firstHlBeginTag < 0 || firstHlEndTag < 0) {
088      return abbreviate(text, maxLength);
089    }
090    // highlighted text larger than max length - return it all to keep highlighting tags intact
091    if (hlTextSize >= maxLength) {
092      return text.substring(firstHlBeginTag, firstHlEndTag);
093    }
094    int sizeBefore = (maxLength - hlTextSize) / 3;
095    int start = Math.max(0, firstHlBeginTag - sizeBefore);
096    final String leftLimited = start == 0 ? text : abrevMarker + text.substring(start + (sizeBefore > 0 ? 1 : 0));
097    return limitHighlightedText(leftLimited, maxLength);
098  }
099
100  /**
101   * Used by UI to determine if a text is highlighted.
102   * 
103   * @param text
104   * @return
105   */
106  public static boolean isHighlightedText(String text) {
107    return !Strings.isNullOrEmpty(text) && text.contains(HL_PRE);
108  }

110  /**
111   * Takes a highlighted text and returns the given number of initial characters, not counting html tags
112   * and making sure we always keep html tags intact and closed.
113   * 
114   * @param highlightedText the highlighted text to be abbreviated
115   * @return the beginning of the highlighted text up to a given number of characters
116   */
117  public static String limitHighlightedText(String highlightedText, int max) {
118    StringBuffer sb = new StringBuffer();
119
120    Matcher m = HL_MATCHER.matcher(highlightedText);
121    int matched = 0;
122    boolean appendTail = true;
123    while (m.find()) {
124      m.appendReplacement(sb, "");
125      int currLength = sb.length() - matched * HL_MARKER_LENGTH;
126      int hlLength = m.group(1).length();
127      if (currLength + hlLength < max) {
128        sb.append(m.group());
129        matched++;
130      } else {
131        sb.append(abrevMarker);
132        appendTail = false;
133        break;
134      }
135    }
136    if (appendTail) {
137      m.appendTail(sb);
138    }
139
140    // Case where highlightedText wraps entire input string, and it exceeds the max length.
141    // E.g. highlightedText = "<em class=\"gbifHl\">kyle</em>", and max length is 2.
142    if (sb.toString().equals(abrevMarker) && highlightedText.startsWith(HL_PRE) && highlightedText.endsWith(HL_POST)
143        && ((highlightedText.length() - HL_MARKER_LENGTH) > max)) {
144      String txt = highlightedText.substring(HL_PRE.length(), highlightedText.indexOf(HL_POST));
145      // if we're sure it doesn't contain other highlighting tags,
146      // slide end tag backwards so highlighted text length = max length, then append "...", and closing tag
147      if (!txt.contains(HL_PRE) && !txt.contains(HL_POST)) {
148        return highlightedText.substring(0, max) + abrevMarker + HL_POST;
149      }
150    }
151
152    return abbreviate(sb.toString(), max + matched * HL_MARKER_LENGTH);
153  }
154
155  /**
156   * Takes a highlighted text and removes all tags used for highlighting.
157   * 
158   * @param highlightedText the highlighted text to be cleaned
159   * @return a cleaned plain text version of the highlighted text
160   */
161  public static String removeHighlighting(String highlightedText) {
162    return highlightedText.replaceAll(HL_PRE, "").replaceAll(HL_POST, "");
163  }
164
165  private static String abbreviate(String str, int maxWidth) {
166    if (str == null) {
167      return null;
168    }
169    if (maxWidth < 2) {
170      throw new IllegalArgumentException("Minimum abbreviation width is 2");
171    }
172    if (str.length() <= maxWidth) {
173      return str;
174    }
175    return str.substring(0, maxWidth - 1) + abrevMarker;
176  }

178  @Override
179  public String execute() {
180    LOG.info("Search for [{}]", getQ());
181    // default query parameters
182    searchRequest.setQ(getQ());
183    // Turn off highlighting for empty query strings
184    searchRequest.setHighlight(!Strings.isNullOrEmpty(q));
185    // adds parameters processed by subclasses
186    readFilterParams();
187    // issues the search operation
188    searchResponse = searchService.search(searchRequest);
189    LOG.debug("Search for [{}] returned {} results", getQ(), searchResponse == null ? "null" : searchResponse.getCount());
190    return SUCCESS;
191  }

193  /**
194   * The input search pattern used to issue a search operation.
195   * 
196   * @return the q, input search pattern defaulting to "" if none is provided
197   */
198  public String getQ() {
199    // To enable simple linking without a query as per http://dev.gbif.org/issues/browse/POR-274
200    return (q == null) ? "" : q;
201  }

237  /**
238   * Checks if a parameter value is already selected in the current request filters.
239   * Public method used by html templates.
240   * 
241   * @param param the facet name according to
242   */
243  public boolean isInFilter(P param, String value) {
244    if (param != null && searchRequest.getParameters().containsKey(param)) {
245      for (String v : searchRequest.getParameters().get(param)) {
246        if (v.equalsIgnoreCase(value)) {
247          return true;
248        }
249      }
250    }
251    return false;
252  }
253
254  /**
255   * Checks if a parameter value is already selected in the current request filters.
256   * Public method used by html templates.
257   * 
258   * @param paramName the name according to
259   */
260  public boolean isInFilter(String paramName, String value) {
261    return isInFilter(getSearchParam(paramName), value);
262  }
263
264  /**
265   * Implement this method to parse the filter parameters from the request and populate the request parameters.
266   * Make sure that empty parameters are set too to filter null values!
267   */
268  public void readFilterParams() {
269    final Map<String, String[]> params = request.getParameterMap();
270
271    for (String p : params.keySet()) {
272      // recognize facets by enum name
273      P param = getSearchParam(p);
274      if (param != null) {
275        // filter found
276        for (String v : params.get(p)) {
277          searchRequest.addParameter(param, translateFilterValue(param, v));
278        }
279      }
280    }
281  }

298  /**
299   * Optional hook for concrete search actions to define custom translations of filter values
300   * before they are send to the search service.
301   * For example to enable a simple checklist=nub filter without the need to know the real nub UUID.
302   * The values will NOT be translated for the UI and request parameters, only for the search and title lookup service!
303   * This method can be overriden to modify the returned value, by default it keeps it as it is.
304   *
305   * @param param the filter parameter the value belongs to
306   * @param value the value to translate or return as is
307   */
308  protected String translateFilterValue(P param, String value) {
309    // dont do anything by default
310    return value;
311  }

137       loadDataset(c.getDatasetKey());
138     }
139     for (NameUsageComponent c : usage.getVernacularNames()) {
140       loadDataset(c.getDatasetKey());
141     }
142     for (UUID uuid : sublist(Lists.newArrayList(occurrenceDatasetCounts.keySet()), MAX_COMPONENTS)) {
143       loadDataset(uuid);
144     }

194   /**
195    * Performs service calls fleshing out the usage model object.
196    */
197   private void loadUsageDetails() {
198     if (usage.getBasionymKey() != null) {
199       basionym = usageService.get(usage.getBasionymKey(), getLocale());
200     }
201     if (usage.getNubKey() != null) {
202       List<NameUsage> relatedResponse = usageService.listRelated(usage.getNubKey(), getLocale());
203       for (NameUsage u : relatedResponse) {
204         // skip oneself
205         if (!u.getKey().equals(usage.getKey())) {
206           related.add(u);
207         }
208       }
209       occurrenceDatasetCounts = occurrenceDatasetService.occurrenceDatasetsForNubKey(usage.getNubKey());
210     }
211     usage.setSynonyms(usageService.listSynonyms(id, getLocale(), page6).getResults());
212     usage.setVernacularNames(vernacularNameService.listByUsage(id, page100).getResults());
213     usage.setDistributions(distributionService.listByUsage(id, page10).getResults());
214     usage.setImages(imageService.listByUsage(id, page1).getResults()); // first only
215     usage.setTypeSpecimens(typeSpecimenService.listByUsage(id, page10).getResults());
216     TypesAction.removeInvalidTypes(usage.getTypeSpecimens());
217     usage.setSpeciesProfiles(speciesProfileService.listByUsage(id, page20).getResults());
218     for (Description d : descriptionService.listByUsage(id, page50).getResults()) {
219       descriptionToc.addDescription(d);
220     }
221 
222     final Set<String> seenRefs = Sets.newHashSet();
223     usage.setReferences(FluentIterable.from(referenceService.listByUsage(id, page15).getResults())
224       .filter(new Predicate<Reference>() {
225         public boolean apply(@Nullable Reference r) {
226           if (r == null) return false;
227           // deduplicate references only for nub species
228           if (usage.isNub() && seenRefs.contains(r.getCitation())) {
229             return false;
230           }
231           seenRefs.add(r.getCitation());
232           return true;
233         }
234       })
235       .toSortedList(Ordering.natural().onResultOf(new Function<Reference, String>() {
236         @Override
237         public String apply(@Nullable Reference r) {
238           return r == null ? null : r.getCitation();
239         }
240       }))
241     );
242 
243     if (Origin.SOURCE == usage.getOrigin()) {
244       verbatimExists = usageService.getVerbatim(id) != null;
245     }
246 
247     if (usage.isNub() && usage.getSourceKey() != null) {
248       // check if the source record actually exists
249       nubSourceExists = usageService.get(usage.getSourceKey(), null) != null;
250     }
251   }

253   /**
254    * Populates the habitats first using the boolean flags associated with the usage, and then appending any others.
255    */
256   private void populateHabitats() {
257     appendHabitat(usage.isTerrestrial(), "enum.habitat.terrestrial");
258     appendHabitat(usage.isMarine(), "enum.habitat.marine");
259     appendHabitat(usage.isFreshwater(), "enum.habitat.freshwater");
260     for (String h : usage.getHabitats()) {
261       habitats.add(h);
262     }

265   /**
266    * Populates the index of vernacular names.
267    * This is a sorted map keyed on the name + language(optional) and order by name and by the rules imposed by
268    * {@link #vernacularLocaleComparator}
269    */
270   @VisibleForTesting
271   void populateVernacularNames() {
272     List<VernacularName> source = usage.getVernacularNames();
273     vernacularNames.clear();
274     // total ordering of the names, which is then respected in the sorted map
275     for (VernacularName name : Ordering.from(vernacularLocaleComparator).immutableSortedCopy(source)) {
276 
277       // skip those that can't be displayed
278       if (Strings.isNullOrEmpty(name.getVernacularName())) {
279         continue;
280       }
281 
282       // The names are keyed using name||language where language might be null
283       String languageCode = (name.getLanguage() == null) ? null : name.getLanguage().getIso2LetterCode();
284       String key = VERNACULAR_JOINER
285         .join(
286           name.getVernacularName().toLowerCase(),
287           "||",
288           languageCode);
289 
290       // Add to our map index
291       List<VernacularName> values = vernacularNames.get(key);
292       if (values == null) {
293         values = Lists.newArrayList();
294         vernacularNames.put(key, values);
295       }
296       values.add(name);
297     }

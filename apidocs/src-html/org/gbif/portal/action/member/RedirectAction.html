001package org.gbif.portal.action.member;
002
003import org.gbif.api.model.registry.NetworkEntity;
004import org.gbif.api.service.registry.DatasetService;
005import org.gbif.api.service.registry.InstallationService;
006import org.gbif.api.service.registry.NetworkService;
007import org.gbif.api.service.registry.NodeService;
008import org.gbif.api.service.registry.OrganizationService;
009import org.gbif.portal.action.BaseAction;
010import org.gbif.portal.exception.NotFoundException;
011
012import java.util.UUID;
013
014import com.google.inject.Inject;
015
016/**
017 * Redirects to the typed url for the member or throws NotFoundException.
018 */
019public class RedirectAction extends BaseAction {
020
021  @Inject
022  private OrganizationService organizationService;
023  @Inject
024  private NodeService nodeService;
025  @Inject
026  private NetworkService networkService;
027  @Inject
028  private InstallationService technicalInstallationService;
029  @Inject
030  private DatasetService datasetService;
031
032  private UUID id;
033  private String redirectUrl;
034
035  @Override
036  public String execute() {
037    if (id != null) {
038      NetworkEntity member = organizationService.get(id);
039      if (member != null) {
040        return redirect(MemberType.PUBLISHER);
041      }
042
043      member = nodeService.get(id);
044      if (member != null) {
045        return redirect(MemberType.NODE);
046      }
047
048      member = networkService.get(id);
049      if (member != null) {
050        return redirect(MemberType.NETWORK);
051      }
052
053      member = technicalInstallationService.get(id);
054      if (member != null) {
055        return redirect(MemberType.INSTALLATION);
056      }
057
058      member = datasetService.get(id);
059      if (member != null) {
060        return redirect(MemberType.DATASET);
061      }
062    }
063    throw new NotFoundException();
064  }
065
066  public String getRedirectUrl() {
067    return redirectUrl;
068  }
069
070  public void setId(String id) {
071    try {
072      this.id = UUID.fromString(id);
073    } catch (Exception e) {
074      this.id = null;
075    }
076  }
077
078  private String redirect(MemberType type) {
079    redirectUrl = getBaseUrl() + "/" + type.name().toLowerCase() + "/" + id.toString();
080    return SUCCESS;
081  }
082}

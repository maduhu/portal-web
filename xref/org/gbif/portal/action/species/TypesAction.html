30    /**
31     * Iterates over the types in a usage and remove the ones which have bad content - mainly from wikipedia.
32     * See http://dev.gbif.org/issues/browse/POR-409
33     */
34    public static void removeInvalidTypes(List<TypeSpecimen> types){
35      Iterator<TypeSpecimen> iter = types.iterator();
36      while (iter.hasNext()) {
37        TypeSpecimen ts = iter.next();
38        if (ts == null || (TYPE_NAME_STATUS.contains(ts.getTypeStatus()) && Strings.isNullOrEmpty(ts.getScientificName()))) {
39          iter.remove();
40        }
41      }
42    }

31  /**
32   * Provides the basic structure and functionality for: free text search, paginated navigation and faceting navigation.
33   * Besides the inherited functionality from {@link BaseSearchAction} this class provides several features:
34   * - Execution of the search request using an instance of {@link SearchService}.
35   * - Holds the user selected values of a facet.
36   * - Provides the required information for displaying the facet counts.
37   * 
38   * @param <T> type of the results content
39   * @param <P> the search parameter enum
40   * @param <R> the request type
41   */

72    /**
73     * Adds facets to the search before it gets executed and initializes the facets to be displayed in the user interface.
74     */
75    @Override
76    public String execute() {
77      searchRequest.setMultiSelectFacets(true);
78      // add all available facets to the request
79      for (P fEnum : searchType.getEnumConstants()) {
80        searchRequest.addFacets(fEnum);
81      }
82      // execute search
83      final String result = super.execute();
84      // initializes the elements required by the UI
85      initializeFacetsForUI();
86      initSelectedFacetCounts();
87      initMinCounts();
88      return result;
89    }

91    /**
92     * Translates current url query parameter values via the translateFacetValue method.
93     * 
94     * @return current url with translated values
95     */
96    @Override
97    public String getCurrentUrl() {
98      StringBuffer currentUrl = request.getRequestURL();
99      if (request.getQueryString() != null) {
100       boolean first = true;
101       for (String p : querySplitter.split(request.getQueryString())) {
102         Iterator<String> kvIter = paramSplitter.split(p).iterator();
103         // lowercase to handle URL hacking gracefully (http://dev.gbif.org/issues/browse/POR-522)
104         String key = kvIter.next().toLowerCase();
105         if (first) {
106           currentUrl.append("?");
107         } else {
108           currentUrl.append("&");
109         }
110         if (kvIter.hasNext()) { // parameter could have no value
111           String val = kvIter.next();
112           // potentially translate facet values
113           P facet = getSearchParam(key);
114           if (facet != null) {
115             val = translateFilterValue(facet, val);
116           }
117           currentUrl.append(key);
118           currentUrl.append("=");
119           currentUrl.append(val);
120         } else {
121           currentUrl.append(key);
122         }
123         first = false;
124       }
125     }
126     return currentUrl.toString();
127   }

174   /**
175    * Utility function that sets facet titles.
176    * The function uses a function parameter to accomplish this task.
177    * The getTitleFunction could provide the actual communication with the service later to provide the required title.
178    * 
179    * @param facet the facet
180    * @param getTitleFunction function that returns title using a facet name
181    */
182   protected void lookupFacetTitles(P facet, Function<String, String> getTitleFunction) {
183     // "cache"
184     Map<String, String> names = Maps.newHashMap();
185     // facet counts
186     if (facetCounts.containsKey(facet)) {
187       for (int idx = 0; idx < facetCounts.get(facet).size(); idx++) {
188         FacetInstance c = getFacetCounts().get(facet).get(idx);
189         if (names.containsKey(c.getName())) {
190           c.setTitle(names.get(c.getName()));
191         } else {
192           try {
193             c.setTitle(getTitleFunction.apply(c.getName()));
194             names.put(c.getName(), c.getTitle());
195           } catch (Exception e) {
196             LOG.warn("Cannot lookup {} title for {}", new Object[] {facet.name(), c.getName(), e});
197           }
198         }
199       }
200     }
201   }

203   /**
204    * Searches for facetInstance.name in the list of FacetInstances.
205    * 
206    * @param facet search parameter to find the value in
207    * @param name facet name to find, case insensitive
208    * @return the existing facet instance from the counts or null
209    */
210   private FacetInstance findFacetInstanceByName(P facet, String name) {
211     List<FacetInstance> fis = facetCounts.get(facet);
212     if (fis != null) {
213       for (FacetInstance fi : fis) {
214         if (fi.getName() != null && fi.getName().equalsIgnoreCase(name)) {
215           return fi;
216         }
217       }
218     }
219     return null;
220   }

222   /**
223    * By using the response object, this method initializes the facetCounts field.
224    * If the response object contains facets, iterates over the facets for copying the count information into the
225    * facetCounts (see {@link Facet.Count}) field.
226    */
227   private void initializeFacetsForUI() {
228     if (searchResponse!=null && searchResponse.getFacets() != null && !searchResponse.getFacets().isEmpty()) {
229       // there are facets in the response
230       for (Facet<P> facet : searchResponse.getFacets()) {
231         if (facet.getCounts() != null) {// the facet.Count are stored in the facetCounts field
232           // facetFilters.put(facet.getField(), toFacetInstance(facet.getCounts()));
233           facetCounts.put(facet.getField(), toFacetInstance(facet.getCounts()));
234         }
235       }
236     }
237   }
238 
239   private void initMinCounts() {
240     if (searchResponse!=null && !searchRequest.getParameters().isEmpty()) {
241       // calculate the minimum count for each facet
242       for (Facet<P> facet : searchResponse.getFacets()) {
243         Long min = null;
244         for (Facet.Count cnt : facet.getCounts()) {
245           if (cnt.getCount() != null && (min == null || cnt.getCount() < min)) {
246             min = cnt.getCount();
247           }
248         }
249         facetMinimumCount.put(facet.getField(), min);
250       }
251     }
252   }
253 
254   private void initSelectedFacetCounts() {
255     if (!searchRequest.getParameters().isEmpty()) {
256       // convert request filters into facets instances
257       for (P facet : searchRequest.getParameters().keySet()) {
258         selectedFacetCounts.put(facet, Lists.<FacetInstance>newArrayList());
259         for (String filterValue : searchRequest.getParameters().get(facet)) {
260           if (StringUtils.trimToNull(filterValue) != null) {
261             FacetInstance facetInstance = findFacetInstanceByName(facet, filterValue);
262             if (facetInstance == null) {
263               facetInstance = new FacetInstance(filterValue);
264               facetInstance.setCount(null);
265               // also add them to the list of all counts so we will lookup titles
266               if (!facetCounts.containsKey(facet)) {
267                 facetCounts.put(facet, Lists.<FacetInstance>newArrayList());
268               }
269               facetCounts.get(facet).add(facetInstance);
270             }
271             selectedFacetCounts.get(facet).add(facetInstance);
272           }
273         }
274       }
275     }
276   }

278   private List<FacetInstance> toFacetInstance(List<Facet.Count> counts) {
279     List<FacetInstance> instances = Lists.newArrayList();
280     for (Facet.Count c : counts) {
281       // only show counts with at least 1 matching record
282       if (c.getCount() > 0) {
283         instances.add(new FacetInstance(c));
284       }
285     }
286     return instances;
287   }

001package org.gbif.portal.action.species;
002
003import org.gbif.api.model.checklistbank.TypeSpecimen;
004import org.gbif.api.service.checklistbank.TypeSpecimenService;
005import org.gbif.api.vocabulary.TypeStatus;
006
007import java.util.Iterator;
008import java.util.List;
009import java.util.Set;
010
011import com.google.common.base.Strings;
012import com.google.common.collect.Sets;
013import com.google.inject.Inject;
014
015
016public class TypesAction extends SeeMoreAction<TypeSpecimen> {
017  private static final Set<TypeStatus> TYPE_NAME_STATUS = Sets.newHashSet(TypeStatus.TYPE_GENUS, TypeStatus.TYPE_SPECIES);
018  @Inject
019  public TypesAction(TypeSpecimenService service) {
020    super(service);
021  }
022
023  @Override
024  public String execute() {
025    super.execute();
026    TypesAction.removeInvalidTypes(getPage().getResults());
027    return SUCCESS;
028  }
029
030  /**
031   * Iterates over the types in a usage and remove the ones which have bad content - mainly from wikipedia.
032   * See http://dev.gbif.org/issues/browse/POR-409
033   */
034  public static void removeInvalidTypes(List<TypeSpecimen> types){
035    Iterator<TypeSpecimen> iter = types.iterator();
036    while (iter.hasNext()) {
037      TypeSpecimen ts = iter.next();
038      if (ts == null || (TYPE_NAME_STATUS.contains(ts.getTypeStatus()) && Strings.isNullOrEmpty(ts.getScientificName()))) {
039        iter.remove();
040      }
041    }
042  }
043}

057  /**
058   * @param maxSize
059   * @param <T>
060   * @return list of a size not larger then maxSize
061   */
062  protected static <T> List<T> sublist(List<T> list, int maxSize) {
063    return sublist(list, 0, maxSize);
064  }
065
066  /**
067   * @param maxSize
068   * @param <T>
069   * @return list of a size not larger then maxSize
070   */
071  protected static <T> List<T> sublist(List<T> list, int start, int maxSize) {
072    if (list.size() <= maxSize) {
073      if (start == 0) {
074        return list;
075      }
076      return list.subList(start, list.size());
077    }
078    return list.subList(start, maxSize);
079  }

124  /**
125   * Loads a name usage and its checklist by the id parameter.
126   * 
127   * @throws NotFoundException if no usage for the given id can be found
128   */
129  public void loadUsage() {
130    if (id == null) {
131      throw new NotFoundException("No checklist usage id given");
132    }
133    NameUsage u = usageService.get(id, getLocale());
134    if (u == null) {
135      throw new NotFoundException("No usage found with id " + id);
136    }
137    usage = new NameUsageContainer(u);
138    usageMetrics = usageService.getMetrics(id);
139    // make sure we got an empty one at least - its all ints and its references from lots of templates
140    if (usageMetrics == null) {
141      usageMetrics = new NameUsageMetrics();
142      usageMetrics.setKey(id);
143    }
144    // load checklist
145    dataset = datasetService.get(usage.getDatasetKey());
146    if (dataset == null) {
147      throw new ReferentialIntegrityException(NameUsage.class, id, "Missing dataset " + usage.getDatasetKey());
148    }
149
150    // load constituent dataset too if we have one
151    if (usage.getConstituentKey() != null) {
152      constituent = datasetService.get(usage.getConstituentKey());
153    }
154
155    // load publisher
156    publisher = organizationService.get(dataset.getOwningOrganizationKey());
157
158    try {
159      metrics = metricsService.get(usage.getDatasetKey());
160    } catch (ServiceUnavailableException e) {
161      LOG.error("Failed to load checklist metrics for dataset {}", usage.getDatasetKey());
162    }
163
164    try {
165      numOccurrences = occurrenceCubeService.get(new ReadBuilder().at(OccurrenceCube.NUB_KEY, usage.getKey()));
166      numGeoreferencedOccurrences =
167        occurrenceCubeService.get(new ReadBuilder().at(OccurrenceCube.NUB_KEY, usage.getKey()).at(
168          OccurrenceCube.IS_GEOREFERENCED, true));
169    } catch (ServiceUnavailableException e) {
170      LOG.error("Failed to load occurrence metrics for usage {}", usage.getKey(), e);
171    }
172  }

174  public void setId(Integer id) {

178  /**
179   * Populates the checklists map with the checklists for the given keys.
180   * The method does not remove existing entries and can be called many times to add additional, new checklists.
181   */
182  protected void loadChecklists(Collection<UUID> checklistKeys) {
183    for (UUID u : checklistKeys) {
184      loadDataset(u);
185    }
186  }
187
188  protected void loadDataset(UUID datasetKey) {
189    if (datasetKey != null && !datasets.containsKey(datasetKey)) {
190      try {
191        datasets.put(datasetKey, datasetService.get(datasetKey));
192      } catch (ServiceUnavailableException e) {
193        LOG.error("Failed to load dataset with key {}", datasetKey);
194        datasets.put(datasetKey, null);
195      }
196    }
197  }

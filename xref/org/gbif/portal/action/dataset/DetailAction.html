47    private static final List<Kingdom> KINGDOMS = ImmutableList.of(Kingdom.ANIMALIA, Kingdom.ARCHAEA, Kingdom.BACTERIA,

69    private static final int PAGE_SIZE = 10;

71    @Inject
72    public DetailAction(DatasetService datasetService, CubeService occurrenceCubeService,
73      DatasetMetricsService metricsService, OrganizationService organizationService) {
74      super(datasetService, occurrenceCubeService, metricsService, organizationService);
75    }
76  
77    @Override
78    public String execute() {
79      loadDetail();
80      populateContacts(member.getContacts());
81      populateLinks(member.getEndpoints());
82      // only datasets with a key (internal) can have constituents
83      if (id != null) {
84        constituents = datasetService.listConstituents(id, new PagingRequest(0, PAGE_SIZE));
85        super.loadCountWrappedDatasets(constituents);
86      }
87  
88      // the map article is rendered only if the cube has indicated georeferenced records, or if there are
89      // coverages that are not global (they don't really warrant visualizing).
90      renderMaps = getNumGeoreferencedOccurrences() != null && getNumGeoreferencedOccurrences() > 0;
91      if (!renderMaps) {
92        for (GeospatialCoverage gc : member.getGeographicCoverages()) {
93          renderMaps = renderMaps
94            || (gc.getBoundingBox() != null && !gc.getBoundingBox().isGlobalCoverage());
95          if (renderMaps) {
96            break; // for speed
97          }
98        }
99      }
100     // get networks
101     networks = datasetService.listNetworks(id);
102 
103     return SUCCESS;
104   }

177   /**
178    * Populates the preferred and other contact lists from the unified contact list.
179    */
180   private void populateContacts(@NotNull List<Contact> contacts) {
181     Preconditions.checkNotNull(contacts, "Contacts cannot be null - expected empty list for no contacts");
182     preferredContacts.clear(); // for safety
183     otherContacts.clear();
184     for (Contact contact : contacts) {
185       if (contact.isPrimary()) {
186         preferredContacts.add(contact);
187       } else {
188         otherContacts.add(contact);
189       }
190     }
191   }
192 
193   /**
194    * Populates the data, metadata and other links from the unified endpoint list.
195    */
196   private void populateLinks(@NotNull List<Endpoint> endpoints) {
197     Preconditions.checkNotNull(endpoints, "Enpoints cannot be null - expected empty list for no endpoints");
198     dataLinks.clear(); // for safety
199     metaLinks.clear();
200     links.clear();
201     for (Endpoint p : member.getEndpoints()) {
202       if (DATA_CODES.contains(p.getType())) {
203         dataLinks.add(p);
204       } else if (METADATA_CODES.contains(p.getType())) {
205         metaLinks.add(p);
206       } else {
207         links.add(p);
208       }
209     }
210   }

044  @Test
045  public void testBuild() {
046    PredicateFactory pf = new PredicateFactory();
047    Map<String, String[]> params = new HashMap<String, String[]>();
048    params.put("BASIS_OF_RECORD", new String[] {"PRESERVED_SPECIMEN"});
049    Predicate p = pf.build(params);
050
051    assertTrue(p instanceof EqualsPredicate);
052    EqualsPredicate eq = (EqualsPredicate) p;
053    assertEquals(OccurrenceSearchParameter.BASIS_OF_RECORD, eq.getKey());
054    assertEquals("PRESERVED_SPECIMEN", eq.getValue());
055
056    // add a second country to check an OR is built
057    params.put("BASIS_OF_RECORD", new String[] {"OBSERVATION", "PRESERVED_SPECIMEN"});
058    p = pf.build(params);
059    assertTrue(p instanceof DisjunctionPredicate);
060    DisjunctionPredicate dq = (DisjunctionPredicate) p;
061    assertEquals(2, dq.getPredicates().size());
062    Iterator<Predicate> iter = dq.getPredicates().iterator();
063    p = iter.next();
064    assertTrue(p instanceof EqualsPredicate);
065    eq = (EqualsPredicate) p;
066    assertEquals(OccurrenceSearchParameter.BASIS_OF_RECORD, eq.getKey());
067    assertEquals("OBSERVATION", eq.getValue());
068    p = iter.next();
069    assertTrue(p instanceof EqualsPredicate);
070    eq = (EqualsPredicate) p;
071    assertEquals(OccurrenceSearchParameter.BASIS_OF_RECORD, eq.getKey());
072    assertEquals("PRESERVED_SPECIMEN", eq.getValue());
073
074    // add a third to check an AND is built
075    params.put("LATITUDE", new String[] {"10.07, *"}); // greater than or equals
076    p = pf.build(params);
077    assertTrue(p instanceof ConjunctionPredicate);
078    ConjunctionPredicate cq = (ConjunctionPredicate) p;
079    assertEquals(2, cq.getPredicates().size());
080    iter = cq.getPredicates().iterator();
081    p = iter.next();
082    assertTrue(p instanceof DisjunctionPredicate);
083    p = iter.next();
084    assertTrue(p instanceof GreaterThanOrEqualsPredicate); // the OR'ed scientific names (tested above)
085    GreaterThanOrEqualsPredicate gp = (GreaterThanOrEqualsPredicate) p;
086    assertEquals(OccurrenceSearchParameter.LATITUDE, gp.getKey());
087    assertEquals("10.07", gp.getValue());
088  }

151  @Test
152  public void testRange() {
153    PredicateFactory pf = new PredicateFactory();
154    Map<String, String[]> params = Maps.newHashMap();
155
156    params.put("YEAR", new String[] {"1900", "1800,1840", "*, 1700", "2000 , *"});
157
158    Predicate p = pf.build(params);
159    assertTrue(p instanceof DisjunctionPredicate);
160    DisjunctionPredicate or = (DisjunctionPredicate) p;
161    assertEquals(4, or.getPredicates().size());
162    Iterator<Predicate> iter = or.getPredicates().iterator();
163
164    p = iter.next();
165    assertSimplePredicate(p, EqualsPredicate.class, "1900", OccurrenceSearchParameter.YEAR);
166
167    p = iter.next();
168    assertTrue(p instanceof ConjunctionPredicate);
169    ConjunctionPredicate and = (ConjunctionPredicate) p;
170    assertEquals(2, and.getPredicates().size());
171    Iterator<Predicate> andIter = and.getPredicates().iterator();
172    assertSimplePredicate(andIter.next(), GreaterThanOrEqualsPredicate.class, "1800", OccurrenceSearchParameter.YEAR);
173    assertSimplePredicate(andIter.next(), LessThanOrEqualsPredicate.class, "1840", OccurrenceSearchParameter.YEAR);
174
175    p = iter.next();
176    assertSimplePredicate(p, LessThanOrEqualsPredicate.class, "1700", OccurrenceSearchParameter.YEAR);
177
178    p = iter.next();
179    assertSimplePredicate(p, GreaterThanOrEqualsPredicate.class, "2000", OccurrenceSearchParameter.YEAR);
180  }

182  private void assertSimplePredicate(Predicate p, Class<? extends SimplePredicate> expectedClass, String expectedValue,
183    OccurrenceSearchParameter expectedParam) {
184    assertEquals(expectedClass, p.getClass());
185    SimplePredicate sp = (SimplePredicate) p;
186    assertEquals(expectedParam, sp.getKey());
187    assertEquals(expectedValue, sp.getValue());
188  }

190  private String randomValue(OccurrenceSearchParameter p) {
191    if (OccurrenceSearchParameter.COUNTRY == p) {
192      return Country.SPAIN.getIso2LetterCode();
193
194    } else if (OccurrenceSearchParameter.GEOMETRY == p) {
195      return "30.12 10, 10 20, 20 40, 40 40, 30.12 10";
196
197    } else if (UUID.class.isAssignableFrom(p.type())) {
198      return UUID.randomUUID().toString();
199
200    } else if (Boolean.class.isAssignableFrom(p.type())) {
201      return "true";
202
203    } else if (OccurrenceSearchParameter.MONTH == p) {
204      return String.format("%02d", 1 + rnd.nextInt(11));
205
206    } else if (OccurrenceSearchParameter.LONGITUDE == p) {
207      return Integer.toString(rnd.nextInt(180));
208
209    } else if (OccurrenceSearchParameter.LATITUDE == p) {
210      return Integer.toString(rnd.nextInt(90));
211
212    } else if (OccurrenceSearchParameter.COUNTRY == p || OccurrenceSearchParameter.PUBLISHING_COUNTRY == p) {
213      return Country.AFGHANISTAN.getIso2LetterCode();
214
215    } else if (Enum.class.isAssignableFrom(p.type())) {
216      Class<? extends Enum<?>> vocab = (Class<? extends Enum<?>>) p.type();
217      return vocab.getEnumConstants()[0].name();
218    }
219
220    return String.format("%04d", 1 + rnd.nextInt(2012));
221  }

001package org.gbif.portal.action.installation;
002
003import org.gbif.api.model.common.paging.PagingRequest;
004import org.gbif.api.model.common.paging.PagingResponse;
005import org.gbif.api.model.registry.Dataset;
006import org.gbif.api.model.registry.Installation;
007import org.gbif.api.model.registry.Organization;
008import org.gbif.api.service.checklistbank.DatasetMetricsService;
009import org.gbif.api.service.metrics.CubeService;
010import org.gbif.api.service.registry.InstallationService;
011import org.gbif.api.service.registry.OrganizationService;
012import org.gbif.portal.action.member.MemberBaseAction;
013import org.gbif.portal.action.member.MemberType;
014
015import com.google.inject.Inject;
016
017public class DetailAction extends MemberBaseAction<Installation> {
018
019  private final OrganizationService organizationService;
020  private final InstallationService installationService;
021
022  private Organization organization;
023  private PagingResponse<Dataset> page;
024  private long offset = 0;
025
026  @Inject
027  public DetailAction(OrganizationService organizationService, InstallationService installationService,
028    CubeService cubeService, DatasetMetricsService datasetMetricsService) {
029    super(MemberType.INSTALLATION, installationService, cubeService, datasetMetricsService, organizationService);
030    this.organizationService = organizationService;
031    this.installationService = installationService;
032  }
033
034  public String datasets() throws Exception {
035    super.execute();
036    page = installationService.getHostedDatasets(id, new PagingRequest(offset, 25));
037    return SUCCESS;
038  }
039
040  @Override
041  public String execute() throws Exception {
042    super.execute();
043    // load hosting organization
044    if (member.getOrganizationKey() != null) {
045      organization = organizationService.get(member.getOrganizationKey());
046    }
047    // load first 10 datasets
048    page = installationService.getHostedDatasets(id, new PagingRequest(0, 10));
049    super.loadCountWrappedDatasets(page);
050
051    return SUCCESS;
052  }
053
054  public Organization getOrganization() {
055    return organization;
056  }
057
058  public PagingResponse<Dataset> getPage() {
059    return page;
060  }
061
062  public void setOffset(long offset) {
063    if (offset >= 0) {
064      this.offset = offset;
065    }
066  }
067}

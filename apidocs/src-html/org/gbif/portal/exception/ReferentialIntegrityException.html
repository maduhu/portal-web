001package org.gbif.portal.exception;
002
003/**
004 * Exception to be thrown if a primary resource can not load all of its related, secondary components.
005 * For example if a dataset has just been deleted but a name usage still exists, this exception should be thrown
006 * when trying to load the dataset.
007 */
008public class ReferentialIntegrityException extends RuntimeException {
009  private final Class primaryResourceClass;
010  private final Object primaryResourceKey;
011
012  private static String buildMessage(Class primaryResourceClass, Object primaryResourceKey) {
013
014    return primaryResourceClass.getSimpleName() + " ["+primaryResourceKey.toString() + "] "
015           + "has referential integrity problems.";
016  }
017
018  /**
019   * @param primaryResourceClass
020   * @param primaryResourceKey
021   * @param msg additional exception message
022   */
023  public ReferentialIntegrityException(Class primaryResourceClass, Object primaryResourceKey, String msg) {
024    super(buildMessage(primaryResourceClass, primaryResourceKey) + ". " + msg);
025    this.primaryResourceClass = primaryResourceClass;
026    this.primaryResourceKey = primaryResourceKey;
027  }
028
029  /**
030   * @param primaryResourceClass
031   * @param primaryResourceKey
032   */
033  public ReferentialIntegrityException(Class primaryResourceClass, Object primaryResourceKey) {
034    super(buildMessage(primaryResourceClass, primaryResourceKey));
035    this.primaryResourceClass = primaryResourceClass;
036    this.primaryResourceKey = primaryResourceKey;
037  }
038
039
040  public Class getPrimaryResourceClass() {
041    return primaryResourceClass;
042  }
043
044  public Object getPrimaryResourceKey() {
045    return primaryResourceKey;
046  }
047}

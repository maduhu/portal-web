153  @Override
154  protected void loadDetail() {
155    if (id == null) {
156      throw new NotFoundException("No identifier provided to load the dataset");
157    }
158
159    member = datasetService.get(id);
160    if (member == null) {
161      throw new NotFoundException("No dataset found with id " + id);
162    }
163
164    parentDataset = member.getParentDatasetKey() != null ? datasetService.get(member.getParentDatasetKey()) : null;
165
166    publisher = member.getOwningOrganizationKey() != null ?
167      organizationService.get(member.getOwningOrganizationKey()) : publisher;
168
169    installation = installationService.get(member.getInstallationKey());
170
171    host = organizationService.get(installation.getOrganizationKey());
172
173    organizedCoverages = member.getTaxonomicCoverages() != null ?
174      constructOrganizedTaxonomicCoverages(member.getTaxonomicCoverages()) : organizedCoverages;
175
176    metrics = metricsService.get(id);
177    if (metrics == null) {
178      metrics = new DatasetMetrics();
179    }
180
181    populateOccurrenceCounts(); // only when a key exists
182  }

184  /**
185   * Takes a list of the resource's TaxonomicCoverages, and for each one, creates a new OrganizedTaxonomicCoverage
186   * that gets added to this class' list of OrganizedTaxonomicCoverage.
187   * 
188   * @param coverages list of resource's OrganizedTaxonomicCoverage
189   */
190  @VisibleForTesting
191  List<OrganizedTaxonomicCoverages> constructOrganizedTaxonomicCoverages(List<TaxonomicCoverages> coverages) {
192    List<OrganizedTaxonomicCoverages> organizedCoverages = new ArrayList<OrganizedTaxonomicCoverages>();
193    for (TaxonomicCoverages coverage : coverages) {
194      OrganizedTaxonomicCoverages organizedCoverage = new OrganizedTaxonomicCoverages();
195      organizedCoverage.setDescription(coverage.getDescription());
196      organizedCoverage.setCoverages(organizeTaxonomicCoverages(coverage.getCoverages()));
197      organizedCoverages.add(organizedCoverage);
198    }
199    return organizedCoverages;
200  }

202  /**
203   * Produces a list of rank names comprising the distinct uppercase values from the union of the {@link Rank} values
204   * and the provided coverages.
205   */
206  @VisibleForTesting
207  List<String> newRankList(List<TaxonomicCoverage> coverages) {
208    LinkedHashSet<String> s = Sets.newLinkedHashSet();
209    for (Rank rank : Rank.values()) {
210      s.add(rank.name().toUpperCase());
211    }
212    for (TaxonomicCoverage cov : coverages) {
213      if (cov.getRank() != null && hasNonWhitespace(cov.getRank().getVerbatim())) {
214        s.add(cov.getRank().getVerbatim().toUpperCase().trim());
215      }
216    }
217    // include "RANK NOT SPECIFIED" used to categorize coverages with no rank specified
218    s.add(EMPTY_RANK);
219    return ImmutableList.<String>copyOf(s);
220  }

222  /**
223   * Construct the display name from TaxonomicCoverage's scientific name and common name properties. It will look like:
224   * scientific name (common name) provided both properties are not null. Otherwise, it will be either the scientific
225   * name or common name by themselves.
226   * 
227   * @return constructed display name or an empty string if none could be constructed
228   */
229  private String createDisplayNameForCoverage(TaxonomicCoverage coverage) {
230    String combined = null;
231    if (coverage != null) {
232
233      String scientificName = hasNonWhitespace(coverage.getScientificName()) ?
234        CharMatcher.WHITESPACE.trimFrom(coverage.getScientificName()) : null;
235
236      String commonName = hasNonWhitespace(coverage.getCommonName()) ?
237        CharMatcher.WHITESPACE.trimFrom(coverage.getCommonName()) : null;
238
239      if (!Strings.isNullOrEmpty(scientificName) && !Strings.isNullOrEmpty(commonName)) {
240        combined = scientificName + " (" + commonName + ")";
241      } else if (scientificName != null) {
242        combined = Strings.emptyToNull(scientificName);
243      } else if (commonName != null) {
244        combined = Strings.emptyToNull(commonName);
245      }
246    }
247    return Strings.nullToEmpty(combined);
248  }

250  /**
251   * This method iterates through a list of TaxonomicCoverage, and groups them all by rank. For each unique rank
252   * represented in the list, there will be 1 OrganizedTaxonomicCoverage, that has a list of
253   * DisplayableTaxonomicCoverage. A DisplayableTaxonomicCoverage is basically the same as TaxonomicCoverage, only that
254   * it has a field called display name. The display name is the way the TaxonomicCoverage should be shown in the UI.
255   * 
256   * @param coverages list of TaxonomicCoverages' TaxonomicCoverage
257   * @return list of OrganizedTaxonomicCoverage (one for each unique rank represented in the list of
258   *         TaxonomicCoverage), or an empty list if none were added
259   */
260  private List<OrganizedTaxonomicCoverage> organizeTaxonomicCoverages(List<TaxonomicCoverage> coverages) {
261    List<OrganizedTaxonomicCoverage> organizedTaxonomicCoveragesList = new ArrayList<OrganizedTaxonomicCoverage>();
262
263    // create Rank name list, made from Rank vocab names + uninterpreted rank names discovered from coverages list
264    List<String> rankNames = newRankList(coverages);
265
266    for (String rankName : rankNames) {
267      // initiate a new OrganizedTaxonomicCoverage for each rank
268      OrganizedTaxonomicCoverage organizedCoverage = new OrganizedTaxonomicCoverage(rankName);
269
270      // iterate through all coverages, and match all with same rank
271      for (TaxonomicCoverage coverage : coverages) {
272        // create display name
273        String displayName = createDisplayNameForCoverage(coverage);
274        // proceed if display name created (meaning coverage has at least a scientific name)
275        if (!Strings.isNullOrEmpty(displayName)) {
276          // Check if the interpreted rank or the verbatim rank matches.
277          // Check for cases where there is no rank specified, only scientific name (grouped under RANK NOT SPECIFIED)
278          Rank interpreted = coverage.getRank() == null ? null : coverage.getRank().getInterpreted();
279          String verbatim = coverage.getRank() == null ? null : coverage.getRank().getVerbatim();
280          if ((interpreted != null && rankName.equalsIgnoreCase(interpreted.name())) || (verbatim != null && rankName
281            .equalsIgnoreCase(verbatim)) || (verbatim == null && interpreted == null && rankName
282            .equalsIgnoreCase(EMPTY_RANK))) {
283            // add DisplayableTaxonomicCoverage into OrganizedTaxonomicCoverage
284            DisplayableTaxonomicCoverage displayable = new DisplayableTaxonomicCoverage(coverage, displayName);
285            organizedCoverage.getDisplayableNames().add(displayable);
286          }
287        }
288      }
289      // add to list if OrganizedTaxonomicCoverage contained at least one DisplayableTaxonomicCoverage
290      if (organizedCoverage.getDisplayableNames().size() > 0) {
291        organizedTaxonomicCoveragesList.add(organizedCoverage);
292      }
293    }
294    // return list
295    return organizedTaxonomicCoveragesList;
296  }

298  /**
299   * Populates the occurrence counts using the cube, only when the key exists.
300   */
301  private void populateOccurrenceCounts() {
302    if (id != null) {
303      try {
304        numOccurrences = occurrenceCubeService.get(new ReadBuilder().at(OccurrenceCube.DATASET_KEY, id));
305        numGeoreferencedOccurrences =
306          occurrenceCubeService.get(new ReadBuilder().at(OccurrenceCube.DATASET_KEY, id).at(
307            OccurrenceCube.IS_GEOREFERENCED, true));
308      } catch (Exception e) {
309        LOG.error("Failed to load occurrence metrics for dataset {}", id, e);
310      }
311    }
312  }

001package org.gbif.portal.action.species;
002
003import org.gbif.api.model.Constants;
004import org.gbif.api.model.checklistbank.DatasetMetrics;
005import org.gbif.api.service.checklistbank.DatasetMetricsService;
006import org.gbif.api.vocabulary.Extension;
007import org.gbif.api.vocabulary.Rank;
008import org.gbif.portal.action.BaseAction;
009
010import java.util.UUID;
011
012import com.google.inject.Inject;
013
014import static org.gbif.api.model.Constants.NUB_DATASET_KEY;
015
016@SuppressWarnings("serial")
017public class HomeAction extends BaseAction {
018
019  public static final UUID COL_KEY = UUID.fromString("7ddf754f-d193-4cc9-b351-99906754a03b");
020
021
022  @Inject
023  private DatasetMetricsService metricsService;
024
025  private DatasetMetrics colMetrics;
026  private DatasetMetrics nubMetrics;
027
028  @Override
029  public String execute() {
030    nubMetrics = metricsService.get(Constants.NUB_DATASET_KEY);
031    colMetrics = metricsService.get(COL_KEY);
032    // make sure we have at least an empty one
033    // this should hardly ever happen, but maybe we just killed all metrics or just start a fresh CLB
034    if (nubMetrics==null) {
035      nubMetrics = new DatasetMetrics();
036    }
037    if (colMetrics==null) {
038      colMetrics = new DatasetMetrics();
039    }
040    return SUCCESS;
041  }
042
043  /**
044   * Exposed to allow easy access in freemarker.
045   */
046  public UUID getNubDatasetKey() {
047    return NUB_DATASET_KEY;
048  }
049
050  public DatasetMetrics getNubMetrics() {
051    return nubMetrics;
052  }
053
054  public int getNubCommonNames() {
055    return nubMetrics.getExtensionRecordCount(Extension.VERNACULAR_NAME);
056  }
057
058  public int getNubInfraSpecies() {
059    return nubMetrics.getCountByRank(Rank.INFRASPECIFIC_NAME);
060  }
061
062  public int getNubLanguages() {
063    return nubMetrics.getCountNamesByLanguage().size();
064  }
065
066  public int getNubSpecies() {
067    return nubMetrics.getCountByRank(Rank.SPECIES);
068  }
069
070  public int getColSpecies() {
071    return colMetrics.getCountByRank(Rank.SPECIES);
072  }
073
074  public DatasetMetrics getColMetrics() {
075    return colMetrics;
076  }
077
078  public static UUID getColKey() {
079    return COL_KEY;
080  }
081
082}

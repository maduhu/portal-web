001package org.gbif.portal.action.country;
002
003import org.gbif.api.service.checklistbank.DatasetMetricsService;
004import org.gbif.api.service.metrics.CubeService;
005import org.gbif.api.service.occurrence.OccurrenceCountryIndexService;
006import org.gbif.api.service.occurrence.OccurrenceDatasetIndexService;
007import org.gbif.api.service.registry.DatasetSearchService;
008import org.gbif.api.service.registry.DatasetService;
009import org.gbif.api.service.registry.NodeService;
010import org.gbif.api.service.registry.OrganizationService;
011import org.gbif.portal.config.DrupalCountryTagMap;
012
013import com.google.inject.Inject;
014
015public class NewsAction extends CountryBaseAction {
016
017  private DrupalCountryTagMap drupalCountryTagMap;
018
019  @Inject
020  public NewsAction(NodeService nodeService, CubeService cubeService,
021    OccurrenceDatasetIndexService datasetIndexService,
022    OccurrenceCountryIndexService countryIndexService, DatasetService datasetService,
023    DatasetSearchService datasetSearchService, DatasetMetricsService datasetMetricsService,
024    OrganizationService organizationService, DrupalCountryTagMap drupalCountryTagMap) {
025    super(nodeService, cubeService, datasetIndexService, countryIndexService, datasetService, datasetSearchService,
026      datasetMetricsService, organizationService);
027
028    this.drupalCountryTagMap = drupalCountryTagMap;
029  }
030
031  public Integer getDrupalTagId() {
032    return drupalCountryTagMap.lookupTag(country);
033  }
034}

001package org.gbif.portal.action.publisher;
002
003import org.gbif.api.model.common.paging.PagingRequest;
004import org.gbif.api.model.common.paging.PagingResponse;
005import org.gbif.api.model.registry.Node;
006import org.gbif.api.model.registry.Organization;
007import org.gbif.api.service.registry.NodeService;
008import org.gbif.api.service.registry.OrganizationService;
009import org.gbif.portal.action.BaseAction;
010
011import java.util.Map;
012import java.util.UUID;
013
014import com.google.common.collect.Maps;
015import com.google.inject.Inject;
016
017/**
018 * This is a simple paging search action that uses registry (v2+) web services to support basic organization listings. 
019 */
020@SuppressWarnings("serial")
021public class SearchAction extends BaseAction {
022
023  private static final int PAGE_SIZE = 20;
024  
025  private final OrganizationService organizationService;
026  private final NodeService nodeService;
027  
028  private long offset = 0;
029  private String q;
030  private PagingResponse<Organization> page;
031  // required to show the titles
032  private Map<UUID,Node> nodeIndex = Maps.newHashMap();
033
034  @Inject
035  public SearchAction(OrganizationService organizationService, NodeService nodeService) {
036    this.organizationService = organizationService;
037    this.nodeService = nodeService;
038  }
039
040  @Override
041  public String execute() throws Exception {
042    // TODO: Support ordering
043    // http://dev.gbif.org/issues/browse/REG-415
044    page = organizationService.search(q, new PagingRequest(offset, PAGE_SIZE));
045    
046    for (Organization o : page.getResults()) {
047      if (!nodeIndex.containsKey(o.getEndorsingNodeKey())) {
048        nodeIndex.put(o.getEndorsingNodeKey(), nodeService.get(o.getEndorsingNodeKey()));  
049      }
050    }
051    
052    LOG.info("Searching for publisher with query[{}] produced {} results", q, page.getCount());
053    return SUCCESS;
054  }
055
056  public PagingResponse<Organization> getPage() {
057    return page;
058  }
059
060  // safe setting to 0 or more
061  public void setOffset(long offset) {
062    this.offset = offset > 0 ? offset : 0; 
063  }
064  
065  public String getQ() {
066    return q;
067  }
068  
069  public void setQ(String q) {
070    this.q = q;
071  }
072  
073  public Map<UUID, Node> getNodeIndex() {
074    return nodeIndex;
075  }
076}

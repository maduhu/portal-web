062  /**
063   * To safeguard against configuration issues, this ensures that trailing slashes exist where required.
064   * A future enhancement would be for those not to be required by the depending components, but currently
065   * this is the state of affairs. This does at least help guard against incorrect use in the property
066   * files.
067   */
068  public static Config buildFromProperties() {
069    Config cfg = new Config();
070    try {
071      Properties properties = PropertiesUtil.loadProperties(APPLICATION_PROPERTIES);
072      properties.putAll(PropertiesUtil.loadProperties(STRUTS_PROPERTIES));
073
074      cfg.serverName = getServerName(properties);
075      cfg.googleAnalytics = StringUtils.trimToNull(properties.getProperty("google.analytics"));
076      cfg.drupal = getPropertyUrl(properties, "drupal.url", false);
077      cfg.drupalCookieName = properties.getProperty("drupal.cookiename");
078      cfg.wsClb = getPropertyUrl(properties, "checklistbank.ws.url", true);
079      final String clbSearchWs = getPropertyUrl(properties, "checklistbank.search.ws.url", true);
080      cfg.wsClbSearch = clbSearchWs + "search";
081      cfg.wsClbSuggest = clbSearchWs + "suggest";
082      cfg.wsReg = getPropertyUrl(properties, "registry.ws.url", true);
083      cfg.wsRegSearch = cfg.wsReg + "dataset/search";
084      cfg.wsRegSuggest = cfg.wsReg + "dataset/suggest";
085      cfg.wsOcc = getPropertyUrl(properties, "occurrence.ws.url", true);
086      cfg.wsOccSearch = cfg.wsOcc + OCC_SEARCH_PATH;
087      cfg.wsOccDownload = getPropertyUrl(properties, "occurrencedownload.ws.url", true);
088      cfg.wsOccDownloadForPublicLink = getPropertyUrl(properties, "occurrencedownload.ws.url.for.public.link", true);
089      cfg.maxOccDowloadSize = Integer.parseInt(properties.getProperty("occurrencedownload.size.limit"));
090      cfg.maxOccSearchOffset = Integer.parseInt(properties.getProperty("occurrence.search.maxoffset"));
091      cfg.wsMetrics = getPropertyUrl(properties, "metrics.ws.url", true);
092      cfg.wsOccCatalogNumberSearch = cfg.wsOcc + OCC_SEARCH_PATH + '/' + CATALOG_NUMBER_PATH;
093      cfg.wsOccCollectorNameSearch = cfg.wsOcc + OCC_SEARCH_PATH + '/' + COLLECTOR_NAME_PATH;
094      cfg.wsOccCollectionCodeSearch = cfg.wsOcc + OCC_SEARCH_PATH + '/' + COLLECTION_CODE_PATH;
095      cfg.wsOccInstitutionCodeSearch = cfg.wsOcc + OCC_SEARCH_PATH + '/' + INSTITUTION_CODE_PATH;
096      cfg.tileServerBaseUrl = getPropertyUrl(properties, "tile-server.url", false);
097      cfg.wsImageCache = getPropertyUrl(properties, "image-cache.url", false);
098      cfg.includeContext = Boolean.parseBoolean(properties.getProperty("struts.url.includeContext"));
099    } catch (IOException e) {
100      throw new ConfigurationException("application.properties cannot be read", e);
101    }
102    return cfg;
103  }

106  /**
107   * Reads the property as a URL, and will optionally force a trailing slash as required by
108   * http://dev.gbif.org/issues/browse/POR-260
109   */
110  private static String getPropertyUrl(Properties properties, String propName, boolean forceTrailingSlash) {
111    String value = null;
112    try {
113      value = properties.getProperty(propName);
114      value = (forceTrailingSlash && !value.endsWith("/")) ? value + '/' : value;
115      URI uri = URI.create(value);
116      return uri.toString();
117    } catch (Exception e) {
118      throw new ConfigurationException(String.format(PROP_CONF_ERROR_MSG, value, propName), e);
119    }
120  }
121
122  /**
123   * Gets the server name parameter from the properties.
124   */
125  private static String getServerName(Properties properties) {
126    // prefer system variable if existing, required (e.g.) by selenim
127    String serverName;
128    try {
129      serverName = URI.create(System.getProperty(SERVERNAME)).toString();
130      LOG.debug("Using servername system variable");
131    } catch (Exception e) {
132      serverName = getPropertyUrl(properties, SERVERNAME, false);
133    }
134    LOG.debug("Setting servername to {}", serverName);
135    return serverName;
136  }

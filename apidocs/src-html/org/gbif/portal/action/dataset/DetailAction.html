040/**
041 * Populates the model objects for the dataset detail page.
042 */
043@SuppressWarnings("serial")
044public class DetailAction extends DatasetBaseAction {
045
046  private static final List<Extension> EXTENSIONS = ImmutableList.copyOf(Extension.values());
047  private static final List<Kingdom> KINGDOMS = ImmutableList.of(Kingdom.ANIMALIA, Kingdom.ARCHAEA, Kingdom.BACTERIA,
048    Kingdom.CHROMISTA,
049    Kingdom.FUNGI, Kingdom.PLANTAE, Kingdom.PROTOZOA, Kingdom.VIRUSES, Kingdom.INCERTAE_SEDIS);
050
051  private static final Set<EndpointType> DATA_CODES = ImmutableSet.<EndpointType>builder()
052    .add(EndpointType.BIOCASE).add(EndpointType.TAPIR).add(EndpointType.DIGIR).add(EndpointType.DIGIR_MANIS)
053    .add(EndpointType.DWC_ARCHIVE).add(EndpointType.TCS_RDF).add(EndpointType.TCS_XML).add(EndpointType.WFS)
054    .build();
055  private static final Set<EndpointType> METADATA_CODES = ImmutableSet.<EndpointType>builder()
056    .add(EndpointType.EML).add(EndpointType.OAI_PMH).build();
057
058  private final List<Contact> preferredContacts = Lists.newArrayList();
059  private final List<Contact> otherContacts = Lists.newArrayList();
060  private final List<Endpoint> links = Lists.newArrayList();
061  private final List<Endpoint> dataLinks = Lists.newArrayList();
062  private final List<Endpoint> metaLinks = Lists.newArrayList();
063  private List<Network> networks;
064
065  @Nullable
066  private PagingResponse<Dataset> constituents;
067  private boolean renderMaps = false; // flag controlling map article rendering or not
068
069  private static final int PAGE_SIZE = 10;
070
071  @Inject
072  public DetailAction(DatasetService datasetService, CubeService occurrenceCubeService,
073    DatasetMetricsService metricsService, OrganizationService organizationService) {
074    super(datasetService, occurrenceCubeService, metricsService, organizationService);
075  }
076
077  @Override
078  public String execute() {
079    loadDetail();
080    populateContacts(member.getContacts());
081    populateLinks(member.getEndpoints());
082    // only datasets with a key (internal) can have constituents
083    if (id != null) {
084      constituents = datasetService.listConstituents(id, new PagingRequest(0, PAGE_SIZE));
085      super.loadCountWrappedDatasets(constituents);
086    }
087
088    // the map article is rendered only if the cube has indicated georeferenced records, or if there are
089    // coverages that are not global (they don't really warrant visualizing).
090    renderMaps = getNumGeoreferencedOccurrences() != null && getNumGeoreferencedOccurrences() > 0;
091    if (!renderMaps) {
092      for (GeospatialCoverage gc : member.getGeographicCoverages()) {
093        renderMaps = renderMaps
094          || (gc.getBoundingBox() != null && !gc.getBoundingBox().isGlobalCoverage());
095        if (renderMaps) {
096          break; // for speed
097        }
098      }
099    }
100    // get networks
101    networks = datasetService.listNetworks(id);
102
103    return SUCCESS;
104  }
105
106  /**
107   * Will not be null following execute() invocation.
108   */
109  @Nullable
110  public PagingResponse<Dataset> getConstituents() {
111    return constituents;
112  }
113

115  public List<Endpoint> getDataLinks() {
116    return dataLinks;
117  }

123    return EXTENSIONS;
124  }
125
126  /**
127   * Exposed to allow easy access in freemarker.
128   */
129  public List<Kingdom> getKingdoms() {
130    return KINGDOMS;
131  }
132
133  @NotNull
134  public List<Endpoint> getLinks() {
135    return links;
136  }
137
138  @NotNull
139  public List<Endpoint> getMetaLinks() {
140    return metaLinks;
141  }
142
143  @NotNull
144  public List<Contact> getOtherContacts() {
145    return otherContacts;
146  }
147
148  @Override
149  @Nullable
150  public Dataset getParentDataset() {
151    return parentDataset;
152  }
153
154  @NotNull
155  public List<Contact> getPreferredContacts() {
156    return preferredContacts;
157  }
158
159  /**
160   * Exposed to allow easy access in freemarker.
161   */
162  public List<Rank> getSortedMetricRanks() {
163    List<Rank> sortedRanks = Lists.newArrayList(metrics.getCountByRank().keySet());
164    Collections.sort(sortedRanks);
165    return sortedRanks;
166  }
167
168  @NotNull
169  public Map<String, String> getResourceBundleProperties() {
170    return getResourceBundleProperties("enum.rank.");
171  }
172
173  public boolean isRenderMaps() {
174    return renderMaps;
175  }
176
177  /**
178   * Populates the preferred and other contact lists from the unified contact list.
179   */
180  private void populateContacts(@NotNull List<Contact> contacts) {
181    Preconditions.checkNotNull(contacts, "Contacts cannot be null - expected empty list for no contacts");
182    preferredContacts.clear(); // for safety
183    otherContacts.clear();
184    for (Contact contact : contacts) {
185      if (contact.isPrimary()) {
186        preferredContacts.add(contact);
187      } else {
188        otherContacts.add(contact);
189      }
190    }
191  }
192
193  /**
194   * Populates the data, metadata and other links from the unified endpoint list.
195   */
196  private void populateLinks(@NotNull List<Endpoint> endpoints) {
197    Preconditions.checkNotNull(endpoints, "Enpoints cannot be null - expected empty list for no endpoints");
198    dataLinks.clear(); // for safety
199    metaLinks.clear();
200    links.clear();
201    for (Endpoint p : member.getEndpoints()) {
202      if (DATA_CODES.contains(p.getType())) {
203        dataLinks.add(p);
204      } else if (METADATA_CODES.contains(p.getType())) {
205        metaLinks.add(p);
206      } else {
207        links.add(p);
208      }
209    }
210  }
211
212  public List<Network> getNetworks() {
213    return networks;
214  }

030/**
031 * A base class that provides accession to utilities for freemarker.
032 */
033@SuppressWarnings("serial")
034public abstract class BaseAction extends ActionSupport
035  implements SessionAware, ServletRequestAware, ServletContextAware {
036
037  protected static Logger LOG = LoggerFactory.getLogger(BaseAction.class);
038  private static Joiner QUERY_JOINER = Joiner.on('&');
039  private static Splitter QUERY_SPLITTER = Splitter.on("&");
040
041  protected Map<String, Object> session;
042  protected HttpServletRequest request;
043  protected ServletContext ctx;
044
045  /**
046   * The threshold that switches between 1px resolution and 4px resolution.
047   */
048  public static final int MAP_RESOLUTION_THRESHOLD = 50000;
049
050  @Inject
051  private Config cfg;
052
053  /**
054   * Checks whether a string starts with any of the prefixes specified
055   *
056   * @return true if string matches against any prefix. false otherwise.
057   */
058  private static boolean containsPrefix(String propertyKey, String[] prefixes) {
059    if (propertyKey != null) {
060      for (String prefix : prefixes) {
061        if (propertyKey.startsWith(prefix)) {
062          return true;
063        }
064      }
065    }
066    return false;
067  }

070  /**
071   * Exposed to make configurations available to javascript.
072   */
073  public Config getCfg() {
074    return cfg;
075  }
076
077  /**
078   * Returns the application's base url. Exposed to simplify javascript.
079   * It takes into account the includeContext struts settings to strip of the context if not used.
080   *
081   * @return the base url of the application
082   */
083  public String getBaseUrl() {
084    if (cfg.isIncludeContext()) {
085      return cfg.getServerName() + ctx.getContextPath();
086    }
087    return cfg.getServerName();
088  }
089
090  /**
091   * Returns the absolute url to the current page including the query string.
092   * Exposed to simplify freemarker and javascript.
093   * It takes into account the includeContext struts settings to strip of the context if not used.
094   *
095   * @return the absolute, current url
096   */
097  public String getCurrentUrl() {
098    return getCurrentUrl(false);
099  }
100
101  /**
102   * Returns the absolute url to the current page including the query string, but having the paging
103   * parameters offset and limit removed.
104   * Exposed for the paging macros to not have to deal with this complexity in freemarker.
105   */
106  public String getCurrentUrlWithoutPage() {
107    return getCurrentUrl(true);
108  }
109
110  /**
111   * Used by the drupal login destination parameter which expects a relative URL of the current page.
112   * We need to remove any scheme, host or port information and also the root slash.
113   */
114  public String getCurrentDestinationParam() {
115    String url = getCurrentUrl();
116    try {
117      URI u = URI.create(url);
118      return URIUtils.rewriteURI(u, null).toString().substring(1);
119
120    } catch (Exception e) {
121      LOG.warn("Cannot produce the login destination URL for {}", url);
122      // return root if we ever should encounter an invalid URI
123      return "/";
124    }
125  }
126
127  private String getCurrentUrl(boolean removePagingParams) {
128    StringBuffer currentUrl = request.getRequestURL();
129
130    if (!cfg.isIncludeContext()) {
131      // dont replace context name in host name
132      final int start = currentUrl.indexOf(ctx.getContextPath(), cfg.getServerName().length()-1);
133      currentUrl.replace(start, start+ctx.getContextPath().length(), "");
134    }
135
136    if (request.getQueryString() != null) {
137      currentUrl.append("?");
138      if (removePagingParams) {
139        currentUrl.append(QUERY_JOINER.join(
140          Iterables.filter(QUERY_SPLITTER.split(request.getQueryString()), new Predicate<String>() {
141            @Override
142            public boolean apply(@Nullable String input) {
143              if (input.toLowerCase().startsWith("offset=") || input.toLowerCase().startsWith("limit=")) {
144                return false;
145              }
146              return true;
147            }
148          })
149        ));
150      } else {
151        currentUrl.append(request.getQueryString());
152      }
153    }
154    return currentUrl.toString();
155  }

157  /**
158   * @return the currently logged in user.
159   */
160  public User getCurrentUser() {
161    return (User) session.get(Constants.SESSION_USER);
162  }
163
164  /**
165   * Returns a map representing properties from the resource bundle but just those
166   * properties whose keys match one or more of the given prefixes.
167   *
168   * @return a map which the matched properties
169   */
170  public Map<String, String> getResourceBundleProperties(String... prefix) {
171    Map<String, String> bundleProps = new TreeMap<String, String>();
172    ResourceBundle bundle = ResourceBundle.getBundle("resources", getLocale());
173    // properties should be filtered out
174    if (prefix != null && prefix.length != 0) {
175      for (String key : bundle.keySet()) {
176        // only add those properties whose key starts with one of the prefixes given
177        if (containsPrefix(key, prefix)) {
178          bundleProps.put(key, bundle.getString(key));
179        }
180      }
181    } else { // just get all properties without any filtering at all
182      for (String key : bundle.keySet()) {
183        bundleProps.put(key, bundle.getString(key));
184      }
185    }
186    return bundleProps;
187  }

205  }

222  public String getImageCache(String url, String size) {
223    try {
224      return cfg.getWsImageCache() + "?url=" + URLEncoder.encode(url, "utf8") + "&size=" + size;
225    } catch (UnsupportedEncodingException e) {
226      return cfg.getWsImageCache() + "?url=" + url + "&size=" + size;
227    }
228  }
229
230  /**
231   * A utility to provide the map resolution based on the provided record count.
232   * 
233   * @return the suggested resolution to use for the map.
234   */
235  public int getMapResolution(int recordCount) {
236    return (recordCount > MAP_RESOLUTION_THRESHOLD) ? 1 : 4;
237  }

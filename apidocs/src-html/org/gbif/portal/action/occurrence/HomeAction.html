001package org.gbif.portal.action.occurrence;
002
003import org.gbif.api.model.metrics.cube.OccurrenceCube;
004import org.gbif.api.model.metrics.cube.ReadBuilder;
005import org.gbif.api.service.metrics.CubeService;
006import org.gbif.api.service.occurrence.OccurrenceDistributionIndexService;
007import org.gbif.api.vocabulary.BasisOfRecord;
008import org.gbif.api.vocabulary.Kingdom;
009import org.gbif.portal.action.BaseAction;
010
011import java.util.Calendar;
012import java.util.Map;
013
014import com.google.common.base.Enums;
015import com.google.common.base.Optional;
016import com.google.inject.Inject;
017
018
019/**
020 * Occurrence search home action.
021 */
022public class HomeAction extends BaseAction {
023
024  private static final long serialVersionUID = 374193477998601641L;
025
026  private static final int MIN_YEAR = 1950;
027
028  private static final int maxYear = Calendar.getInstance().get(Calendar.YEAR); // this year
029
030  protected CubeService occurrenceCubeService;
031
032  private final OccurrenceDistributionIndexService occurrenceDistributionIndexService;
033
034  private Integer numGeoreferenced;
035
036  private Integer numOccurrences;
037  
038  private Map<Kingdom, Long> kingdomCounts; 
039  
040  private Map<BasisOfRecord, Long> borCounts;
041  
042  private Map<Integer, Long> yearCounts;
043
044  @Inject
045  public HomeAction(CubeService occurrenceCubeService,
046    OccurrenceDistributionIndexService occurrenceDistributionIndexService) {
047    this.occurrenceCubeService = occurrenceCubeService;
048    this.occurrenceDistributionIndexService = occurrenceDistributionIndexService;
049  }
050
051  @Override
052  public String execute() {
053    numGeoreferenced = (int) occurrenceCubeService.get(new ReadBuilder().at(OccurrenceCube.IS_GEOREFERENCED, true));
054    numOccurrences = (int) occurrenceCubeService.get(new ReadBuilder());
055    borCounts = occurrenceDistributionIndexService.getBasisOfRecordCounts();
056    kingdomCounts = occurrenceDistributionIndexService.getKingdomCounts();
057    yearCounts = occurrenceDistributionIndexService.getYearCounts(MIN_YEAR, maxYear);
058    return SUCCESS;
059  }
060
061  public Map<BasisOfRecord, Long> getBasisOfRecordCounts() {
062    return borCounts;
063  }
064
065  public Map<Kingdom, Long> getKingdomCounts() {
066    return kingdomCounts;
067  }
068
069
070  public String getKingdomNubUsageId(String kingdom) {
071    Optional<Kingdom> opKingdom = Enums.getIfPresent(Kingdom.class, kingdom);
072    if (opKingdom.isPresent()) {
073      return opKingdom.get().nubUsageID().toString();
074    }
075    return kingdom;
076  }
077
078  public Integer getMaxYear() {
079    return maxYear;
080  }
081
082  public Integer getMinYear() {
083    return MIN_YEAR;
084  }
085
086  public Integer getNumGeoreferenced() {
087    return numGeoreferenced;
088  }
089
090  public Integer getNumOccurrences() {
091    return numOccurrences;
092  }
093
094  public Map<Integer, Long> getYearCounts() {
095    return yearCounts;
096  }
097}

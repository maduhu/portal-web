066  @Override
067  public String execute() throws Exception {
068    country = Country.fromIsoCode(id);
069    if (country == null) {
070      throw new NotFoundException("No country found with ISO code [" + id + ']');
071    }
072
073    member = nodeService.getByCountry(country);
074    sortContacts();
075
076    return SUCCESS;
077  }

117  /**
118   * populates the about field and optionally also loads the first requested datasets into the datasets property.
119   * This allows to only call the index service once effectively and process its response.
120   * 
121   * @param numDatasetsToLoad number of datasets to load, if zero or negative doesnt load any
122   */
123  protected void buildAboutMetrics(int numDatasetsToLoad, int numCountriesToLoad) {
124    // TODO: use checklist search to populate this???
125    final long chkRecords = -1;
126    final long chkDatasets = -1;
127    final int organizations = -1;
128
129    final long occRecords = cubeService.get(new ReadBuilder().at(OccurrenceCube.COUNTRY, country));
130
131    final long occDatasets = loadAboutDatasetsPage(numDatasetsToLoad);
132
133    // find number of external datasets
134    DatasetSearchRequest search = new DatasetSearchRequest();
135    search.addTypeFilter(DatasetType.METADATA);
136    search.addCountryFilter(country);
137    search.setLimit(0);
138    SearchResponse<DatasetSearchResult, DatasetSearchParameter> resp = datasetSearchService.search(search);
139    final long extDatasets = resp.getCount();
140
141    int countryCount = loadCountryPage(true, numCountriesToLoad);
142
143    about =
144      new CountryMetrics(occDatasets, occRecords, chkDatasets, chkRecords, extDatasets, organizations, countryCount);
145  }
146
147  protected void buildByMetrics(int numDatasetsToLoad, int numCountriesToLoad) {
148    final long occRecords = cubeService.get(new ReadBuilder().at(OccurrenceCube.HOST_COUNTRY, country));
149
150    // we only want the counts here
151    PagingRequest p = new PagingRequest(0, 0);
152    final long occDatasets = datasetService.listByCountry(country, DatasetType.OCCURRENCE, p).getCount();
153    final long chkDatasets = datasetService.listByCountry(country, DatasetType.CHECKLIST, p).getCount();
154    final long extDatasets = datasetService.listByCountry(country, DatasetType.METADATA, p).getCount();
155    final int organizations = -1;
156
157    // for total checklist counts we need to retrieve the stats for every single checklist
158    // we do not have a checklist cube yet
159    long chkRecords = 0;
160    p = new PagingRequest(0, 1000);  // there are only 100 checklists alltogether!
161    for (Dataset chkl : datasetService.listByCountry(country, DatasetType.CHECKLIST, p).getResults()) {
162      DatasetMetrics metric = datasetMetricsService.get(chkl.getKey());
163      if (metric != null) {
164        // count them all
165        chkRecords += metric.getUsagesCount();
166      }
167    }
168
169    // load full datasets preview if requested
170    if (numDatasetsToLoad > 0) {
171      p = new PagingRequest(0, numDatasetsToLoad);
172      PagingResponse<Dataset> datasetPage = datasetService.listByCountry(country, null, p);
173      super.loadCountWrappedDatasets(datasetPage);
174    }
175
176    int countryCount = loadCountryPage(false, numCountriesToLoad);
177
178    by = new CountryMetrics(occDatasets, occRecords, chkDatasets, chkRecords, extDatasets, organizations, countryCount);
179  }

181  /**
182   * Honors the offset paging parameter.
183   * 
184   * @return the number of all datasets having data about this country
185   */
186  protected int loadAboutDatasetsPage(int limit) {
187    Map<UUID, Integer> dsMetrics = datasetIndexService.occurrenceDatasetsForCountry(country);
188    int idx = 0;
189    for (Map.Entry<UUID, Integer> metric : dsMetrics.entrySet()) {
190      if (idx >= getOffset() + limit) {
191        break;
192      }
193      // only load the requested page
194      if (idx >= getOffset()) {
195        Dataset d = datasetService.get(metric.getKey());
196        if (d == null) {
197          LOG.warn("Dataset in cube, but not in registry; {}", metric.getKey());
198
199        } else {
200          long total = cubeService.get(new ReadBuilder()
201            .at(OccurrenceCube.DATASET_KEY, d.getKey()));
202          datasets.add(new CountWrapper<Dataset>(d, metric.getValue(), total));
203        }
204      }
205      idx++;
206    }
207
208    datasetPage = new PagingResponse<Dataset>(getOffset(), limit, (long) dsMetrics.size());
209
210    return dsMetrics.size();
211  }
212
213  protected int loadCountryPage(boolean isAboutCountry, int limit) {
214    try {
215      Map<Country, Long> cMap;
216      if (isAboutCountry) {
217        cMap = countryIndexService.publishingCountriesForCountry(country);
218      } else {
219        cMap = countryIndexService.countriesForPublishingCountry(country);
220      }
221
222      countryPage = new PagingResponse<Country>(getOffset(), limit, (long) cMap.size());
223      loadCountryList(cMap, isAboutCountry, limit);
224
225      return cMap.size();
226
227    } catch (RuntimeException e) {
228      LOG.error("Cannot get country metrics", e);
229    }
230    return 0;
231  }
232
233  private void loadCountryList(Map<Country, Long> cMetrics, boolean isAboutCountry, int limit) {
234    ReadBuilder rb = new ReadBuilder().at(OccurrenceCube.IS_GEOREFERENCED, true);
235    if (isAboutCountry) {
236      rb.at(OccurrenceCube.COUNTRY, country);
237    } else {
238      rb.at(OccurrenceCube.HOST_COUNTRY, country);
239    }
240
241    int idx = 0;
242    for (Map.Entry<Country, Long> metric : cMetrics.entrySet()) {
243      if (idx >= getOffset() + limit) {
244        break;
245      }
246
247      // only load the requested page
248      if (idx >= getOffset()) {
249        if (isAboutCountry) {
250          rb.at(OccurrenceCube.HOST_COUNTRY, metric.getKey());
251        } else {
252          rb.at(OccurrenceCube.COUNTRY, metric.getKey());
253        }
254
255        long geoCnt = cubeService.get(rb);
256        countries.add(new CountWrapper<Country>(metric.getKey(), metric.getValue(), geoCnt));
257      }
258
259      idx++;
260    }
261  }

001package org.gbif.portal.struts;
002
003import java.io.Writer;
004
005import com.opensymphony.xwork2.inject.Inject;
006import org.apache.struts2.components.Form;
007import org.apache.struts2.components.ServletUrlRenderer;
008import org.apache.struts2.components.UrlProvider;
009import org.slf4j.Logger;
010import org.slf4j.LoggerFactory;
011
012/**
013 * Custom struts URL renderer that takes a global configuration to turn on/off the addition of the servlet context
014 * to a URL. This is needed for GBIF to run the portal with a /portal context, but expose it publicly behind
015 * varnish in the root context.
016 */
017public class ConfigurableServletUrlRenderer extends ServletUrlRenderer {
018  private static final Logger LOG = LoggerFactory.getLogger(ConfigurableServletUrlRenderer.class);
019
020  private final boolean addContext;
021
022  @Inject("struts.url.includeContext")
023  public ConfigurableServletUrlRenderer(String addContext) {
024    LOG.info("Configure struts tags to include the servlet context: " + addContext);
025    this.addContext = Boolean.parseBoolean(addContext);
026  }
027
028  @Override
029  public void renderUrl(Writer writer, UrlProvider urlComponent) {
030    urlComponent.setIncludeContext(addContext);
031    super.renderUrl(writer, urlComponent);
032  }
033
034  @Override
035  public void renderFormUrl(Form formComponent) {
036    formComponent.setIncludeContext(addContext);
037    super.renderFormUrl(formComponent);
038  }
039}

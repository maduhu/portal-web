64    public String verbatim() {
65      LOG.debug("Loading raw details for occurrence id [{}]", id);
66  
67      loadDetail();
68  
69      // prepare verbatim map
70      verbatim = Maps.newLinkedHashMap();
71      try {
72        fragmentExists = occurrenceService.getFragment(id) != null;
73        VerbatimOccurrence v = verbatimService.getVerbatim(id);
74        for (String group : DwcTerm.GROUPS) {
75          for (DwcTerm t : DwcTerm.listByGroup(group)) {
76            if (v.getFields().containsKey(t.simpleName())) {
77              if (!verbatim.containsKey(group)) {
78                verbatim.put(group, new TreeMap<String, String>());
79              }
80              verbatim.get(group).put(t.simpleName(), v.getFields().get(t.simpleName()));
81            }
82          }
83        }
84      } catch (Exception e) {
85        LOG.error("Can't load verbatim data for occurrence {}: {}", id, e);
86      }
87  
88      return SUCCESS;
89    }

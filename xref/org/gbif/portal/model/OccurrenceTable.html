130   private boolean isParameterPresent(SearchRequest<OccurrenceSearchParameter> request,
131     OccurrenceSearchParameter... parameters) {

140   /**
141    * Retrieve the columns from the request parameter.
142    * The default fields are returned if no value is gotten from the request.
143    */
144   private EnumSet<OccurrenceColumn> retrieveColumns(HttpServletRequest request,
145     SearchRequest<OccurrenceSearchParameter> searchRequest) {
146     EnumSet<OccurrenceColumn> requestCols = retrieveEnumParams(request, OccurrenceColumn.class, COLUMNS_PARAM);
147     if (requestCols.isEmpty()) {
148       requestCols = EnumSet.allOf(OccurrenceColumn.class);
149     }
150     if (isParameterPresent(searchRequest, OCC_LOCATION_PARAMS)) {
151       requestCols.add(OccurrenceColumn.LOCATION);
152     }
153     if (isParameterPresent(searchRequest, OccurrenceSearchParameter.BASIS_OF_RECORD)) {
154       requestCols.add(OccurrenceColumn.BASIS_OF_RECORD);
155     }
156     if (isParameterPresent(searchRequest, OCC_DATE_PARAMS)) {
157       requestCols.add(OccurrenceColumn.DATE);
158     }
159     return requestCols;
160   }

163   /**
164    * Retrieve an EnumSet containing a list of Enum literals if type T that can be obtained from the request parameter
165    * paramName.
166    */
167   private <T extends Enum<T>> EnumSet<T> retrieveEnumParams(HttpServletRequest request, Class<T> enumClass,
168     String paramName) {
169     EnumSet<T> allCols = EnumSet.noneOf(enumClass);
170     final String[] values = request.getParameterValues(paramName);
171     if (values != null) {
172       for (String paramValue : values) {
173         for (String value : paramValue.split(",")) {
174           Optional<T> optValue = Enums.getIfPresent(enumClass, value);
175           if (optValue.isPresent()) {
176             allCols.add(optValue.get());
177           }
178         }
179       }
180     }
181     return allCols;
182   }

185   /**
186    * Retrieve the summary fields from the request.
187    * The default fields are returned if no value is gotten from the request.
188    */
189   private EnumSet<OccurrenceSummaryField> retrieveSummaryFields(HttpServletRequest request,
190     SearchRequest<OccurrenceSearchParameter> searchRequest) {
191     EnumSet<OccurrenceSummaryField> fields =
192       retrieveEnumParams(request, OccurrenceSummaryField.class, SUMMARY_FIELDS_PARAM);
193     if (fields.isEmpty()) {
194       fields.addAll(defaulSummaryFields);
195     }
196     if (isParameterPresent(searchRequest, OccurrenceSearchParameter.CATALOG_NUMBER)) {
197       fields.add(OccurrenceSummaryField.CATALOG_NUMBER);
198     }
199     if (isParameterPresent(searchRequest, OccurrenceSearchParameter.COLLECTION_CODE)) {
200       fields.add(OccurrenceSummaryField.COLLECTION_CODE);
201     }
202     if (isParameterPresent(searchRequest, OccurrenceSearchParameter.COLLECTOR_NAME)) {
203       fields.add(OccurrenceSummaryField.COLLECTOR_NAME);
204     }
205     if (isParameterPresent(searchRequest, OccurrenceSearchParameter.INSTITUTION_CODE)) {
206       fields.add(OccurrenceSummaryField.INSTITUTION);
207     }
208     if (isParameterPresent(searchRequest, OccurrenceSearchParameter.TAXON_KEY)) {
209       fields.add(OccurrenceSummaryField.SCIENTIFIC_NAME);
210     }
211     if (isParameterPresent(searchRequest, OccurrenceSearchParameter.MODIFIED)) {
212       fields.add(OccurrenceSummaryField.MODIFIED);
213     }
214     return fields;
215   }

001package org.gbif.portal.action.country;
002
003import org.gbif.api.service.registry.NodeService;
004import org.gbif.api.vocabulary.Country;
005import org.gbif.portal.action.BaseAction;
006import org.gbif.portal.config.ContinentCountryMap;
007
008import java.util.List;
009import java.util.Set;
010import javax.annotation.Nullable;
011
012import com.google.common.base.Function;
013import com.google.common.base.Predicate;
014import com.google.common.collect.FluentIterable;
015import com.google.common.collect.Lists;
016import com.google.common.collect.Ordering;
017import com.google.common.collect.Sets;
018import com.google.inject.Inject;
019
020public class HomeAction extends BaseAction {
021
022  /**
023   * Ensure we are ordered by title, removing noise.
024   */
025  private static List<Country> countries = FluentIterable.from(Lists.newArrayList(Country.values()))
026    // remove unofficial countries
027    .filter(new Predicate<Country>() {
028      public boolean apply(@Nullable Country n) {
029        return n.isOfficial();
030      }
031    })
032    // sort alphabetically
033    .toSortedList(Ordering.natural().onResultOf(new Function<Country, String>() {
034      @Nullable
035      @Override
036      public String apply(@Nullable Country n) {
037        return n == null ? null : n.getTitle();
038      }
039    }));
040
041  private Set<Country> activeNodes;
042  @Inject
043  private NodeService nodeService;
044  @Inject
045  private ContinentCountryMap continentMap;
046
047  @Override
048  public String execute() throws Exception {
049    activeNodes = Sets.newHashSet(nodeService.listActiveCountries());
050    return SUCCESS;
051  }
052
053  public Set<Country> getActiveNodes() {
054    return activeNodes;
055  }
056
057  public List<Country> getCountries() {
058    return countries;
059  }
060
061}

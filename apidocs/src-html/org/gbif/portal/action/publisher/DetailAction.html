001package org.gbif.portal.action.publisher;
002
003import org.gbif.api.model.common.paging.PagingRequest;
004import org.gbif.api.model.common.paging.PagingResponse;
005import org.gbif.api.model.registry.Contact;
006import org.gbif.api.model.registry.Dataset;
007import org.gbif.api.model.registry.Installation;
008import org.gbif.api.model.registry.Node;
009import org.gbif.api.model.registry.Organization;
010import org.gbif.api.service.checklistbank.DatasetMetricsService;
011import org.gbif.api.service.metrics.CubeService;
012import org.gbif.api.service.registry.NodeService;
013import org.gbif.api.service.registry.OrganizationService;
014import org.gbif.portal.action.member.MemberBaseAction;
015import org.gbif.portal.action.member.MemberType;
016
017import java.util.List;
018
019import com.google.common.collect.Lists;
020import com.google.inject.Inject;
021
022public class DetailAction extends MemberBaseAction<Organization> {
023
024  @Inject
025  private NodeService nodeService;
026  private final OrganizationService organizationService;
027
028  private Node node;
029  private PagingResponse<Dataset> page;
030  private PagingResponse<Installation> installationsPage;
031  private long offset = 0;
032  private List<Contact> primaryContacts;
033  private List<Contact> otherContacts;
034
035  @Inject
036  public DetailAction(OrganizationService organizationService, CubeService cubeService,
037    DatasetMetricsService datasetMetricsService) {
038    super(MemberType.PUBLISHER, organizationService, cubeService, datasetMetricsService, organizationService);
039    this.organizationService = organizationService;
040  }
041
042  public String datasets() throws Exception {
043    super.execute();
044    page = organizationService.ownedDatasets(id, new PagingRequest(offset, 25));
045    return SUCCESS;
046  }
047
048  public String installations() throws Exception {
049    super.execute();
050    installationsPage = organizationService.installations(id, new PagingRequest(offset, 25));
051    return SUCCESS;
052  }
053
054  @Override
055  public String execute() throws Exception {
056    super.execute();
057    // load endorsing node
058    if (member.getEndorsingNodeKey() != null) {
059      node = nodeService.get(member.getEndorsingNodeKey());
060    }
061    // load first 10 datasets
062    page = organizationService.ownedDatasets(id, new PagingRequest(0, 10));
063    super.loadCountWrappedDatasets(page);
064
065    // load first 10 hosted installations
066    installationsPage = organizationService.installations(id, new PagingRequest(0, 10));
067    // separate contacts into 2 lists: primary contacts and other contacts
068    loadSeparateContacts();
069
070    return SUCCESS;
071  }
072
073  public Node getNode() {
074    return node;
075  }
076
077  public PagingResponse<Dataset> getPage() {
078    return page;
079  }
080
081  public PagingResponse<Installation> getInstallationsPage() {
082    return installationsPage;
083  }
084
085  public void setOffset(long offset) {
086    if (offset >= 0) {
087      this.offset = offset;
088    }
089  }
090
091  /**
092   * Get a list of publisher's contacts that are designated as primary.
093   *
094   * @return a list of publisher's contacts that are designated as primary
095   */
096  public List<Contact> getPrimaryContacts() {
097    return primaryContacts;
098  }
099
100  /**
101   * Get a list of publisher's contacts that are not designated as primary.
102   *
103   * @return a list of publisher's contacts that are not designated as primary
104   */
105  public List<Contact> getOtherContacts() {
106    return otherContacts;
107  }
108
109  /**
110   * Separate member's contacts into 2 lists: primary contacts, and others.
111   */
112  private void loadSeparateContacts() {
113    // reinitialize lists
114    primaryContacts = Lists.newLinkedList();
115    otherContacts = Lists.newLinkedList();
116    for (Contact contact : member.getContacts()) {
117      if (contact.isPrimary()) {
118        primaryContacts.add(contact);
119      } else {
120        otherContacts.add(contact);
121      }
122    }
123  }
124}

001package org.gbif.portal.config;
002
003import org.gbif.api.model.common.User;
004import org.gbif.api.model.common.UserPrincipal;
005
006import java.security.Principal;
007import javax.servlet.http.HttpSession;
008
009import com.google.inject.Inject;
010import com.google.inject.Provider;
011import com.google.inject.Singleton;
012
013@Singleton
014public class SessionAuthProvider implements Provider<Principal>{
015  private static final String SESSION_USER = "current_user";
016  private final Provider<HttpSession> sessionProvider;
017
018  @Inject
019  public SessionAuthProvider(Provider<HttpSession> sessionProvider) {
020    this.sessionProvider = sessionProvider;
021  }
022
023  @Override
024  public Principal get() {
025    HttpSession session = sessionProvider.get();
026    if (session != null && session.getAttribute(SESSION_USER) != null){
027      return new UserPrincipal( (User) session.getAttribute(SESSION_USER) );
028    }
029    return null;
030  }
031}

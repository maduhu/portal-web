001package org.gbif.portal.action;
002
003import org.junit.Test;
004
005import static org.junit.Assert.assertEquals;
006
007public class BaseSearchActionTest {
008
009  @Test
010  public void testGetHighlightedText() throws Exception {
011    assertEquals("hallo markus", BaseSearchAction.getHighlightedText("hallo markus", 20));
012    assertEquals("hallo markus, hallo…", BaseSearchAction.getHighlightedText("hallo markus, hallo pia, hallo carla", 20));
013    final String text = "Strings are constant; their values cannot be changed after they are created. String <em class=\"gbifHl\">buffers</em> support mutable strings. Because String objects are immutable they can be shared.";
014    assertEquals("<em class=\"gbifHl\">buffers</em>", BaseSearchAction.getHighlightedText(text, 6));
015    assertEquals("<em class=\"gbifHl\">buffers</em>", BaseSearchAction.getHighlightedText(text, 7));
016    assertEquals("…<em class=\"gbifHl\">buffers</em> …", BaseSearchAction.getHighlightedText(text, 10));
017    assertEquals("…g <em class=\"gbifHl\">buffers</em> supp…", BaseSearchAction.getHighlightedText(text, 16));
018    assertEquals("…tring <em class=\"gbifHl\">buffers</em> support mutabl…", BaseSearchAction.getHighlightedText(text, 30));
019    assertEquals("…created. String <em class=\"gbifHl\">buffers</em> support mutable strings. Because S…", BaseSearchAction.getHighlightedText(text, 60));
020    assertEquals(text, BaseSearchAction.getHighlightedText(text, 200));
021
022    final String text2 = "Strings are <em class=\"gbifHl\">constant</em>; their values cannot be changed after they are created. String buffers support mutable strings. Because String objects are immutable they can be shared.";
023    assertEquals("…<em class=\"gbifHl\">constant</em>…", BaseSearchAction.getHighlightedText(text2, 10));
024    assertEquals("Strings are <em class=\"gbifHl\">constant</em>; their values cannot be changed after they are created. St…", BaseSearchAction.getHighlightedText(text2, 80));
025  }
026
027
028  @Test
029  public void testLimitHighlightedText() throws Exception {
030    assertEquals("hall…", BaseSearchAction.limitHighlightedText("hallo markus", 5));
031    assertEquals("hallo markus", BaseSearchAction.limitHighlightedText("hallo markus", 15));
032
033    final String text = "String <em class=\"gbifHl\">buffers</em> support mutable strings.";
034    assertEquals("String …", BaseSearchAction.limitHighlightedText(text, 8));
035    assertEquals("String …", BaseSearchAction.limitHighlightedText(text, 12));
036    assertEquals("String <em class=\"gbifHl\">buffers</em>…", BaseSearchAction.limitHighlightedText(text, 15));
037    assertEquals("String <em class=\"gbifHl\">buffers</em> support mutabl…", BaseSearchAction.limitHighlightedText(text, 30));
038    // Test addresses this issue: http://dev.gbif.org/issues/browse/PF-1247
039    final String longText = "<em class=\"gbifHl\">FADA Ephemeroptera: World checklist of freshwater Ephemeroptera species in the Catalogue of Life</em>";
040    assertEquals("<em class=\"gbifHl\">FADA Ephemeroptera: World checklist of freshwater…</em>", BaseSearchAction.limitHighlightedText(longText, 68));
041  }
042}

001package org.gbif.portal.struts;
002
003import java.util.Iterator;
004import java.util.Set;
005
006import javax.validation.ConstraintViolation;
007import javax.validation.Validator;
008
009import com.google.inject.Inject;
010import com.opensymphony.xwork2.ActionInvocation;
011import com.opensymphony.xwork2.Validateable;
012import com.opensymphony.xwork2.interceptor.MethodFilterInterceptor;
013import com.opensymphony.xwork2.validator.DelegatingValidatorContext;
014import com.opensymphony.xwork2.validator.ValidatorContext;
015import org.slf4j.Logger;
016import org.slf4j.LoggerFactory;
017
018/*
019 * This interceptor provides validation using the BVal validation framework
020 */
021public class ValidationInterceptor extends MethodFilterInterceptor {
022
023  private static final Logger LOG = LoggerFactory.getLogger(ValidationInterceptor.class);
024
025  private Validator validator;
026
027  /**
028   * @param validator
029   */
030  @Inject
031  public ValidationInterceptor(Validator validator) {
032    this.validator = validator;
033  }
034
035  @Override
036  protected String doIntercept(ActionInvocation invocation) throws Exception {
037    Object action = invocation.getAction();
038
039    if (LOG.isDebugEnabled()) {
040      LOG.debug("Action class: [{}]", action);
041    }
042
043    // perform validation
044    performValidation(action);
045
046    return invocation.invoke();
047  }
048
049  /**
050   * Performs the validation.
051   * 
052   * @param action current action class.
053   */
054  private void performValidation(Object action) {
055    if (action instanceof Validateable) {
056      // validates the action's class members which have been annotated.
057      Set<ConstraintViolation<Object>> violations = validator.validate(action);
058      Iterator iterator = violations.iterator();
059      ValidatorContext validatorContext = new DelegatingValidatorContext(action);
060      // iterate over all violations.
061      while (iterator.hasNext()) {
062        ConstraintViolation violation = (ConstraintViolation) iterator.next();
063        String fieldName = violation.getPropertyPath().toString();
064        String i18nMsg = fieldName + ".error";
065        // adds the error to a given field. The error message is a i18n key so that it will be easy to internationalize.
066        // the i18n key will take the form 'fieldName.error'.
067        validatorContext.addFieldError(fieldName, i18nMsg);
068      }
069      // TODO: perform normal validation?
070      // Validateable validateable = (Validateable) action;
071      // validateable.validate();
072    }
073  }
074}

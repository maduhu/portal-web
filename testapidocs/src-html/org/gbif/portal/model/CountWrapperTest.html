001package org.gbif.portal.model;
002
003import com.google.common.collect.Lists;
004import org.junit.Test;
005
006import static junit.framework.Assert.assertEquals;
007import static org.junit.Assert.assertFalse;
008import static org.junit.Assert.assertTrue;
009
010public class CountWrapperTest {
011  CountWrapper<String> cw1 = new CountWrapper<String>("Tim", 10);
012  CountWrapper<String> cw2 = new CountWrapper<String>("Timmy", 20);
013  CountWrapper<String> cw3 = new CountWrapper<String>("Timo", 31);
014  CountWrapper<String> cw1b = new CountWrapper<String>("Tim", 10);
015  CountWrapper<String> cw1c = new CountWrapper<String>("Tim", 1);
016  CountWrapper<String> cw1d = new CountWrapper<String>("Tim", 10, 8);
017
018  @Test
019  public void testSum() throws Exception {
020    assertEquals(61l, CountWrapper.sum(Lists.newArrayList(cw1, cw2, cw3)));
021  }
022
023  @Test
024  public void testCompareTo() throws Exception {
025    for (CountWrapper cw : Lists.newArrayList(cw1, cw2, cw1b, cw1c, cw1d)) {
026      assertTrue(cw3.compareTo(cw) > 0);
027    }
028    for (CountWrapper cw : Lists.newArrayList(cw1, cw2, cw1b, cw3, cw1d)) {
029      assertTrue(cw1c.compareTo(cw) < 0);
030    }
031    assertEquals(0, cw1.compareTo(cw1b));
032  }
033
034  @Test
035  public void testEquals() throws Exception {
036    assertTrue(cw1.equals(cw1b));
037    assertFalse(cw1.equals(cw1c));
038    assertFalse(cw1.equals(cw1d));
039    assertFalse(cw1.equals(cw2));
040    assertFalse(cw1.equals(cw3));
041  }
042}

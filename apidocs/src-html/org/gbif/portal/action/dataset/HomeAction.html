001package org.gbif.portal.action.dataset;
002
003import org.gbif.api.model.common.search.SearchResponse;
004import org.gbif.api.model.registry.search.DatasetSearchParameter;
005import org.gbif.api.model.registry.search.DatasetSearchRequest;
006import org.gbif.api.model.registry.search.DatasetSearchResult;
007import org.gbif.api.service.registry.DatasetSearchService;
008import org.gbif.api.vocabulary.DatasetType;
009import org.gbif.portal.action.BaseAction;
010
011import com.google.inject.Inject;
012import com.google.inject.Singleton;
013
014/**
015 * The home action uses the dataset search index to produce basic metrics.
016 */
017@Singleton
018public class HomeAction extends BaseAction {
019
020  private static final long serialVersionUID = 3791960822111920288L;
021
022  private long numOccurrenceDatasets;
023  private long numChecklistDatasets;
024  private long numMetadataDatasets;
025  private long numDatasets;
026
027  private final DatasetSearchService service;
028
029  @Inject
030  public HomeAction(DatasetSearchService service) {
031    this.service = service;
032  }
033
034  @Override
035  public String execute() {
036    numOccurrenceDatasets = count(DatasetType.OCCURRENCE);
037    numChecklistDatasets = count(DatasetType.CHECKLIST);
038    numMetadataDatasets = count(DatasetType.METADATA);
039    numDatasets = numOccurrenceDatasets + numChecklistDatasets + numMetadataDatasets;
040    return SUCCESS;
041  }
042
043  public long getNumChecklistDatasets() {
044    return numChecklistDatasets;
045  }
046
047  public long getNumDatasets() {
048    return numDatasets;
049  }
050
051  public long getNumMetadataDatasets() {
052    return numMetadataDatasets;
053  }
054
055  public long getNumOccurrenceDatasets() {
056    return numOccurrenceDatasets;
057  }
058
059  /**
060   * Uses the {@link DatasetSearchService} to find the count of records.
061   * 
062   * @param type To count
063   * @return The count, or 0 should the service be unresponsive
064   */
065  private long count(DatasetType type) {
066    DatasetSearchRequest dsr = new DatasetSearchRequest();
067    dsr.setLimit(0); // we don't need any
068    dsr.addParameter(DatasetSearchParameter.TYPE, type);
069    SearchResponse<DatasetSearchResult, DatasetSearchParameter> response = service.search(dsr);
070    return (response == null || response.getCount() == null) ? 0 : response.getCount(); // NPE safe
071  }
072}

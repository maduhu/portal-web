001package org.gbif.portal.action.dataset;
002
003import org.gbif.api.model.registry.Contact;
004import org.gbif.api.vocabulary.ContactType;
005import org.gbif.portal.action.node.NodeAction;
006
007import java.util.Collections;
008import java.util.List;
009
010import com.google.common.collect.Lists;
011import org.junit.Test;
012
013import static org.junit.Assert.assertEquals;
014
015public class NodeContactOrderTest {
016
017  int nextContactId = 1;
018
019  @Test
020  public void test() {
021    List<Contact> contacts = Lists.newArrayList(
022      build(ContactType.AUTHOR, "Pia", "Possible"),
023      build(ContactType.ADDITIONAL_DELEGATE, "Pia", "Possible"),
024      build(ContactType.NODE_STAFF, "Karl", "Orff"),
025      build(ContactType.NODE_STAFF, "Iggy", "Pop"),
026      build(ContactType.SYSTEM_ADMINISTRATOR, "Cip", "Munk"),
027      build(ContactType.NODE_MANAGER, "Dag", "Dig"),
028      build(ContactType.REGIONAL_NODE_REPRESENTATIVE, "Retro", "Mayer"),
029      build(null, null, null),
030      build(ContactType.HEAD_OF_DELEGATION, "Tip", "Top")
031      );
032
033    Collections.sort(contacts, new NodeAction.NodeContactOrder());
034
035    for (Contact c : contacts) {
036      System.out.println(c);
037    }
038
039    assertEquals((Integer) 9, contacts.get(0).getKey());
040    assertEquals((Integer) 7, contacts.get(1).getKey());
041    assertEquals((Integer) 6, contacts.get(2).getKey());
042    assertEquals((Integer) 4, contacts.get(3).getKey());
043    assertEquals((Integer) 3, contacts.get(4).getKey());
044    assertEquals((Integer) 1, contacts.get(5).getKey());
045    assertEquals((Integer) 5, contacts.get(6).getKey());
046    assertEquals((Integer) 2, contacts.get(7).getKey());
047    assertEquals((Integer) 8, contacts.get(8).getKey());
048  }
049
050
051  private Contact build(ContactType type, String first, String last) {
052    Contact c = new Contact();
053    c.setType(type);
054    c.setFirstName(first);
055    c.setLastName(last);
056    c.setKey(nextContactId++);
057    return c;
058  }
059}

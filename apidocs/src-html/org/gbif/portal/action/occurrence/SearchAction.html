073  @Inject

080  /*
081   * (non-Javadoc)
082   * @see org.gbif.portal.action.BaseSearchAction#execute()
083   */
084  @Override
085  public String execute() {
086
087    // read filter parameters in order to have them available even when the search wasn't executed.
088    readFilterParams();
089
090    table = new OccurrenceTable(request, getSearchRequest());
091
092    initSuggestions();
093
094    if (getSearchRequest().getOffset() > getMaxAllowedOffset()) {
095      addFieldError(
096        OFFSET_FIELD,
097        getText(MAXOFFSET_ERROR_KEY,
098          new String[] {Long.toString(getSearchRequest().getOffset()), getMaxAllowedOffset().toString()}));
099
100      return SUCCESS;
101    }
102
103    // process taxon/scientific-name suggestions
104    nameUsagesSuggestions = filtersActionHelper.processNameUsagesSuggestions(request);
105
106    // process dataset title suggestions
107    datasetsSuggestions = filtersActionHelper.processDatasetSuggestions(request);
108
109    // replace known name usages
110    filtersActionHelper.processNameUsageReplacements(searchRequest, nameUsagesSuggestions);
111    // replace known datasets
112    filtersActionHelper.processDatasetReplacements(searchRequest, datasetsSuggestions);
113
114    // Search is executed only if there aren't suggestions that need to be notified to the user
115    if (!hasSuggestions() && validateSearchParameters()) {
116      return executeSearch();
117    }
118    return SUCCESS;
119  }
120
121  /**
122   * Utility method that executes the search.
123   * Differs to the BaseSearchAction.execute() method in that this method doesn't execute the method
124   * BaseSearchAction.readFilterParams().
125   */
126  public String executeSearch() {
127    LOG.info("Search for [{}]", getQ());
128    // default query parameters
129    searchRequest.setQ(getQ());
130    // Turn off highlighting for empty query strings
131    searchRequest.setHighlight(!Strings.isNullOrEmpty(q));
132    // issues the search operation
133    searchResponse = searchService.search(searchRequest);
134
135    // Provide suggestions for catalog numbers and collector names
136    provideSuggestions();
137
138    LOG.debug("Search for [{}] returned {} results", getQ(), searchResponse.getCount());
139    return SUCCESS;
140  }

296  /**
297   * Validates if the download functionality should be shown.
298   */
299  public boolean showDownload() {
300    return searchResponse != null && searchResponse.getCount() > 0
301      && (getCfg().getMaxOccDowloadSize() < 0 || searchResponse.getCount() <= getCfg().getMaxOccDowloadSize())
302      && !hasErrors() && !hasSuggestions();
303  }

305  @Override
306  protected String translateFilterValue(OccurrenceSearchParameter param, String value) {
307    if (param == OccurrenceSearchParameter.GEOMETRY) {
308      return String.format(FiltersActionHelper.POLYGON_PATTERN, value);
309    }
310    return value;
311  }

313  /**
314   * Calculates the maximum offset allowed using the current limit parameter.
315   */
316  public Integer getMaxAllowedOffset() {
317    return getCfg().getMaxOccSearchOffset() - getSearchRequest().getLimit();
318  }

320  /**
321   * Initializes the suggestion objects with empty values.
322   */
323  private void initSuggestions() {
324    nameUsagesSuggestions = new SearchSuggestions<NameUsageSearchResult>();
325    datasetsSuggestions = new SearchSuggestions<DatasetSearchResult>();
326    collectorSuggestions = new SearchSuggestions<String>();
327    catalogNumberSuggestions = new SearchSuggestions<String>();
328    institutionCodeSuggestions = new SearchSuggestions<String>();
329    collectionCodeSuggestions = new SearchSuggestions<String>();
330  }
331
332  /**
333   * Checks if the parameter is in any lists of suggestions.
334   */
335  private boolean isSuggestion(String value) {
336    return nameUsagesSuggestions.getSuggestions().containsKey(value)
337      || datasetsSuggestions.getSuggestions().containsKey(value);
338  }
339
340  /**
341   * Provides post-search suggestions. The suggestions are provided only if the count of results id 0.
342   */
343  private void provideSuggestions() {
344    if (searchResponse.getCount() == 0) {
345      if (searchRequest.getParameters().containsKey(OccurrenceSearchParameter.COLLECTOR_NAME)) {
346        collectorSuggestions = filtersActionHelper.processCollectorSuggestions(request);
347      }
348      if (searchRequest.getParameters().containsKey(OccurrenceSearchParameter.CATALOG_NUMBER)) {
349        catalogNumberSuggestions = filtersActionHelper.processCatalogNumberSuggestions(request);
350      }
351      if (searchRequest.getParameters().containsKey(OccurrenceSearchParameter.COLLECTION_CODE)) {
352        collectionCodeSuggestions = filtersActionHelper.processCatalogNumberSuggestions(request);
353      }
354      if (searchRequest.getParameters().containsKey(OccurrenceSearchParameter.INSTITUTION_CODE)) {
355        institutionCodeSuggestions = filtersActionHelper.processCatalogNumberSuggestions(request);
356      }
357    }
358  }
359
360  /**
361   * Process the occurrence search parameters to validate if the values are correct.
362   * The list of errors is store in the "errors" field.
363   */
364  private boolean validateSearchParameters() {
365    validationErrors = filtersActionHelper.validateSearchParameters(request, OCC_VALIDATION_DISCARDED);
366    return validationErrors.isEmpty();
367  }

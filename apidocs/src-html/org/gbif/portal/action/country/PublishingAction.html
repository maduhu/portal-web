001package org.gbif.portal.action.country;
002
003import org.gbif.api.model.metrics.cube.OccurrenceCube;
004import org.gbif.api.model.metrics.cube.ReadBuilder;
005import org.gbif.api.service.checklistbank.DatasetMetricsService;
006import org.gbif.api.service.metrics.CubeService;
007import org.gbif.api.service.occurrence.OccurrenceCountryIndexService;
008import org.gbif.api.service.occurrence.OccurrenceDatasetIndexService;
009import org.gbif.api.service.registry.DatasetSearchService;
010import org.gbif.api.service.registry.DatasetService;
011import org.gbif.api.service.registry.NodeService;
012import org.gbif.api.service.registry.OrganizationService;
013
014import com.google.inject.Inject;
015import org.slf4j.Logger;
016import org.slf4j.LoggerFactory;
017
018public class PublishingAction extends CountryBaseAction {
019
020  private static Logger LOG = LoggerFactory.getLogger(PublishingAction.class);
021
022  private int otherCountries;
023  private long otherCountryRecords;
024  private double otherCountryPercentage;
025
026  @Inject
027  public PublishingAction(NodeService nodeService, CubeService cubeService,
028    OccurrenceDatasetIndexService datasetIndexService, OccurrenceCountryIndexService countryIndexService,
029    DatasetService datasetService, DatasetSearchService datasetSearchService,
030    DatasetMetricsService datasetMetricsService, OrganizationService organizationService) {
031    super(nodeService, cubeService, datasetIndexService, countryIndexService, datasetService, datasetSearchService,
032      datasetMetricsService, organizationService);
033  }
034
035  @Override
036  public String execute() throws Exception {
037    super.execute();
038
039    buildByMetrics(7, 7);
040
041    // calc other countries metrics
042    long inCountryRecords = cubeService.get(
043      new ReadBuilder().at(OccurrenceCube.HOST_COUNTRY, country).at(OccurrenceCube.COUNTRY, country)
044      );
045    otherCountryRecords = getBy().getOccurrenceRecords() - inCountryRecords;
046    otherCountryPercentage = 100d * otherCountryRecords / getBy().getOccurrenceRecords();
047    otherCountries = getBy().getCountries() - 1;
048
049    return SUCCESS;
050  }
051
052  public int getOtherCountries() {
053    return otherCountries;
054  }
055
056  public double getOtherCountryPercentage() {
057    return otherCountryPercentage;
058  }
059
060  public long getOtherCountryRecords() {
061    return otherCountryRecords;
062  }
063}

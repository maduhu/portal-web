070  /**
071   * From partial gathering date's year, month, day parts construct string in following format: Oct 7, 2006.
072   * Basically, only valid integers gets persisted to the index, and any invalid data will be flagged/logged.
073   * This method must ensure the values gets displayed exactly as they are persisted so that if there are errors
074   * they can be detected via the occurrence page also.
075   * 
076   * @return date string
077   */
078  protected String constructPartialGatheringDate(Integer year, Integer month, Integer day) {
079    StringBuffer st = new StringBuffer();
080
081    // start by adding month
082    String monthString = null;
083    if (month != null && month > 0) {
084
085      switch (month) {
086        case 1:
087          monthString = "Jan";
088          break;
089        case 2:
090          monthString = "Feb";
091          break;
092        case 3:
093          monthString = "Mar";
094          break;
095        case 4:
096          monthString = "Apr";
097          break;
098        case 5:
099          monthString = "May";
100          break;
101        case 6:
102          monthString = "Jun";
103          break;
104        case 7:
105          monthString = "Jul";
106          break;
107        case 8:
108          monthString = "Aug";
109          break;
110        case 9:
111          monthString = "Sep";
112          break;
113        case 10:
114          monthString = "Oct";
115          break;
116        case 11:
117          monthString = "Nov";
118          break;
119        case 12:
120          monthString = "Dec";
121          break;
122      }
123      st.append(monthString);
124
125      // add day to month separated by single whitespace, provided month has been added
126      if (day != null && day > 0) {
127        st.append(" ");
128        st.append(String.valueOf(day));
129      }
130    }
131
132    // add year
133    if (year != null && year > 0) {
134      // before adding year, write comma (",") if at least month has been appended to string date
135      if (!Strings.isNullOrEmpty(st.toString())) {
136        st.append(", ");
137      }
138      // now add year
139      st.append(String.valueOf(year));
140    }
141
142    return st.toString();
143  }
144
145  protected void loadDetail() {
146    if (id == null) {
147      throw new NotFoundException("No occurrence id given");
148    }
149    occ = occurrenceService.get(id);
150    if (occ == null) {
151      throw new NotFoundException("No occurrence found with id " + id);
152    }
153    // load dataset
154    dataset = datasetService.get(occ.getDatasetKey());
155    if (dataset == null) {
156      throw new ReferentialIntegrityException(Occurrence.class, id, "Missing dataset " + occ.getDatasetKey());
157    }
158    // load name usage nub object
159    if (occ.getTaxonKey() != null) {
160      nub = usageService.get(occ.getTaxonKey(), getLocale());
161      if (nub == null) {
162        throw new ReferentialIntegrityException(Occurrence.class, id, "Missing nub usage " + occ.getTaxonKey());
163      }
164    }
165    // TODO: load metrics for occurrence once implemented
166    metrics = null;
167    // construct partial gathering date if occurrenceDate was empty and there is at least a year or month
168    if (occ.getEventDate() == null && (occ.getYear() != null || occ.getMonth() != null)) {
169      partialGatheringDate =
170        constructPartialGatheringDate(occ.getYear(), occ.getMonth(), occ.getDay());
171    }
172  }

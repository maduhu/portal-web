24   * Landing page for non country nodes.
25   */
26  public class ListAction extends BaseAction {
27    public static final UUID GBIF_TEMP_NODE_KEY = UUID.fromString("02c40d2a-1cba-4633-90b7-e36e5e97aba8");
28  
29    private List<Node> voting = Lists.newArrayList();
30    private List<Node> associate = Lists.newArrayList();
31    private List<Node> other = Lists.newArrayList();
32  
33    @Inject
34    private NodeService nodeService;
35  
36    @Override
37    public String execute() throws Exception {
38      PagingResponse<Node> resp = nodeService.list(new PagingRequest(0, 1000));
39      List<Node> sorted = FluentIterable.from(resp.getResults())
40        // sort alphabetically
41        .filter(new Predicate<Node>() {
42          public boolean apply(@Nullable Node n) {
43            return !GBIF_TEMP_NODE_KEY.equals(n.getKey());
44          }
45        })
46        .toSortedList(Ordering.natural().nullsLast().onResultOf(new Function<Node, Country>() {
47          @Nullable
48          @Override
49          public Country apply(@Nullable Node n) {
50            return n == null ? null : n.getCountry();
51          }
52        }));
53  
54      for (Node n: sorted) {
55        if (GBIF_TEMP_NODE_KEY.equals(n.getKey())) {
56          continue;
57        }
58        if (NodeType.COUNTRY == n.getType()) {
59          if (ParticipationStatus.VOTING == n.getParticipationStatus()) {
60            voting.add(n);
61          } else if (ParticipationStatus.ASSOCIATE == n.getParticipationStatus()) {
62            associate.add(n);
63          }
64        } else if (ParticipationStatus.ASSOCIATE == n.getParticipationStatus()){
65          other.add(n);
66        }
67      }
68  
69      // sort others according to node title, in case of taiwan use the country title
70      other = FluentIterable.from(other)
71        // sort alphabetically
72        .toSortedList(Ordering.natural().nullsLast().onResultOf(new Function<Node, String>() {
73          @Nullable
74          @Override
75          public String apply(@Nullable Node n) {
76            // yet another taiwan country hack: http://dev.gbif.org/issues/browse/PF-1031
77            return n == null ? null : (n.getCountry()==Country.TAIWAN ? Country.TAIWAN.getTitle() : n.getTitle()) ;
78          }
79        }));
80  
81      return SUCCESS;
82    }
83

85      return voting;
86    }
87  
88    public List<Node> getAssociate() {
89      return associate;
90    }
91  
92    public List<Node> getOther() {
93      return other;
94    }

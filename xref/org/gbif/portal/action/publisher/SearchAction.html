20  @SuppressWarnings("serial")

23    private static final int PAGE_SIZE = 20;

25    private final OrganizationService organizationService;
26    private final NodeService nodeService;
27    
28    private long offset = 0;
29    private String q;
30    private PagingResponse<Organization> page;
31    // required to show the titles
32    private Map<UUID,Node> nodeIndex = Maps.newHashMap();
33  
34    @Inject
35    public SearchAction(OrganizationService organizationService, NodeService nodeService) {
36      this.organizationService = organizationService;
37      this.nodeService = nodeService;
38    }
39  
40    @Override
41    public String execute() throws Exception {
42      // TODO: Support ordering
43      // http://dev.gbif.org/issues/browse/REG-415
44      page = organizationService.search(q, new PagingRequest(offset, PAGE_SIZE));
45      
46      for (Organization o : page.getResults()) {
47        if (!nodeIndex.containsKey(o.getEndorsingNodeKey())) {
48          nodeIndex.put(o.getEndorsingNodeKey(), nodeService.get(o.getEndorsingNodeKey()));  
49        }
50      }
51      
52      LOG.info("Searching for publisher with query[{}] produced {} results", q, page.getCount());
53      return SUCCESS;
54    }
55

014  /**
015   * @param name            the q name to search for
016   * @param searchNub       true if only the nub is being searched, false to search all
017   * @param expectedUsageId the list of expected name usage ids in the result, starting with first entry
018   */
019  private void assertNameSearch(String name, boolean searchNub, Integer expectedNumResults, int... expectedUsageId) {
020    LOG.debug("Asserting name search '{}'", name);
021    getUrl(getPortalUrl("species"));
022    // Find input form and enter seach text
023    WebElement input = driver.findElement(By.id("q"));
024    input.sendKeys(name);
025    LOG.debug("SEARCH FORM VALUE: {}", input.getAttribute("value"));
026    driver.findElement(By.id("submitSearch")).click();
027
028    // do we want a search across all checklists?
029    if (!searchNub) {
030      // TODO: doesnt seem to work. Maybe some javascript fix needed?
031      driver.findElement(By.id("facetfilterDATASET_KEY")).findElement(By.cssSelector("a")).click();
032    }
033
034    // main content div on page
035    WebElement content = driver.findElement(By.id("content"));
036
037    // assert number of hits
038    if (expectedNumResults != null) {
039      assertEquals("Expected number of shown search results wrong", expectedNumResults,
040        (Integer) content.findElements(By.cssSelector("div.result")).size());
041    }
042
043    // assert exact results
044    LinkedList<WebElement> links = new LinkedList(content.findElements(By.cssSelector("div.result h2 a")));
045    LOG.debug("{} search results found", links.size());
046    for (int uid : expectedUsageId) {
047      int speciesKey = parseIdLink(links.remove().getAttribute("href"));
048      assertEquals("Expected result usage id different", uid, speciesKey);
049    }
050  }

069  @Test
070  public void testIdPattern() {
071    assertEquals((Integer) 2685464, parseIdLink("http://staging.gbif.org:8080/portal-web-dynamic/species/2685464"));

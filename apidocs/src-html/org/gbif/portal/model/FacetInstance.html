001package org.gbif.portal.model;
002
003import org.gbif.api.model.common.search.Facet;
004
005
006/**
007 * Extended facet count with an additional optional title field
008 * that can be used to display other values than the facet being used in solr.
009 * This allows facets to be keys only while the title can be used to show real names.
010 */
011public class FacetInstance extends Facet.Count {
012
013  private String title;
014
015  public FacetInstance() {
016  }
017
018  public FacetInstance(Facet.Count c) {
019    super(c.getName(), c.getCount());
020  }
021
022  public FacetInstance(String name) {
023    super(name, null);
024  }
025
026  public FacetInstance(String name, String title, Long count) {
027    super(name, count);
028    this.title = title;
029  }
030
031  /**
032   * @return the title to display, defaulting to name in case its null.
033   */
034  public String getTitle() {
035    return title == null ? getName() : title;
036  }
037
038  public void setTitle(String title) {
039    this.title = title;
040  }
041}

001package org.gbif.portal.struts;
002
003import java.util.Collection;
004import java.util.List;
005import java.util.Map;
006
007import com.opensymphony.xwork2.ActionInvocation;
008import com.opensymphony.xwork2.Result;
009import com.opensymphony.xwork2.ValidationAware;
010import com.opensymphony.xwork2.interceptor.MethodFilterInterceptor;
011import org.apache.struts2.dispatcher.ServletRedirectResult;
012
013/**
014 * An Interceptor to preserve an actions ValidationAware messages across a
015 * redirect result.
016 * It makes the assumption that you always want to preserve messages across a
017 * redirect and restore them to the next action if they exist.
018 * The way this works is it looks at the result type after a action has executed
019 * and if the result was a redirect (ServletRedirectResult) or a redirectAction
020 * (ServletActionRedirectResult) and there were any errors, messages, or
021 * fieldErrors they are stored in the session. Before the next action executes
022 * it will check if there are any messages stored in the session and add them to
023 * the next action.
024 *
025 * @see <a href="http://glindholm.wordpress.com/2008/07/02/preserving-messages-across-a-redirect-in-struts-2/">Preserving
026 *      Messages Across a Redirect in Struts
027 *      2</a>
028 */
029public class RedirectMessageInterceptor extends MethodFilterInterceptor {
030
031  private static final long serialVersionUID = -1847557437429753540L;
032
033  public static final String FIELD_ERRORS_KEY = "RedirectMessageInterceptor_FieldErrors";
034  public static final String ACTION_ERRORS_KEY = "RedirectMessageInterceptor_ActionErrors";
035  public static final String ACTION_MESSAGES_KEY = "RedirectMessageInterceptor_ActionMessages";
036
037  @Override
038  public String doIntercept(final ActionInvocation invocation) throws Exception {
039    final Object action = invocation.getAction();
040    if (action instanceof ValidationAware) {
041      before(invocation, (ValidationAware) action);
042    }
043
044    final String result = invocation.invoke();
045
046    if (action instanceof ValidationAware) {
047      after(invocation, (ValidationAware) action);
048    }
049    return result;
050  }
051
052  /**
053   * Retrieve the errors and messages from the session and add them to the
054   * action.
055   */
056  protected void before(final ActionInvocation invocation, final ValidationAware validationAware) throws Exception {
057    @SuppressWarnings("unchecked")
058    final Map<String, ?> session = invocation.getInvocationContext().getSession();
059
060    @SuppressWarnings("unchecked")
061    final Collection<String> actionErrors = (Collection) session.remove(ACTION_ERRORS_KEY);
062    if (actionErrors != null && !actionErrors.isEmpty()) {
063      for (final String error : actionErrors) {
064        validationAware.addActionError(error);
065      }
066    }
067
068    @SuppressWarnings("unchecked")
069    final Collection<String> actionMessages = (Collection) session.remove(ACTION_MESSAGES_KEY);
070    if (actionMessages != null && !actionMessages.isEmpty()) {
071      for (final String message : actionMessages) {
072        validationAware.addActionMessage(message);
073      }
074    }
075
076    @SuppressWarnings("unchecked")
077    final Map<String, List<String>> fieldErrors = (Map) session.remove(FIELD_ERRORS_KEY);
078    if (fieldErrors != null && !fieldErrors.isEmpty()) {
079      for (final Map.Entry<String, List<String>> fieldError : fieldErrors.entrySet()) {
080        for (final String message : fieldError.getValue()) {
081          validationAware.addFieldError(fieldError.getKey(), message);
082        }
083      }
084    }
085  }
086
087  /**
088   * If the result is a redirect then store error and messages in the session.
089   */
090  protected void after(final ActionInvocation invocation, final ValidationAware validationAware) throws Exception {
091    final Result result = invocation.getResult();
092
093    if (result != null && result instanceof ServletRedirectResult) {
094      final Map<String, Object> session = invocation.getInvocationContext().getSession();
095
096      final Collection<String> actionErrors = validationAware.getActionErrors();
097      if (actionErrors != null && !actionErrors.isEmpty()) {
098        session.put(ACTION_ERRORS_KEY, actionErrors);
099      }
100
101      final Collection<String> actionMessages = validationAware.getActionMessages();
102      if (actionMessages != null && !actionMessages.isEmpty()) {
103        session.put(ACTION_MESSAGES_KEY, actionMessages);
104      }
105
106      final Map<String, List<String>> fieldErrors = validationAware.getFieldErrors();
107      if (fieldErrors != null && !fieldErrors.isEmpty()) {
108        session.put(FIELD_ERRORS_KEY, fieldErrors);
109      }
110    }
111  }
112}

001package org.gbif.portal.config;
002
003import org.gbif.api.vocabulary.Continent;
004import org.gbif.api.vocabulary.Country;
005
006import java.util.Iterator;
007import java.util.List;
008import java.util.Map;
009import java.util.Set;
010
011import com.google.common.base.CharMatcher;
012import com.google.common.base.Splitter;
013import com.google.common.collect.FluentIterable;
014import com.google.common.collect.ImmutableList;
015import com.google.common.collect.Lists;
016import com.google.common.collect.Maps;
017import com.google.common.collect.Ordering;
018import com.google.common.collect.Sets;
019import org.slf4j.Logger;
020import org.slf4j.LoggerFactory;
021
022/**
023 * A utility class to map countries to specific drupal tag ids used in drupal news feeds for example.
024 */
025public class ContinentCountryMap {
026  private static final Logger LOG = LoggerFactory.getLogger(ContinentCountryMap.class);
027  private static final Splitter ROW_SPLITTER = Splitter.on(CharMatcher.WHITESPACE).trimResults().omitEmptyStrings();
028  private final Map<Continent, List<Country>> continents = Maps.newHashMap();
029  private final List<Continent> continentsSortedAlphabetically;
030
031  /**
032   * @param lines key=continent enum value, value=country iso code
033   */
034  public ContinentCountryMap(Set<String> lines) {
035    Map<Continent, List<Country>> cm = Maps.newHashMap();
036    for (String line : lines) {
037      Iterator<String> cols = ROW_SPLITTER.split(line).iterator();
038
039      Continent continent = Continent.fromString(cols.next());
040      String iso = cols.next();
041      Country c = Country.fromIsoCode(iso);
042      if (c == null) {
043        LOG.debug("Ignore unknown country {}", iso);
044        continue;
045      }
046      if (!cm.containsKey(continent)) {
047        cm.put(continent, Lists.<Country>newArrayList());
048      }
049      cm.get(continent).add(c);
050    }
051    // for checking if all countries are covered!
052    Set<Country> countries = Sets.newHashSet();
053    // sort countries into immutable list
054    for (Continent c : cm.keySet()) {
055      continents.put(c, FluentIterable.from(cm.get(c)).toSortedList(Ordering.natural()));
056      countries.addAll(cm.get(c));
057    }
058    // check if all countries are covered!
059    for (Country c : Country.values()) {
060      if (c.isOfficial()) {
061        if (!countries.contains(c)) {
062          LOG.error("Country {} missing from continent country map!", c);
063        }
064      }
065    }
066
067    // continent enum is not sorted
068    continentsSortedAlphabetically = ImmutableList.copyOf(Ordering.usingToString().sortedCopy(continents.keySet()));
069  }
070
071  public List<Continent> alphabeticallyOrderedContinents() {
072    return continentsSortedAlphabetically;
073  }
074
075  public List<Country> listCountries(Continent c) {
076    return continents.get(c);
077  }
078}

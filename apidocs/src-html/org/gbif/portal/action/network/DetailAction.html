001package org.gbif.portal.action.network;
002
003import org.gbif.api.model.common.paging.PagingRequest;
004import org.gbif.api.model.common.paging.PagingResponse;
005import org.gbif.api.model.registry.Dataset;
006import org.gbif.api.model.registry.Network;
007import org.gbif.api.service.checklistbank.DatasetMetricsService;
008import org.gbif.api.service.metrics.CubeService;
009import org.gbif.api.service.registry.NetworkService;
010import org.gbif.api.service.registry.OrganizationService;
011import org.gbif.portal.action.member.MemberBaseAction;
012import org.gbif.portal.action.member.MemberType;
013
014import com.google.inject.Inject;
015
016public class DetailAction extends MemberBaseAction<Network> {
017
018  private final NetworkService networkService;
019  private PagingResponse<Dataset> page;
020  private long offset = 0;
021
022  @Inject
023  public DetailAction(NetworkService networkService, OrganizationService organizationService, CubeService cubeService,
024    DatasetMetricsService datasetMetricsService) {
025    super(MemberType.NETWORK, networkService, cubeService, datasetMetricsService, organizationService);
026    this.networkService = networkService;
027  }
028
029  @Override
030  public String execute() throws Exception {
031    super.execute();
032
033    // load first 10 datasets
034    page = networkService.listConstituents(id, new PagingRequest(0, 10));
035    super.loadCountWrappedDatasets(page);
036
037    return SUCCESS;
038  }
039
040  public String datasets() throws Exception {
041    super.execute();
042    page = networkService.listConstituents(id, new PagingRequest(offset, 25));
043    return SUCCESS;
044  }
045
046  public PagingResponse<Dataset> getPage() {
047    return page;
048  }
049
050  public void setOffset(long offset) {
051    if (offset >= 0) {
052      this.offset = offset;
053    }
054  }
055}

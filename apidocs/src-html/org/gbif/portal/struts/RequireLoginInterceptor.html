001package org.gbif.portal.struts;
002
003import org.gbif.api.model.common.User;
004import org.gbif.portal.config.Constants;
005
006import java.util.Map;
007
008import com.opensymphony.xwork2.ActionInvocation;
009import com.opensymphony.xwork2.interceptor.AbstractInterceptor;
010
011/**
012 * An Interceptor that makes sure a user is currently logged in and returns a notLoggedIn otherwise.
013 */
014public class RequireLoginInterceptor extends AbstractInterceptor {
015
016  @Override
017  public String intercept(final ActionInvocation invocation) throws Exception {
018    final Map session = invocation.getInvocationContext().getSession();
019    final User user = (User) session.get(Constants.SESSION_USER);
020    if (user != null) {
021      return invocation.invoke();
022    }
023    return Constants.RESULT_LOGIN_REQUIRED;
024  }
025
026}

001package org.gbif.portal.action.species;
002
003import org.gbif.api.model.checklistbank.NameUsageComponent;
004import org.gbif.api.model.common.paging.PagingRequest;
005import org.gbif.api.model.common.paging.PagingResponse;
006import org.gbif.api.service.checklistbank.NameUsageComponentService;
007
008import java.util.HashSet;
009import java.util.Set;
010import java.util.UUID;
011
012public class SeeMoreAction<T extends NameUsageComponent> extends UsageBaseAction {
013
014  private PagingResponse<T> page;
015  private long offset = 0;
016  private final NameUsageComponentService<T> service;
017
018  public SeeMoreAction(NameUsageComponentService<T> service) {
019    this.service = service;
020  }
021
022  @Override
023  public String execute() {
024    loadUsage();
025
026    PagingRequest p = new PagingRequest(offset, 25);
027    page = service.listByUsage(id, p);
028
029    // load checklist lookup map if its a nub usage
030    if (usage.isNub()) {
031      Set<UUID> cids = new HashSet<UUID>();
032      for (T comp : page.getResults()) {
033        cids.add(comp.getDatasetKey());
034      }
035      loadChecklists(cids);
036    }
037
038    return SUCCESS;
039  }
040
041  public void setOffset(long offset) {
042    if (offset >= 0) {
043      this.offset = offset;
044    }
045  }
046
047  public PagingResponse<T> getPage() {
048    return page;
049  }
050}

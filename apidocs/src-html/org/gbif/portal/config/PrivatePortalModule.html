001package org.gbif.portal.config;
002
003import org.gbif.utils.file.FileUtils;
004import org.gbif.ws.client.filter.HttpGbifAuthFilter;
005import org.gbif.ws.client.guice.GbifApplicationAuthModule;
006import org.gbif.ws.security.GbifAppAuthService;
007
008import java.io.IOException;
009import java.util.Map;
010import java.util.Properties;
011import java.util.Set;
012
013import com.google.inject.Inject;
014import com.google.inject.PrivateModule;
015import com.google.inject.Provides;
016import com.google.inject.Scopes;
017import com.google.inject.Singleton;
018import com.google.inject.name.Names;
019import com.sun.jersey.api.client.filter.ClientFilter;
020
021public class PrivatePortalModule extends PrivateModule{
022  private final Properties properties;
023  private final String appKey;
024  private final String appSecret;
025
026  public PrivatePortalModule(Properties properties) {
027    this.properties = properties;
028    this.appKey = properties.getProperty(GbifApplicationAuthModule.PROPERTY_APP_KEY);
029    this.appSecret = properties.getProperty(GbifApplicationAuthModule.PROPERTY_APP_SECRET);
030  }
031
032  @Override
033  protected void configure() {
034    Names.bindProperties(binder(), properties);
035
036    bind(SessionAuthProvider.class).in(Scopes.SINGLETON);
037
038    expose(ClientFilter.class);
039    expose(DrupalCountryTagMap.class);
040    expose(ContinentCountryMap.class);
041  }
042
043  @Provides
044  @Singleton
045  public GbifAppAuthService provideGbifAppAuthService() {
046    return new GbifAppAuthService(appKey, appSecret);
047  }
048
049  @Provides
050  @Singleton
051  @Inject
052  public ClientFilter provideSessionAuthFilter(GbifAppAuthService authService, SessionAuthProvider sessionAuthProvider) {
053    return new HttpGbifAuthFilter(appKey, authService, sessionAuthProvider);
054  }
055
056  @Provides
057  @Singleton
058  public DrupalCountryTagMap provideDrupalCountryTagMap() throws IOException {
059    Map<String,String> rawMap = FileUtils.streamToMap(FileUtils.classpathStream("drupal_country_tags.txt"), 0, 1, true);
060    return new DrupalCountryTagMap(rawMap);
061  }
062
063  @Provides
064  @Singleton
065  public ContinentCountryMap provideContinentCountryMap() throws IOException {
066    Set<String> raw = FileUtils.streamToSet(FileUtils.classpathStream("country_by_continent.txt"));
067    return new ContinentCountryMap(raw);
068  }
069}

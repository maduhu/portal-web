55    /**
56     * Ordering based on contact types with head of delegation and node managers coming first.
57     */
58    public static class NodeContactOrder implements Comparator<Contact> {
59  
60      @Override
61      public int compare(Contact o1, Contact o2) {
62        if (o1.getType() != null && o2.getType() != null && o1.getType() != o2.getType()) {
63          return contactTypeOrder.indexOf(o1.getType()) - contactTypeOrder.indexOf(o2.getType());
64        }
65        return ComparisonChain.start()
66          .compare(o1.isPrimary(), o2.isPrimary(), Ordering.natural().nullsLast())
67          .compare(o1.getType(), o2.getType(), Ordering.natural().nullsLast())
68          .compare(o1.getFirstName(), o2.getFirstName(), Ordering.natural().nullsLast())
69          .compare(o1.getLastName(), o2.getLastName(), Ordering.natural().nullsLast())
70          .compare(o1.getKey(), o2.getKey())
71          .result();
72      }
73    }

001package org.gbif.portal.action.dataset;
002
003import org.gbif.api.model.registry.search.DatasetSearchResult;
004import org.gbif.portal.action.ActionTestUtil;
005
006import java.util.LinkedList;
007import java.util.List;
008
009import com.google.inject.Injector;
010import org.junit.Before;
011import org.junit.BeforeClass;
012import org.junit.Test;
013
014import static junit.framework.Assert.assertEquals;
015import static junit.framework.Assert.assertFalse;
016import static junit.framework.Assert.assertTrue;
017import static org.junit.Assert.assertNotNull;
018
019public class SearchActionTest {
020
021  private static SearchAction sa;
022  DatasetSearchResult result;
023
024  @BeforeClass
025  public static void first() {
026    // initiate SearchAction
027    Injector injector = ActionTestUtil.initTestInjector();
028    sa = injector.getInstance(org.gbif.portal.action.dataset.SearchAction.class);
029  }
030
031  @Before
032  public void setup() {
033    // populate DatasetSearchResult without any highlighted fields
034    result = populateTestDatasetSearchResult();
035  }
036
037  /**
038   * The registry search module does many things at startup, including for example
039   * pinging SOLR. This test simply ensures that this process works.
040   */
041  @Test
042  public void test() {
043    assertNotNull(sa);
044  }
045
046  @Test
047  public void testAddMissingHighlighting() {
048    assertEquals("<em class=\"gbifHl\">pon</em>taurus", sa.addMissingHighlighting("pontaurus", "pon"));
049    assertEquals("<em class=\"gbifHl\">pon</em>taurus <em class=\"gbifHl\">pon</em>TAURUS",
050      sa.addMissingHighlighting("pontaurus PONTAURUS", "pon"));
051    assertEquals("Schulhof Gymnasium Hürth Bonn<em class=\"gbifHl\">strasse</em>",
052      sa.addMissingHighlighting("Schulhof Gymnasium Hürth Bonnstrasse", "straße"));
053    assertEquals(
054      "<em class=\"gbifHl\">FADA Ephemeroptera: World checklist of freshwater Ephemeroptera species in the Catalogue of Life</em>",
055      sa.addMissingHighlighting(
056        "FADA Ephemeroptera: World checklist of freshwater Ephemeroptera species in the Catalogue of Life",
057        "FADA Ephemeroptera: World checklist of freshwater Ephemeroptera species in the Catalogue of Life"));
058  }
059
060  @Test
061  public void testAddMissingHighlightingNotApplied() {
062    assertEquals("<em class=\"gbifHl\">pon</em>taurus",
063      sa.addMissingHighlighting("<em class=\"gbifHl\">pon</em>taurus", "pon"));
064    assertEquals("<em class=\"gbifHl\">pon</em>taurus",
065      sa.addMissingHighlighting("<em class=\"gbifHl\">pon</em>taurus", ""));
066    assertEquals("", sa.addMissingHighlighting("", "pon"));
067  }
068
069  @Test
070  public void testFoldToAscii() throws Exception {
071    assertEquals("strasse", sa.foldToAscii("straße"));
072    assertEquals("strasse Imagenes", sa.foldToAscii("straße Imágenes"));
073  }
074
075  @Test
076  public void testIsFullTextMatchOnly() {
077    // there was no highlighting in any of the test object's fields, thereby it can be inferred that the match must have
078// happened on full_text
079    assertTrue(sa.isFullTextMatchOnly(result, "Pon"));
080  }
081
082  @Test
083  public void testIsFullTextMatchOnlyFalse() {
084    // override title with one that contains some (solr) highlighting
085    result.setTitle("<em class=\"gbifHl\">Pon&lt;/em>Taurus collection");
086    // there was no highlighting in any of the above fields, so infer match must have happened on full_text
087    assertFalse(sa.isFullTextMatchOnly(result, "Pon"));
088  }
089
090  @Test
091  public void testIsFullTextMatchOnlyForEmptyQueryText() {
092    // there was no query text, and thus no match can be reported
093    assertFalse(sa.isFullTextMatchOnly(result, ""));
094  }
095
096  /**
097   * Create a test DatasetSearchResult populated without any (solr) highlighting.
098   * 
099   * @return populated test object
100   */
101  private DatasetSearchResult populateTestDatasetSearchResult() {
102    DatasetSearchResult result = new DatasetSearchResult();
103    result.setTitle("Title");
104    result.setDescription("Desc");
105    List<String> keywords = new LinkedList<String>();
106    keywords.add("k1");
107    result.setKeywords(keywords);
108    result.setOwningOrganizationTitle("Owner");
109    result.setHostingOrganizationTitle("Host");
110    return result;
111  }
112}

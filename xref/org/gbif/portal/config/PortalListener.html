9   package org.gbif.portal.config;
10  
11  import com.google.inject.Guice;
12  import com.google.inject.Injector;
13  import com.google.inject.Provides;
14  import com.google.inject.Singleton;
15  import com.google.inject.servlet.GuiceServletContextListener;
16  import com.google.inject.servlet.ServletModule;
17  import com.google.inject.struts2.Struts2GuicePluginModule;
18  import org.apache.bval.guice.ValidationModule;
19  import org.apache.struts2.dispatcher.ng.filter.StrutsExecuteFilter;
20  import org.apache.struts2.dispatcher.ng.filter.StrutsPrepareFilter;
21  import org.apache.struts2.sitemesh.FreemarkerPageFilter;
22  import org.slf4j.Logger;
23  import org.slf4j.LoggerFactory;
24  
25  /**
26   * Setting up filter and servlets in addition to the ones in web.xml.
27   */
28  public class PortalListener extends GuiceServletContextListener {
29    private static Logger LOG = LoggerFactory.getLogger(PortalListener.class);
30    private final static Config cfg = Config.buildFromProperties();
31  
32    private final ServletModule sm = new ServletModule() {
33  
34      @Provides
35      @Singleton
36      public Config provideConfig(){
37        LOG.info("Configuring Guice with server name {}", cfg.getServerName());
38        return cfg;
39      }
40  
41      @Override
42      protected void configureServlets() {
43        // Struts2
44        bind(StrutsPrepareFilter.class).in(Singleton.class);
45        bind(FreemarkerPageFilter.class).in(Singleton.class);
46        bind(StrutsExecuteFilter.class).in(Singleton.class);
47  
48        // Struts2
49        filter("/*").through(StrutsPrepareFilter.class);
50        filter("/*").through(FreemarkerPageFilter.class);
51        filter("/*").through(StrutsExecuteFilter.class);
52      }
53    };
54    
55  
56    @Override
57    public Injector getInjector() {
58      return Guice.createInjector(new Struts2GuicePluginModule(), sm, new PortalModule(), new ValidationModule());
59    }
60  }

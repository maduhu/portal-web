001package org.gbif.portal.action.member;
002
003import org.gbif.api.model.checklistbank.DatasetMetrics;
004import org.gbif.api.model.common.paging.PagingResponse;
005import org.gbif.api.model.metrics.cube.OccurrenceCube;
006import org.gbif.api.model.metrics.cube.ReadBuilder;
007import org.gbif.api.model.registry.Dataset;
008import org.gbif.api.model.registry.NetworkEntity;
009import org.gbif.api.model.registry.Organization;
010import org.gbif.api.model.registry.Tag;
011import org.gbif.api.model.registry.Taggable;
012import org.gbif.api.service.checklistbank.DatasetMetricsService;
013import org.gbif.api.service.metrics.CubeService;
014import org.gbif.api.service.registry.NetworkEntityService;
015import org.gbif.api.service.registry.OrganizationService;
016import org.gbif.api.vocabulary.DatasetType;
017import org.gbif.portal.exception.NotFoundException;
018import org.gbif.portal.model.CountWrapper;
019
020import java.util.List;
021import java.util.Map;
022import java.util.Set;
023import java.util.UUID;
024
025import com.google.common.base.Strings;
026import com.google.common.collect.Lists;
027import com.google.common.collect.Maps;
028import com.google.common.collect.Sets;
029import org.slf4j.Logger;
030import org.slf4j.LoggerFactory;
031
032public class MemberBaseAction<T extends NetworkEntity & Taggable> extends org.gbif.portal.action.BaseAction {
033
034  protected CubeService cubeService;
035  protected DatasetMetricsService datasetMetricsService;
036  protected OrganizationService organizationService;
037  protected List<CountWrapper<Dataset>> datasets = Lists.newArrayList();
038  protected UUID id;
039  protected T member;
040  private static final Logger LOG = LoggerFactory.getLogger(MemberBaseAction.class);
041  private final NetworkEntityService<T> memberService;
042  private final MemberType type;
043  private List<String> keywords;
044  private Map<UUID, Organization> orgMap = Maps.newHashMap();
045
046  protected MemberBaseAction(MemberType type, NetworkEntityService<T> memberService,
047    CubeService cubeService, DatasetMetricsService datasetMetricsService, OrganizationService organizationService) {
048    this.memberService = memberService;
049    this.type = type;
050    this.cubeService = cubeService;
051    this.datasetMetricsService = datasetMetricsService;
052    this.organizationService = organizationService;
053  }
054
055  @Override
056  public String execute() throws Exception {
057    loadDetail();
058    return SUCCESS;
059  }
060
061  public UUID getId() {
062    return id;
063  }
064
065  /**
066   * The member's list of lower cased, plain string keywords derived from public tags without a namespace.
067   * This method loads keywords lazily.
068   * @return member's list of keywords
069   */
070  public List<String> getKeywords() {
071    if (keywords == null) {
072      // lazy load keywords
073      Set<String> kws = Sets.newTreeSet();
074      for (Tag t : member.getTags()) {
075        if (!Strings.isNullOrEmpty(t.getValue())) {
076          kws.add(t.getValue().trim().toLowerCase());
077        }
078      }
079      keywords = Lists.newArrayList(kws);
080    }
081    return keywords;
082  }
083
084  public T getMember() {
085    return member;
086  }
087
088  public MemberType getType() {
089    return type;
090  }
091
092  public void setId(String id) {
093    try {
094      this.id = UUID.fromString(id);
095    } catch (IllegalArgumentException e) {
096      this.id = null;
097    }
098  }
099
100  protected void loadDetail() {
101    LOG.debug("Getting detail for member key {}", id);
102    member = memberService.get(id);
103    if (member == null) {
104      throw new NotFoundException("No member found with key " + id);
105    }
106  }
107
108  /**
109   * Loads list of CountWrapper Datasets, that is a list of datasets with a record count and a geo reference count
110   * added to each one.
111   *
112   * @param datasetPage page of Datasets
113   */
114  protected void loadCountWrappedDatasets(PagingResponse<Dataset> datasetPage) {
115    for (Dataset d : datasetPage.getResults()) {
116      final long dsCnt;
117      final long dsGeoCnt;
118      if (DatasetType.OCCURRENCE == d.getType()) {
119        dsCnt = cubeService.get(new ReadBuilder().at(OccurrenceCube.DATASET_KEY, d.getKey()));
120        dsGeoCnt = cubeService.get(new ReadBuilder().at(OccurrenceCube.DATASET_KEY, d.getKey()).at(
121          OccurrenceCube.IS_GEOREFERENCED, true));
122
123      } else if (DatasetType.CHECKLIST == d.getType()) {
124        DatasetMetrics metric = datasetMetricsService.get(d.getKey());
125        dsCnt = (metric == null) ? 0 : metric.getUsagesCount();
126        dsGeoCnt = 0;
127      } else {
128        dsCnt = 0;
129        dsGeoCnt = 0;
130      }
131
132      datasets.add(new CountWrapper<Dataset>(d, dsCnt, dsGeoCnt));
133    }
134  }
135
136  public List<CountWrapper<Dataset>> getDatasets() {
137    return datasets;
138  }
139
140  /**
141   * Utility method to access node infos using a small map cache.
142   * Used in templates to show the publisher for datasets for example.
143   */
144  public Organization getOrganization(UUID key) {
145    if (orgMap.containsKey(key)) {
146      return orgMap.get(key);
147    }
148    Organization o = organizationService.get(key);
149    orgMap.put(key, o);
150    return o;
151  }
152}

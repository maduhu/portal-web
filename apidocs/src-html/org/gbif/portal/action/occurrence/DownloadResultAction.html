001package org.gbif.portal.action.occurrence;
002
003import org.gbif.api.model.occurrence.Download;
004import org.gbif.api.service.registry.OccurrenceDownloadService;
005import org.gbif.portal.action.BaseAction;
006import org.gbif.utils.file.FileUtils;
007
008import java.util.Set;
009
010import com.google.inject.Inject;
011import org.slf4j.Logger;
012import org.slf4j.LoggerFactory;
013
014import static org.gbif.portal.action.occurrence.DownloadsActionUtils.DOWNLOAD_EXIST_ERR_KEY;
015import static org.gbif.portal.action.occurrence.DownloadsActionUtils.DOWNLOAD_NULL_ERR_KEY;
016
017public class DownloadResultAction extends BaseAction {
018
019  private static final long serialVersionUID = -976893753100950798L;
020  private static final String TITLE_READY_KEY = "download.title.ready";
021  private static final String TITLE_STARTED_KEY = "download.title.started";
022  private static final String TITLE_ERROR_KEY = "download.title.error";
023  private static final String DOWNLOAD_KEY_PARAM = "key";
024  private static final Logger LOG = LoggerFactory.getLogger(DownloadResultAction.class);
025
026
027  @Inject
028  private OccurrenceDownloadService downloadService;
029
030  private String key;
031
032  private Download download;
033
034
035  @Override
036  public String execute() {
037    if (key != null) {
038      download = downloadService.get(key);
039      if (download == null || !download.getRequest().getCreator().equals(getCurrentUser().getUserName())) {
040        logAndSetError(DOWNLOAD_EXIST_ERR_KEY);
041      } else {
042        key = download.getKey();
043      }
044    } else {
045      logAndSetError(DOWNLOAD_NULL_ERR_KEY);
046    }
047    return SUCCESS;
048  }
049
050  /**
051   * Download object. Null if the jobId is not a valid download.key.
052   */
053  public Download getDownload() {
054    return download;
055  }
056
057  /**
058   * Notification addresses of the download requested.
059   */
060  public Set<String> getEmails() {
061    return download.getRequest().getNotificationAddresses();
062  }
063
064  public String getHumanRedeableBytesSize(long bytes) {
065    return FileUtils.humanReadableByteCount(bytes, true);
066  }
067
068  /**
069   * Download key.
070   */
071  public String getKey() {
072    return key;
073  }
074
075  public String getPageTitle() {
076    if (hasFieldErrors()) {
077      return getText(TITLE_ERROR_KEY);
078    } else if (download != null && download.isAvailable()) {
079      return getText(TITLE_READY_KEY);
080    }
081    return getText(TITLE_STARTED_KEY);
082  }
083
084  /**
085   * Checks if the download is running.
086   */
087  public boolean isDownloadRunning(String strStatus) {
088    return DownloadsActionUtils.isDownloadRunning(strStatus);
089  }
090
091  public void setKey(String key) {
092    this.key = key;
093  }
094
095  /**
096   * Log and error and set an error to the field jobId.
097   */
098  private void logAndSetError(String messageKey) {
099    String errorText = getText(messageKey, new String[] {key});
100    LOG.error(errorText);
101    addFieldError(DOWNLOAD_KEY_PARAM, errorText);
102  }
103}

025/**
026 * Extends the details action to return a different result name based on the dataset type, so we can use
027 * different freemarker templates for them as they are very different.
028 * For external datasets the page is not found.
029 */
030public class ActivityAction extends DetailAction {
031
032  private static Logger LOG = LoggerFactory.getLogger(ActivityAction.class);
033
034  private final DatasetOccurrenceDownloadUsageService downloadUsageService;
035  private final NameUsageService nameUsageService;
036  private PagingResponse<DatasetOccurrenceDownloadUsage> response;
037  private final PagingRequest request;
038  private static final int DEFAULT_LIMIT = 10;
039
040  @Inject
041  public ActivityAction(DatasetService datasetService, DatasetOccurrenceDownloadUsageService downloadUsageService,
042    NameUsageService nameUsageService, CubeService cubeService, DatasetMetricsService datasetMetricsService,
043    OrganizationService organizationService) {
044    super(datasetService, cubeService, datasetMetricsService, organizationService);
045    this.downloadUsageService = downloadUsageService;
046    this.nameUsageService = nameUsageService;
047    request = new PagingRequest(0, DEFAULT_LIMIT);
048  }
049
050  @Override
051  public String execute() {
052    super.execute();
053    response = downloadUsageService.listByDataset(getId(), request);
054    return SUCCESS;
055  }
056
057  // TODO: the same code is also used in DownloadsAction share it showhow!!!
058  public Map<OccurrenceSearchParameter, LinkedList<String>> getHumanFilter(Predicate p) {
059    if (p != null) {
060      try {
061        // not thread safe!
062        HumanFilterBuilder builder = new HumanFilterBuilder(this.getTexts(), datasetService, nameUsageService, true);
063        return builder.humanFilter(p);
064
065      } catch (Exception e) {
066        LOG.warn("Cannot create human representation for predicate {}", p);
067      }
068    }
069    return null;
070  }
071
072  public Pageable getPage() {
073    return response;
074  }
075
076
077  public String getQueryParams(Predicate p) {
078    return DownloadsAction.getQueryParams(p);
079  }
080
081  public void setOffset(long offset) {
082    request.setOffset(offset);
083  }
084}

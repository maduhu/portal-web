001package org.gbif.portal.action.dataset;
002
003import org.gbif.api.model.Constants;
004import org.gbif.api.service.checklistbank.DatasetMetricsService;
005import org.gbif.api.service.metrics.CubeService;
006import org.gbif.api.service.registry.DatasetService;
007import org.gbif.api.service.registry.OrganizationService;
008import org.gbif.api.vocabulary.DatasetType;
009import org.gbif.portal.action.species.HomeAction;
010import org.gbif.portal.exception.NotFoundException;
011
012import java.util.UUID;
013
014import com.google.inject.Inject;
015
016/**
017 * Extends the details action to return a different result name based on the dataset type, so we can use
018 * different freemarker templates for them as they are very different.
019 * For external datasets the page is not found.
020 */
021public class StatsAction extends DetailAction {
022
023  private static final String OCCURRENCE_RESULT = "occurrence";
024  private static final String CHECKLIST_RESULT = "checklist";
025
026  @Inject
027  public StatsAction(DatasetService datasetService, CubeService cubeService,
028    DatasetMetricsService datasetMetricsService, OrganizationService organizationService) {
029    super(datasetService, cubeService, datasetMetricsService, organizationService);
030  }
031
032  @Override
033  public String execute() {
034    super.execute();
035
036    if (DatasetType.OCCURRENCE == member.getType()) {
037      return OCCURRENCE_RESULT;
038    } else if (DatasetType.CHECKLIST == member.getType()) {
039      return CHECKLIST_RESULT;
040    } else {
041      throw new NotFoundException("External datasets don't have a stats page");
042    }
043  }
044
045  public UUID getColKey() {
046    return HomeAction.COL_KEY;
047  }
048
049  public UUID getNubKey() {
050    return Constants.NUB_DATASET_KEY;
051  }
052}

001package org.gbif.portal.action.dataset;
002
003import org.gbif.api.model.common.InterpretedEnum;
004import org.gbif.api.model.registry.eml.TaxonomicCoverage;
005import org.gbif.api.model.registry.eml.TaxonomicCoverages;
006import org.gbif.api.service.checklistbank.DatasetMetricsService;
007import org.gbif.api.service.metrics.CubeService;
008import org.gbif.api.service.registry.DatasetService;
009import org.gbif.api.service.registry.OrganizationService;
010import org.gbif.api.vocabulary.Rank;
011import org.gbif.portal.action.dataset.util.DisplayableTaxonomicCoverage;
012import org.gbif.portal.action.dataset.util.OrganizedTaxonomicCoverage;
013import org.gbif.portal.action.dataset.util.OrganizedTaxonomicCoverages;
014
015import java.util.ArrayList;
016import java.util.List;
017
018import org.junit.Test;
019
020import static junit.framework.Assert.assertEquals;
021import static org.mockito.Mockito.mock;
022
023public class DatasetBaseActionTest {
024
025  @Test
026  public void testPopulateOrganizedCoverages() {
027    DatasetBaseAction action = new DatasetBaseAction(mock(DatasetService.class),
028      mock(CubeService.class), mock(DatasetMetricsService.class), mock(OrganizationService.class));
029
030    // create coverages #1
031    TaxonomicCoverages coverages1 = new TaxonomicCoverages();
032    List<TaxonomicCoverages> coveragesList = new ArrayList<TaxonomicCoverages>();
033
034    TaxonomicCoverage coverage1 = new TaxonomicCoverage();
035    coverage1.setCommonName("Plants");
036    coverage1.setScientificName("Plantae");
037    coverage1.setRank(new InterpretedEnum("kingdom", Rank.KINGDOM));
038
039
040    TaxonomicCoverage coverage2 = new TaxonomicCoverage();
041    coverage2.setCommonName("Animals");
042    coverage2.setScientificName("Animalia");
043    coverage2.setRank(new InterpretedEnum("kingdom", Rank.KINGDOM));
044
045    List<TaxonomicCoverage> coverageList = new ArrayList<TaxonomicCoverage>();
046    coverageList.add(coverage1);
047    coverageList.add(coverage2);
048
049    coverages1.setCoverages(coverageList);
050    coverages1.setDescription("Coverages #1 description");
051
052    coveragesList.add(coverages1);
053
054    // create coverages #2
055    TaxonomicCoverages coverages2 = new TaxonomicCoverages();
056
057    TaxonomicCoverage coverage_2_1 = new TaxonomicCoverage();
058    coverage_2_1.setScientificName("Equisetophyta");
059    coverage_2_1.setRank(new InterpretedEnum("phylum", Rank.PHYLUM));
060
061    TaxonomicCoverage coverage_2_2 = new TaxonomicCoverage();
062    coverage_2_2.setScientificName("Pteridophyta");
063    coverage_2_2.setRank(new InterpretedEnum("phylum", Rank.PHYLUM));
064
065    TaxonomicCoverage coverage_2_3 = new TaxonomicCoverage();
066    coverage_2_3.setScientificName("Pteridophyta");
067    coverage_2_3.setRank(new InterpretedEnum("FEYELUM", null));
068
069    // only having scientific name
070    TaxonomicCoverage coverage_2_4 = new TaxonomicCoverage();
071    coverage_2_4.setScientificName("Euphyllophytina");
072
073    List<TaxonomicCoverage> coverageList2 = new ArrayList<TaxonomicCoverage>();
074    coverageList2.add(coverage_2_1);
075    coverageList2.add(coverage_2_2);
076    coverageList2.add(coverage_2_3);
077    coverageList2.add(coverage_2_4);
078
079    coverages2.setCoverages(coverageList2);
080    coverages2.setDescription("Coverages #2 description");
081
082    // add 2 coverages to list
083    coveragesList.add(coverages2);
084
085    // return organized coverages
086    List<OrganizedTaxonomicCoverages> organizedCoverages = action.constructOrganizedTaxonomicCoverages(coveragesList);
087
088    // assert some things about 1st coverages
089    assertEquals("Coverages #1 description", organizedCoverages.get(0).getDescription());
090
091    OrganizedTaxonomicCoverage organizedCoverage = organizedCoverages.get(0).getCoverages().get(0);
092    assertEquals(2, organizedCoverage.getDisplayableNames().size());
093
094    DisplayableTaxonomicCoverage displayableCoverage1 = organizedCoverage.getDisplayableNames().get(0);
095    assertEquals("Plantae (Plants)", displayableCoverage1.getDisplayName());
096    assertEquals("Plantae", displayableCoverage1.getScientificName());
097    assertEquals("Plants", displayableCoverage1.getCommonName());
098
099    DisplayableTaxonomicCoverage displayableCoverage2 = organizedCoverage.getDisplayableNames().get(1);
100    assertEquals("Animalia (Animals)", displayableCoverage2.getDisplayName());
101    assertEquals("Animalia", displayableCoverage2.getScientificName());
102    assertEquals("Animals", displayableCoverage2.getCommonName());
103
104    // assert some things about 2nd coverages
105    assertEquals("Coverages #2 description", organizedCoverages.get(1).getDescription());
106    // assert there is an OrganizedTaxonomicCoverage for each unique Rank encountered in TaxonomicCoverage list
107    assertEquals(3, organizedCoverages.get(1).getCoverages().size());
108
109    // check OrganizedTaxonomicCoverage
110    OrganizedTaxonomicCoverage organizedCoverage2_1 = organizedCoverages.get(1).getCoverages().get(0);
111    assertEquals(2, organizedCoverage2_1.getDisplayableNames().size());
112    assertEquals("PHYLUM", organizedCoverages.get(1).getCoverages().get(0).getRank());
113
114    // check OrganizedTaxonomicCoverage for verbatim, uninterpreted Rank called FEYELUM
115    OrganizedTaxonomicCoverage organizedCoverage2_2 = organizedCoverages.get(1).getCoverages().get(1);
116    assertEquals(1, organizedCoverage2_2.getDisplayableNames().size());
117    assertEquals("FEYELUM", organizedCoverage2_2.getRank());
118
119    // check OrganizedTaxonomicCoverage for coverage having no rank, only scientific name
120    OrganizedTaxonomicCoverage organizedCoverage2_3 = organizedCoverages.get(1).getCoverages().get(2);
121    assertEquals(1, organizedCoverage2_3.getDisplayableNames().size());
122    assertEquals("Euphyllophytina", organizedCoverage2_3.getDisplayableNames().get(0).getScientificName());
123  }
124
125}

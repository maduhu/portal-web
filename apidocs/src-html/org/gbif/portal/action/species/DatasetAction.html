001package org.gbif.portal.action.species;
002
003import org.gbif.api.model.checklistbank.NameUsage;
004import org.gbif.api.model.common.paging.PagingResponse;
005import org.gbif.api.model.registry.Dataset;
006import org.gbif.api.service.occurrence.OccurrenceDatasetIndexService;
007import org.gbif.api.vocabulary.DatasetType;
008
009import java.util.Iterator;
010import java.util.List;
011import java.util.SortedMap;
012import java.util.UUID;
013
014import com.google.common.collect.Lists;
015import com.google.inject.Inject;
016
017public class DatasetAction extends UsageBaseAction {
018
019  public class DatasetResult implements Comparable<DatasetResult> {
020
021    private final Dataset dataset;
022    private final Integer numOccurrences;
023    private final NameUsage usage;
024
025    public DatasetResult(Dataset dataset, Integer numOccurrences, NameUsage usage) {
026      this.dataset = dataset;
027      this.numOccurrences = numOccurrences;
028      this.usage = usage;
029    }
030
031    @Override
032    public int compareTo(DatasetResult that) {
033      if (this == that) {
034        return 0;
035      }
036      return dataset.getTitle().toLowerCase().compareTo(that.getDataset().getTitle().toLowerCase());
037    }
038
039    public Dataset getDataset() {
040      return dataset;
041    }
042
043    public Integer getNumOccurrences() {
044      return numOccurrences;
045    }
046
047    public NameUsage getUsage() {
048      return usage;
049    }
050  }
051
052  @Inject
053  private OccurrenceDatasetIndexService occurrenceDatasetService;
054  private DatasetType type;
055  private PagingResponse<DatasetResult> page;
056  private int offset = 0;
057
058  private static final int pageSize = 25;
059
060  @Override
061  public String execute() {
062    loadUsage();
063
064    page = new PagingResponse<DatasetResult>(offset, pageSize);
065
066    if (type == null || type == DatasetType.CHECKLIST) {
067      List<NameUsage> relatedUsages = usageService.listRelated(usage.getNubKey(), getLocale());
068      // remove nub usage itself
069      Iterator<NameUsage> iter = relatedUsages.iterator();
070      while (iter.hasNext()) {
071        if (iter.next().getKey().equals(usage.getKey())) {
072          iter.remove();
073        }
074      }
075
076      page.setCount((long) relatedUsages.size());
077      for (NameUsage u : sublist(relatedUsages, offset, offset + pageSize)) {
078        page.getResults().add(new DatasetResult(datasetService.get(u.getDatasetKey()), null, u));
079      }
080    }
081
082    if ((type == null || type == DatasetType.OCCURRENCE) && usage.getNubKey() != null) {
083      SortedMap<UUID, Integer> occurrenceDatasetCounts =
084        occurrenceDatasetService.occurrenceDatasetsForNubKey(usage.getNubKey());
085      page.setCount((long) occurrenceDatasetCounts.size());
086      for (UUID uuid : sublist(Lists.newArrayList(occurrenceDatasetCounts.keySet()), offset, offset + pageSize)) {
087        page.getResults().add(new DatasetResult(datasetService.get(uuid), occurrenceDatasetCounts.get(uuid), null));
088      }
089    }
090
091    return SUCCESS;
092  }
093
094  public int getOffset() {
095    return offset;
096  }
097
098  public PagingResponse<DatasetResult> getPage() {
099    return page;
100  }
101
102  public DatasetType getType() {
103    return type;
104  }
105
106  public void setOffset(int offset) {
107    this.offset = offset;
108  }
109
110  public void setType(DatasetType type) {
111    this.type = type;
112  }
113}

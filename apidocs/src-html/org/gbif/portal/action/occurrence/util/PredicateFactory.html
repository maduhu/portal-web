001package org.gbif.portal.action.occurrence.util;
002
003import org.gbif.api.model.occurrence.predicate.ConjunctionPredicate;
004import org.gbif.api.model.occurrence.predicate.DisjunctionPredicate;
005import org.gbif.api.model.occurrence.predicate.EqualsPredicate;
006import org.gbif.api.model.occurrence.predicate.GreaterThanOrEqualsPredicate;
007import org.gbif.api.model.occurrence.predicate.LessThanOrEqualsPredicate;
008import org.gbif.api.model.occurrence.predicate.Predicate;
009import org.gbif.api.model.occurrence.predicate.WithinPredicate;
010import org.gbif.api.model.occurrence.search.OccurrenceSearchParameter;
011import org.gbif.api.util.SearchTypeValidator;
012import org.gbif.api.util.VocabularyUtils;
013
014import java.text.SimpleDateFormat;
015import java.util.Date;
016import java.util.List;
017import java.util.Map;
018
019import com.google.common.collect.Lists;
020import com.google.common.collect.Range;
021
022/**
023 * Utility for dealing with the decoding of the request parameters to the
024 * query object to pass into the service.
025 * This parses the URL params which should be from something like the following
026 * into a predicate suitable for launching a download service.
027 *
028 * It understands multi valued parameters and interprets the range format *,100
029 * TAXON_KEY=12&ALTITUDE=1000,2000
030 * (ALTITUDE >= 1000 AND ALTITUDE <= 1000)
031 */
032public class PredicateFactory {
033  private final static String POLYGON = "POLYGON((%s))";
034
035  /**
036   * Builds a full predicate filter from the parameters.
037   * In case no filters exist still return a predicate that matches anything.
038   * @return always some predicate
039   */
040  public Predicate build(Map<String, String[]> params) {
041    // predicates for different parameters. If multiple values for the same parameter exist these are in here already
042    List<Predicate> groupedByParam = Lists.newArrayList();
043    for (String p : params.keySet()) {
044      // recognize valid params by enum name, ignore others
045      OccurrenceSearchParameter param = toEnumParam(p);
046      if (param != null) {
047        // valid parameter
048        Predicate predicate = buildParamPredicate(param, params.get(p));
049        if (predicate != null) {
050          groupedByParam.add(predicate);
051        }
052      }
053    }
054
055    if (groupedByParam.isEmpty()) {
056      // no filter at all
057      return null;
058
059    } else if (groupedByParam.size() == 1) {
060      return groupedByParam.get(0);
061
062    } else {
063      // AND the individual params
064      return new ConjunctionPredicate(groupedByParam);
065    }
066  }
067
068  /**
069   * @param name
070   * @return the search enum or null if it cant be converted
071   */
072  private OccurrenceSearchParameter toEnumParam(String name) {
073    try {
074      return (OccurrenceSearchParameter) VocabularyUtils.lookupEnum(name, OccurrenceSearchParameter.class);
075    } catch (IllegalArgumentException e) {
076      return null;
077    }
078  }
079
080  private Predicate buildParamPredicate(OccurrenceSearchParameter param, String[] values) {
081    List<Predicate> predicates = Lists.newArrayList();
082    for (String v : values) {
083      Predicate p = parsePredicate(param, v);
084      if (p != null) {
085        predicates.add(p);
086      }
087    }
088
089    if (predicates.isEmpty()) {
090      return null;
091
092    } else if (predicates.size() == 1) {
093      return predicates.get(0);
094
095    } else {
096      // OR the individual params
097      return new DisjunctionPredicate(predicates);
098    }
099  }
100
101  private String toIsoDate(Date d) {
102    return new SimpleDateFormat("yyyy-MM-dd").format(d);
103  }
104
105  /**
106   * Converts a value with an optional predicate prefix into a real predicate instance, defaulting to EQUALS.
107   */
108  private Predicate parsePredicate(OccurrenceSearchParameter param, String value) {
109    // geometry filters are special
110    if (OccurrenceSearchParameter.GEOMETRY == param) {
111      return new WithinPredicate(String.format(POLYGON, value));
112    }
113
114    // test for ranges
115    if (SearchTypeValidator.isRange(value)) {
116      Range<?> range;
117      if (Double.class.equals(param.type())) {
118        range = SearchTypeValidator.parseDecimalRange(value);
119      } else if (Integer.class.equals(param.type())) {
120        range = SearchTypeValidator.parseIntegerRange(value);
121      } else if (Date.class.equals(param.type())) {
122        range = SearchTypeValidator.parseDateRange(value);
123        // convert date instances back to strings, but keep the new precision which is now always up to the day!
124        range = SearchTypeValidator.buildRange(
125          range.hasLowerBound() ? toIsoDate((Date) range.lowerEndpoint()) : null,
126          range.hasUpperBound() ? toIsoDate((Date) range.upperEndpoint()) : null
127        );
128      } else {
129        throw new IllegalArgumentException("Ranges are only supported for numeric or date parameter types but received " + param);
130      }
131
132
133      List<Predicate> rangePredicates = Lists.newArrayList();
134      if (range.hasLowerBound()) {
135        rangePredicates.add(new GreaterThanOrEqualsPredicate(param, range.lowerEndpoint().toString()));
136      }
137      if (range.hasUpperBound()) {
138        rangePredicates.add(new LessThanOrEqualsPredicate(param, range.upperEndpoint().toString()));
139      }
140
141      if (rangePredicates.size() == 1) {
142        return rangePredicates.get(0);
143      } else if (rangePredicates.size() > 1) {
144        return new ConjunctionPredicate(rangePredicates);
145      }
146      return null;
147
148    } else {
149      // defaults to an equals predicate with the original value
150      return new EqualsPredicate(param, value);
151    }
152  }
153}

66     * The member's list of lower cased, plain string keywords derived from public tags without a namespace.
67     * This method loads keywords lazily.
68     * @return member's list of keywords
69     */
70    public List<String> getKeywords() {
71      if (keywords == null) {
72        // lazy load keywords
73        Set<String> kws = Sets.newTreeSet();
74        for (Tag t : member.getTags()) {
75          if (!Strings.isNullOrEmpty(t.getValue())) {
76            kws.add(t.getValue().trim().toLowerCase());
77          }
78        }
79        keywords = Lists.newArrayList(kws);
80      }
81      return keywords;
82    }

108   /**
109    * Loads list of CountWrapper Datasets, that is a list of datasets with a record count and a geo reference count
110    * added to each one.
111    *
112    * @param datasetPage page of Datasets
113    */
114   protected void loadCountWrappedDatasets(PagingResponse<Dataset> datasetPage) {
115     for (Dataset d : datasetPage.getResults()) {
116       final long dsCnt;
117       final long dsGeoCnt;
118       if (DatasetType.OCCURRENCE == d.getType()) {
119         dsCnt = cubeService.get(new ReadBuilder().at(OccurrenceCube.DATASET_KEY, d.getKey()));
120         dsGeoCnt = cubeService.get(new ReadBuilder().at(OccurrenceCube.DATASET_KEY, d.getKey()).at(
121           OccurrenceCube.IS_GEOREFERENCED, true));
122 
123       } else if (DatasetType.CHECKLIST == d.getType()) {
124         DatasetMetrics metric = datasetMetricsService.get(d.getKey());
125         dsCnt = (metric == null) ? 0 : metric.getUsagesCount();
126         dsGeoCnt = 0;
127       } else {
128         dsCnt = 0;
129         dsGeoCnt = 0;
130       }
131 
132       datasets.add(new CountWrapper<Dataset>(d, dsCnt, dsGeoCnt));
133     }
134   }

11  /**
12   * A utility class to map countries to specific drupal tag ids used in drupal news feeds for example.
13   */
14  public class DrupalCountryTagMap {
15    private static final Logger LOG = LoggerFactory.getLogger(DrupalCountryTagMap.class);
16    private Map<Country, Integer> map = Maps.newHashMap();
17  
18    /**
19     * @param rawMap key=iso code, value=tag id
20     */
21    public DrupalCountryTagMap(Map<String, String> rawMap) {
22      for (Map.Entry<String, String> ent : rawMap.entrySet()) {
23        Country c = Country.fromIsoCode(ent.getKey());
24        if (c != null) {
25          map.put(c, Integer.parseInt(ent.getValue()));
26        } else {
27          LOG.debug("Ignore unknown drupal tag country {}", ent.getKey());
28        }
29      }
30    }
31    public Integer lookupTag(Country c) {
32      return map.get(c);
33    }

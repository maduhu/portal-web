001package org.gbif.portal.model;
002
003import org.gbif.api.model.checklistbank.VernacularName;
004import org.gbif.api.vocabulary.Language;
005
006import java.util.Collections;
007import java.util.List;
008
009import com.google.common.collect.Lists;
010import org.junit.Test;
011
012import static org.junit.Assert.assertEquals;
013
014public class VernacularLocaleComparatorTest {
015
016  @Test
017  public void testCompare() throws Exception {
018    List<VernacularName> names = Lists.newArrayList();
019    names.add( build(1,Language.GERMAN, "Tanne") );
020    names.add( build(2,Language.DUTCH, "Tinne") );
021    names.add( build(3,Language.DANISH, "t√∏nne") );
022    names.add( build(4,null, "tann") );
023    names.add( build(5,Language.ENGLISH, "White fir") );
024    names.add( build(6,null, "TANNENBAUM") );
025    names.add( build(7,Language.ENGLISH, "FIR") );
026    names.add( build(8,Language.AFRIKAANS, "dannebaum") );
027
028    Collections.sort(names, new VernacularLocaleComparator(Language.GERMAN));
029    assertEquals((Integer) 1, names.get(0).getKey());
030    assertEquals((Integer) 7, names.get(1).getKey());
031    assertEquals((Integer) 5, names.get(2).getKey());
032    assertEquals((Integer) 8, names.get(3).getKey());
033    assertEquals((Integer) 4, names.get(4).getKey());
034    assertEquals((Integer) 6, names.get(5).getKey());
035    assertEquals((Integer) 2, names.get(6).getKey());
036    assertEquals((Integer) 3, names.get(7).getKey());
037
038    Collections.sort(names, new VernacularLocaleComparator(Language.ENGLISH));
039    assertEquals((Integer) 7, names.get(0).getKey());
040    assertEquals((Integer) 5, names.get(1).getKey());
041    assertEquals((Integer) 8, names.get(2).getKey());
042    assertEquals((Integer) 4, names.get(3).getKey());
043    assertEquals((Integer) 1, names.get(4).getKey());
044    assertEquals((Integer) 6, names.get(5).getKey());
045    assertEquals((Integer) 2, names.get(6).getKey());
046    assertEquals((Integer) 3, names.get(7).getKey());
047
048    Collections.sort(names, new VernacularLocaleComparator(null));
049    assertEquals((Integer) 7, names.get(0).getKey());
050    assertEquals((Integer) 5, names.get(1).getKey());
051    assertEquals((Integer) 8, names.get(2).getKey());
052    assertEquals((Integer) 4, names.get(3).getKey());
053    assertEquals((Integer) 1, names.get(4).getKey());
054    assertEquals((Integer) 6, names.get(5).getKey());
055    assertEquals((Integer) 2, names.get(6).getKey());
056    assertEquals((Integer) 3, names.get(7).getKey());
057  }
058
059
060
061  private VernacularName build(int key, Language lang, String name) {
062    VernacularName v = new VernacularName();
063    v.setKey(key);
064    v.setLanguage(lang);
065    v.setVernacularName(name);
066    return v;
067  }
068}

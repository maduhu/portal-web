011  /**
012   * This test uses the dataset page to search for all datasets, clicks on the second one and verifies that
013   * the details page for the selected dataset is shown.
014   */
015  @Test
016  public void testSearchForAnyDataset() {
017    getUrl(getBaseUrl() + "/dataset");
018    driver.findElement(By.cssSelector("button.search_button")).click();
019    WebElement element = driver.findElement(By.xpath("//div[@class='result' and position() = 2]/h2/a"));
020    String sourceName = element.getText();
021    element.click();
022    String targetName = driver.findElement(By.xpath("//div[@id='infoband']/div/h1")).getText();
023    assertTrue("Link target and dataset name must be the same", sourceName.contains(targetName.replaceAll("â€¦", "")));
024  }

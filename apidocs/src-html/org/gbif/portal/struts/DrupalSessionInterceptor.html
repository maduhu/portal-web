001package org.gbif.portal.struts;
002
003import org.gbif.api.model.common.User;
004import org.gbif.api.service.common.UserService;
005import org.gbif.drupal.mybatis.UserServiceImpl;
006import org.gbif.portal.config.Config;
007
008import java.util.Map;
009
010import javax.servlet.http.Cookie;
011import javax.servlet.http.HttpServletRequest;
012
013import com.google.inject.Inject;
014import com.opensymphony.xwork2.ActionInvocation;
015import com.opensymphony.xwork2.interceptor.AbstractInterceptor;
016import org.apache.struts2.StrutsStatics;
017import org.slf4j.Logger;
018import org.slf4j.LoggerFactory;
019
020import static org.gbif.portal.config.Constants.SESSION_USER;
021
022/**
023 * An interceptor that puts the current user into the session if its not yet existing and a request principle is given.
024 */
025public class DrupalSessionInterceptor extends AbstractInterceptor {
026
027  private static final long serialVersionUID = 8383908769617385684L;
028
029  private static final Logger LOG = LoggerFactory.getLogger(DrupalSessionInterceptor.class);
030
031  private final UserServiceImpl userService;
032  private final String COOKIE_NAME;
033  private final String DRUPAL_SESSION_NAME = "drupal_session";
034
035  @Inject
036  public DrupalSessionInterceptor(UserService userService, Config cfg) {
037    this.userService = (UserServiceImpl) userService;
038    COOKIE_NAME = cfg.getDrupalCookieName();
039  }
040
041  @Override
042  public String intercept(final ActionInvocation invocation) throws Exception {
043    final Map session = invocation.getInvocationContext().getSession();
044    final HttpServletRequest request =
045      (HttpServletRequest) invocation.getInvocationContext().get(StrutsStatics.HTTP_REQUEST);
046    final Cookie cookie = findDrupalCookie(request);
047
048    // FOR DEBUGGING DURING DEVELOPMENT ONLY - you can switch a user without authenticating him!!!
049    /*
050     * if (!Strings.isNullOrEmpty(request.getParameter("DEBUG_USER"))) {
051     * LOG.info("Force login for user {}", request.getParameter("DEBUG_USER"));
052     * User user = userService.get(request.getParameter("DEBUG_USER"));
053     * session.put(SESSION_USER, user);
054     * session.put(DRUPAL_SESSION_NAME, "");
055     * return invocation.invoke();
056     * }
057     */
058
059    User user = (User) session.get(SESSION_USER);
060
061    // invalidate current user if cookie is missing or drupal session is different
062    if (user != null && (cookie == null || !cookie.getValue().equals(session.get(DRUPAL_SESSION_NAME)))) {
063      LOG.info("Invalidate session for user {}", user.getUserName());
064      user = null;
065      session.clear();
066    }
067
068    if (user == null && cookie != null) {
069      // user logged into drupal
070      user = userService.getBySession(cookie.getValue());
071      if (user == null) {
072        LOG.warn("Drupal cookie contains invalid session {}", cookie.getValue());
073      } else {
074        session.put(SESSION_USER, user);
075        session.put(DRUPAL_SESSION_NAME, cookie.getValue());
076        LOG.info("Activate session {} for user {}", cookie.getValue(), user.getUserName());
077      }
078    }
079
080    return invocation.invoke();
081  }
082
083  private Cookie findDrupalCookie(HttpServletRequest request) {
084    if (request.getCookies() != null) {
085      for (Cookie cookie : request.getCookies()) {
086        if (COOKIE_NAME.equalsIgnoreCase(cookie.getName())) {
087          return cookie;
088        }
089      }
090    }
091    return null;
092  }
093
094}

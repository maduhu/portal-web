001package org.gbif.portal.model;
002
003
004import org.gbif.api.model.common.search.SearchRequest;
005import org.gbif.api.model.occurrence.search.OccurrenceSearchParameter;
006
007import java.util.EnumSet;
008
009import javax.servlet.http.HttpServletRequest;
010
011import com.google.common.base.Enums;
012import com.google.common.base.Optional;
013
014/**
015 * Class that represents the configuration for the HTML table shown in the occurrence search page.
016 */
017public class OccurrenceTable {
018
019  /**
020   * Enum that represents the visible columns in the occurrence page.
021   */
022  public static enum OccurrenceColumn {
023    SUMMARY, LOCATION, BASIS_OF_RECORD, DATE;
024  }
025
026  /**
027   * Enum that represents the visible information of the summary column in the occurrence page.
028   */
029  public static enum OccurrenceSummaryField {
030    OCCURRENCE_KEY, CATALOG_NUMBER, COLLECTION_CODE, COLLECTOR_NAME, INSTITUTION, SCIENTIFIC_NAME, DATASET, MODIFIED;
031  }
032
033  private static final OccurrenceSearchParameter[] OCC_LOCATION_PARAMS = new OccurrenceSearchParameter[] {
034    OccurrenceSearchParameter.ALTITUDE, OccurrenceSearchParameter.DEPTH,
035    OccurrenceSearchParameter.LATITUDE, OccurrenceSearchParameter.LONGITUDE, OccurrenceSearchParameter.GEOMETRY,
036    OccurrenceSearchParameter.GEOREFERENCED, OccurrenceSearchParameter.COUNTRY};
037
038  private static final OccurrenceSearchParameter[] OCC_DATE_PARAMS = new OccurrenceSearchParameter[] {
039    OccurrenceSearchParameter.MONTH, OccurrenceSearchParameter.YEAR};
040
041  // Default list of summary fields
042  private static EnumSet<OccurrenceSummaryField> defaulSummaryFields = EnumSet.of(
043    OccurrenceSummaryField.OCCURRENCE_KEY,
044    OccurrenceSummaryField.CATALOG_NUMBER, OccurrenceSummaryField.SCIENTIFIC_NAME,
045    OccurrenceSummaryField.DATASET);
046
047
048  // Columns HTTP parameter
049  private static final String COLUMNS_PARAM = "columns";
050
051  // summary fields HTTP parameter
052  private static final String SUMMARY_FIELDS_PARAM = "summary";
053
054  private final EnumSet<OccurrenceColumn> columns;
055
056  private final EnumSet<OccurrenceSummaryField> summaryColumn;
057
058
059  /**
060   * Default constructor. Creates a instance containing the default elements.
061   */
062  public OccurrenceTable() {
063    this.columns = EnumSet.allOf(OccurrenceColumn.class);
064    this.summaryColumn = EnumSet.copyOf(defaulSummaryFields);
065  }
066
067
068  /**
069   * Creates an instance container the columns and summary fields set in the request parameter.
070   */
071  public OccurrenceTable(HttpServletRequest request, SearchRequest<OccurrenceSearchParameter> searchRequest) {
072    this.columns = retrieveColumns(request, searchRequest);
073    this.summaryColumn = retrieveSummaryFields(request, searchRequest);
074  }
075
076  /**
077   * Holds the list of visible columns, the default value is OccurrenceColumn.values(). Values can be set by setting the
078   * parameter 'columns'.
079   */
080  public EnumSet<OccurrenceColumn> getColumns() {
081    return columns;
082  }
083
084  /**
085   * Gets the html.table.td colspan value for the summary column.
086   */
087  public int getSummaryColspan() {
088    // +1 because summary column is always shown
089    return columns.size() + 1;
090  }
091
092  /**
093   * Summary column contains basic occurrence information, its default values are OCCURRENCE_KEY, CATALOGUE_NUMBER,
094   * SCIENTIFIC_NAME, INSTITUTION.
095   */
096  public EnumSet<OccurrenceSummaryField> getSummaryColumn() {
097    return summaryColumn;
098  }
099
100
101  /**
102   * Checks if the column parameter exists in the list of columns.
103   */
104  public boolean hasColumn(OccurrenceColumn column) {
105    return columns.contains(column);
106  }
107
108  /**
109   * Checks if the column name parameter exists in the list of columns.
110   */
111  public boolean hasColumn(String column) {
112    return columns.contains(OccurrenceColumn.valueOf(column));
113  }
114
115  /**
116   * Checks if the field parameter exists in the list of summary fields.
117   */
118  public boolean hasSummaryField(OccurrenceSummaryField field) {
119    return summaryColumn.contains(field);
120  }
121
122
123  /**
124   * Checks if the column name parameter exists in the list of summary fields.
125   */
126  public boolean hasSummaryField(String column) {
127    return summaryColumn.contains(OccurrenceSummaryField.valueOf(column));
128  }
129
130  private boolean isParameterPresent(SearchRequest<OccurrenceSearchParameter> request,
131    OccurrenceSearchParameter... parameters) {
132    for (OccurrenceSearchParameter parameter : parameters) {
133      if (request.getParameters().containsKey(parameter)) {
134        return true;
135      }
136    }
137    return false;
138  }
139
140  /**
141   * Retrieve the columns from the request parameter.
142   * The default fields are returned if no value is gotten from the request.
143   */
144  private EnumSet<OccurrenceColumn> retrieveColumns(HttpServletRequest request,
145    SearchRequest<OccurrenceSearchParameter> searchRequest) {
146    EnumSet<OccurrenceColumn> requestCols = retrieveEnumParams(request, OccurrenceColumn.class, COLUMNS_PARAM);
147    if (requestCols.isEmpty()) {
148      requestCols = EnumSet.allOf(OccurrenceColumn.class);
149    }
150    if (isParameterPresent(searchRequest, OCC_LOCATION_PARAMS)) {
151      requestCols.add(OccurrenceColumn.LOCATION);
152    }
153    if (isParameterPresent(searchRequest, OccurrenceSearchParameter.BASIS_OF_RECORD)) {
154      requestCols.add(OccurrenceColumn.BASIS_OF_RECORD);
155    }
156    if (isParameterPresent(searchRequest, OCC_DATE_PARAMS)) {
157      requestCols.add(OccurrenceColumn.DATE);
158    }
159    return requestCols;
160  }
161
162
163  /**
164   * Retrieve an EnumSet containing a list of Enum literals if type T that can be obtained from the request parameter
165   * paramName.
166   */
167  private <T extends Enum<T>> EnumSet<T> retrieveEnumParams(HttpServletRequest request, Class<T> enumClass,
168    String paramName) {
169    EnumSet<T> allCols = EnumSet.noneOf(enumClass);
170    final String[] values = request.getParameterValues(paramName);
171    if (values != null) {
172      for (String paramValue : values) {
173        for (String value : paramValue.split(",")) {
174          Optional<T> optValue = Enums.getIfPresent(enumClass, value);
175          if (optValue.isPresent()) {
176            allCols.add(optValue.get());
177          }
178        }
179      }
180    }
181    return allCols;
182  }
183
184
185  /**
186   * Retrieve the summary fields from the request.
187   * The default fields are returned if no value is gotten from the request.
188   */
189  private EnumSet<OccurrenceSummaryField> retrieveSummaryFields(HttpServletRequest request,
190    SearchRequest<OccurrenceSearchParameter> searchRequest) {
191    EnumSet<OccurrenceSummaryField> fields =
192      retrieveEnumParams(request, OccurrenceSummaryField.class, SUMMARY_FIELDS_PARAM);
193    if (fields.isEmpty()) {
194      fields.addAll(defaulSummaryFields);
195    }
196    if (isParameterPresent(searchRequest, OccurrenceSearchParameter.CATALOG_NUMBER)) {
197      fields.add(OccurrenceSummaryField.CATALOG_NUMBER);
198    }
199    if (isParameterPresent(searchRequest, OccurrenceSearchParameter.COLLECTION_CODE)) {
200      fields.add(OccurrenceSummaryField.COLLECTION_CODE);
201    }
202    if (isParameterPresent(searchRequest, OccurrenceSearchParameter.COLLECTOR_NAME)) {
203      fields.add(OccurrenceSummaryField.COLLECTOR_NAME);
204    }
205    if (isParameterPresent(searchRequest, OccurrenceSearchParameter.INSTITUTION_CODE)) {
206      fields.add(OccurrenceSummaryField.INSTITUTION);
207    }
208    if (isParameterPresent(searchRequest, OccurrenceSearchParameter.TAXON_KEY)) {
209      fields.add(OccurrenceSummaryField.SCIENTIFIC_NAME);
210    }
211    if (isParameterPresent(searchRequest, OccurrenceSearchParameter.MODIFIED)) {
212      fields.add(OccurrenceSummaryField.MODIFIED);
213    }
214    return fields;
215  }
216}

001package org.gbif.portal.config;
002
003import org.gbif.api.model.checklistbank.DatasetMetrics;
004import org.gbif.api.model.registry.Dataset;
005import org.gbif.api.service.checklistbank.DatasetMetricsService;
006import org.gbif.api.service.registry.DatasetService;
007import org.gbif.portal.action.ActionTestUtil;
008
009import java.util.UUID;
010
011import com.google.inject.Injector;
012import org.junit.Before;
013import org.junit.Test;
014
015import static org.junit.Assert.assertNull;
016
017public class PortalModuleIT {
018
019  private Injector injector;
020
021  @Before
022  public void setup() {
023    // will make a new instance of PortalModule(), calling its configure() which installs the various clients
024    injector = ActionTestUtil.initTestInjector();
025  }
026
027  /**
028   * Ultimately, we want to test that the Client has been properly installed, and any interceptors have been properly
029   * bound. In particular, the HttpErrorResponseInterceptor will have been bound to every single public method in
030   * every single client package. This binding is done in the client's super class GbifWsClientModule configure().
031   * </p>
032   * Please note, that if one client has had the NotFoundToNullInterceptor bound, we can rest assured all the others
033   * will have been bound as well. Therefore, we only need to test one service.
034   * </p>
035   * This test uses DatasetMetricsService, bound to DatasetMetricsWsClient in the ChecklistBankWsServiceClientModule.
036   * This test tries to retrieve metrics for a checklist calling: public DatasetMetrics get(UUID datasetKey).
037   * This test intentionally uses a checklist that doesn't exist. The service returns a 204 (no content) response.
038   * If the HttpErrorResponseInterceptor has been bound, the 204 response is converted to NULL.
039   * 
040   * @see org.gbif.ws.client.interceptor.HttpErrorResponseInterceptor#invoke(org.aopalliance.intercept.MethodInvocation)
041   */
042  @Test
043  public void testChecklistBankWsClientModuleInstalled() {
044    DatasetMetricsService cms = injector.getInstance(org.gbif.api.service.checklistbank.DatasetMetricsService.class);
045    DatasetMetrics metrics = cms.get(UUID.randomUUID());
046    assertNull(metrics);
047  }
048
049  /**
050   * This test tries to retrieve a dataset calling: public Dataset get(@PathParam("key") String key).
051   * This test intentionally uses a dataset that doesn't exist. The web service returns a 404 (not found) response.
052   * The final response should be NULL, no exception.
053   * 
054   * @see org.gbif.ws.client.interceptor.HttpErrorResponseInterceptor#invoke(org.aopalliance.intercept.MethodInvocation)
055   */
056  @Test
057  public void testRegistryWsClientModuleInstalled() {
058    DatasetService ds = injector.getInstance(DatasetService.class);
059    UUID key = UUID.randomUUID();
060    Dataset dataset = ds.get(key);
061    assertNull(dataset);
062  }
063}

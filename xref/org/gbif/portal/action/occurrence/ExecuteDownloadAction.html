26    private final PredicateFactory predicateFactory = new PredicateFactory();
27  
28    @Inject
29    private DownloadRequestService downloadRequestService;
30  
31    private String key;
32    // optional additional email notifications
33    private Set<String> emails = Sets.newHashSet();
34  
35    @Override
36    public String execute() {
37      @SuppressWarnings("unchecked")
38      Predicate p = predicateFactory.build(getServletRequest().getParameterMap());
39      LOG.info("Predicate build for passing to download [{}]", p);
40      emails.add(getCurrentUser().getEmail());
41      DownloadRequest download = new DownloadRequest(p, getCurrentUser().getUserName(), emails);
42      key = downloadRequestService.create(download);
43      return SUCCESS;
44    }

12  /**
13   * Custom struts URL renderer that takes a global configuration to turn on/off the addition of the servlet context
14   * to a URL. This is needed for GBIF to run the portal with a /portal context, but expose it publicly behind
15   * varnish in the root context.
16   */
17  public class ConfigurableServletUrlRenderer extends ServletUrlRenderer {
18    private static final Logger LOG = LoggerFactory.getLogger(ConfigurableServletUrlRenderer.class);
19  
20    private final boolean addContext;
21  
22    @Inject("struts.url.includeContext")
23    public ConfigurableServletUrlRenderer(String addContext) {
24      LOG.info("Configure struts tags to include the servlet context: " + addContext);
25      this.addContext = Boolean.parseBoolean(addContext);
26    }
27  
28    @Override
29    public void renderUrl(Writer writer, UrlProvider urlComponent) {
30      urlComponent.setIncludeContext(addContext);
31      super.renderUrl(writer, urlComponent);
32    }
33  
34    @Override
35    public void renderFormUrl(Form formComponent) {
36      formComponent.setIncludeContext(addContext);
37      super.renderFormUrl(formComponent);
38    }
39  }

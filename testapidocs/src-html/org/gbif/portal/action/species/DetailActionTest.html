001package org.gbif.portal.action.species;
002
003import org.gbif.api.model.checklistbank.NameUsageContainer;
004import org.gbif.api.model.checklistbank.VernacularName;
005import org.gbif.api.vocabulary.Language;
006import org.gbif.portal.action.ActionTestUtil;
007
008import java.util.List;
009import java.util.Map;
010
011import com.google.common.collect.ImmutableList;
012import org.apache.struts2.StrutsJUnit4TestCase;
013import org.junit.Before;
014import org.junit.Test;
015
016import static org.junit.Assert.assertEquals;
017import static org.junit.Assert.assertNotNull;
018
019
020public class DetailActionTest extends StrutsJUnit4TestCase<DetailAction> {
021
022  // Utility builder
023  private static VernacularName vernacularOf(String name, Language language, Boolean plural) {
024    VernacularName n = new VernacularName();
025    n.setVernacularName(name);
026    n.setLanguage(language);
027    n.setPlural(plural);
028    return n;
029  }
030
031  private DetailAction action;
032
033  @Before
034  public void setup() {
035    action = ActionTestUtil.initTestInjector().getInstance(DetailAction.class);
036  }
037
038  @Test
039  public void testHabitats() throws Exception {
040    assertNotNull(action);
041    assertNotNull(action.getHabitats());
042    assertEquals(0, action.getHabitats().size());
043    action.appendHabitat(true, "enum.habitat.freshwater");
044    assertEquals(1, action.getHabitats().size());
045    action.appendHabitat(false, "enum.habitat.terrestrial");
046    assertEquals(2, action.getHabitats().size());
047    action.appendHabitat(true, "THE_MOON");
048    assertEquals(3, action.getHabitats().size());
049    // surely we don't translate THE_MOON
050    assertEquals("THE_MOON", action.getHabitats().get(2));
051  }
052
053  @Test
054  public void testVernaculars() {
055    List<VernacularName> source = ImmutableList.of(
056      vernacularOf("Z", Language.ENGLISH, null),
057      vernacularOf("Z", null, null),
058      vernacularOf("A", Language.ABKHAZIAN, null),
059      vernacularOf("A", Language.ABKHAZIAN, true),
060      vernacularOf("B", Language.ABKHAZIAN, null));
061    action.usage = new NameUsageContainer();
062    action.usage.setVernacularNames(source);
063    action.populateVernacularNames();
064
065    Map<String, List<VernacularName>> result = action.getVernacularNames();
066    assertEquals(4, result.size());
067    assertNotNull(result.get("a||ab")); // note that is lowercase
068    assertEquals(2, result.get("a||ab").size());
069  }
070}

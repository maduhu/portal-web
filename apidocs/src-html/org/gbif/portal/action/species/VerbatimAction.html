001package org.gbif.portal.action.species;
002
003import org.gbif.api.model.checklistbank.VerbatimNameUsage;
004import org.gbif.portal.exception.NotFoundException;
005
006import org.slf4j.Logger;
007import org.slf4j.LoggerFactory;
008
009public class VerbatimAction extends UsageBaseAction {
010  private static final Logger LOG = LoggerFactory.getLogger(VerbatimAction.class);
011
012  private VerbatimNameUsage verbatim;
013
014  @Override
015  public String execute() {
016
017    loadUsage();
018
019    verbatim = usageService.getVerbatim(id);
020    if (verbatim == null) {
021      throw new NotFoundException("No verbatim usage found with id " + id);
022    }
023
024    return SUCCESS;
025  }
026
027  public VerbatimNameUsage getVerbatim() {
028    return verbatim;
029  }
030}

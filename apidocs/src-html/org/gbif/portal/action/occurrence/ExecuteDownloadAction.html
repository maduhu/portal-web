001package org.gbif.portal.action.occurrence;
002
003import org.gbif.api.model.occurrence.DownloadRequest;
004import org.gbif.api.model.occurrence.predicate.Predicate;
005import org.gbif.api.service.occurrence.DownloadRequestService;
006import org.gbif.portal.action.BaseAction;
007import org.gbif.portal.action.occurrence.util.PredicateFactory;
008
009import java.util.Set;
010
011import com.google.common.base.Splitter;
012import com.google.common.collect.Sets;
013import com.google.inject.Inject;
014import org.slf4j.Logger;
015import org.slf4j.LoggerFactory;
016
017/**
018 * Executes a download request.
019 * This action has to be executed using a POST method, and error is thrown in any other case.
020 */
021public class ExecuteDownloadAction extends BaseAction {
022
023  private static final long serialVersionUID = 3653614424275432914L;
024  private static final Logger LOG = LoggerFactory.getLogger(ExecuteDownloadAction.class);
025  private static final Splitter EMAIL_SPLITTER = Splitter.on(';').trimResults().omitEmptyStrings();
026  private final PredicateFactory predicateFactory = new PredicateFactory();
027
028  @Inject
029  private DownloadRequestService downloadRequestService;
030
031  private String key;
032  // optional additional email notifications
033  private Set<String> emails = Sets.newHashSet();
034
035  @Override
036  public String execute() {
037    @SuppressWarnings("unchecked")
038    Predicate p = predicateFactory.build(getServletRequest().getParameterMap());
039    LOG.info("Predicate build for passing to download [{}]", p);
040    emails.add(getCurrentUser().getEmail());
041    DownloadRequest download = new DownloadRequest(p, getCurrentUser().getUserName(), emails);
042    key = downloadRequestService.create(download);
043    return SUCCESS;
044  }
045
046
047  /**
048   * Download notification addresses.
049   */
050  public Set<String> getEmails() {
051    return emails;
052  }
053
054  /**
055   * Download key.
056   */
057  public String getKey() {
058    return key;
059  }
060
061  /**
062   * Sets the email field, the parameter is split by ';'.
063   */
064  public void setEmails(String emails) {
065    this.emails = Sets.newHashSet(EMAIL_SPLITTER.split(emails));
066  }
067
068}

001package org.gbif.portal.action.species;
002
003import org.gbif.api.model.checklistbank.Description;
004import org.gbif.api.vocabulary.Language;
005
006import java.util.List;
007import java.util.Map;
008
009import com.google.common.base.Strings;
010import com.google.common.collect.Lists;
011import com.google.common.collect.Maps;
012
013/**
014 * Simple class wrapping all descriptions for a species and exposing methods used to generate a table of content menu.
015 * This could become part of the regular CLB api at some point, but for now we use the basic listByUsage methods of
016 * the DescriptionService to feed the ToC with full descriptions.
017 */
018public class DescriptionToc {
019
020  private final Map<Language, Map<String, List<Integer>>> descriptionToc = Maps.newTreeMap();
021
022  public void addDescription(Description description) {
023    String topic = "general";
024    if (!Strings.isNullOrEmpty(description.getType())) {
025      topic = description.getType().toLowerCase();
026    }
027
028    Language lang;
029    if (description.getLanguage() == null) {
030      // default to english
031      lang = Language.ENGLISH;
032    } else {
033      lang = description.getLanguage();
034    }
035
036    if (!descriptionToc.containsKey(lang)) {
037      descriptionToc.put(lang, Maps.<String, List<Integer>>newTreeMap());
038    }
039    if (!descriptionToc.get(lang).containsKey(topic)) {
040      descriptionToc.get(lang).put(topic, Lists.<Integer>newArrayList());
041    }
042    descriptionToc.get(lang).get(topic).add(description.getKey());
043  }
044
045  public boolean isEmpty() {
046    return descriptionToc.isEmpty();
047  }
048
049  /**
050   * @return list of all languages available for this ToC
051   */
052  public List<Language> listLanguages() {
053    return Lists.newArrayList(descriptionToc.keySet());
054  }
055
056  /**
057   * @return map of all topics for a given language with a list of description keys for each language
058   */
059  public Map<String, List<Integer>> listTopicEntries(Language lang) {
060    if (descriptionToc.containsKey(lang)) {
061      return descriptionToc.get(lang);
062    }
063    return Maps.newHashMap();
064  }
065
066}

72    /**
73     * If the text matches query text, and doesn't contain any highlighting, the missing highlighting is added. Comparison
74     * between the text and query text is case insensitive, and uses the whole query text (no stemming). If the first
75     * comparison fails, the query text is converted to its ASCII equivalent, and a second comparison performed.
76     * </br>
77     * If the title text already contains highlighting, no action is taken and the unchanged .
78     * </br>
79     * Please note: this extraordinary measure is needed for a the full text field named dataset_title_ngram.
80     * Solr supports highlighting for fixed ngram fields only (a fixed ngram is field with
81     * minGramSize = maxGramSize) which is not the case for this field. For this reason, sometimes there is highlighting,
82     * but for times when it's missing, this method will add the missing highlighting.
83     * 
84     * @param t dataset title
85     * @param q search query text
86     * @return dataset title updated with missing highlighting if necessary
87     */
88    public static String addMissingHighlighting(String t, String q) {
89      if (!Strings.isNullOrEmpty(t) && !Strings.isNullOrEmpty(q) && !BaseSearchAction.HL_MATCHER.matcher(t).find()) {
90        Pattern pattern = Pattern.compile(Pattern.quote(q), Pattern.CASE_INSENSITIVE | Pattern.UNICODE_CASE);
91        Matcher matcher = pattern.matcher(t);
92        if (matcher.find()) {
93          t = matcher.replaceAll(HL_PRE + q + HL_POST);
94        } else {
95          // try converting the query text to its ASCII equivalent, and check again. E.g. "straße" to "strasse"
96          q = foldToAscii(q);
97          pattern = Pattern.compile(Pattern.quote(q), Pattern.CASE_INSENSITIVE | Pattern.UNICODE_CASE);
98          matcher = pattern.matcher(t);
99          if (matcher.find()) {
100           t = matcher.replaceAll(HL_PRE + q + HL_POST);
101         }
102       }
103     }
104     return t;
105   }

107   /**
108    * Checks if the dataset search result match (highlighted text) only occurred in the full text field.
109    * This method goes through all highlighted fields that may be highlighted:
110    * 
111    * <pre>
112    * <arr name="hl.fl">
113    *   <str>dataset_title</str>
114    *   <str>keyword</str>
115    *   <str>iso_country_code</str>
116    *   <str>owning_organization_title</str>
117    *   <str>hosting_organization_title</str>
118    *   <str>description</str>
119    * </arr>
120    * </pre>
121    * 
122    * If there is no highlighted text in any of these fields, it can be inferred that the match must have occurred in
123    * the full text field.
124    * </br>
125    * If the query text was null, there can be no matching anyways so the method just returns false.
126    * 
127    * @param result DatasetSearchResult
128    * @return whether a match only occurred on the full text field or not
129    */
130   public static boolean isFullTextMatchOnly(DatasetSearchResult result, String query) {
131     if (Strings.isNullOrEmpty(query)) {
132       return false;
133     }
134     final String keywords = result.getKeywords() == null ? "" : TOKEN_JOINER.join(result.getKeywords());
135     if (isHighlightedText(result.getTitle())
136       || isHighlightedText(result.getDescription())
137       || isHighlightedText(result.getOwningOrganizationTitle())
138       || isHighlightedText(result.getHostingOrganizationTitle())
139       || isHighlightedText(keywords)) {
140 
141       return false;
142     }
143     // otherwise, it must have been a match against the full_text field
144     return true;
145   }

147   /**
148    * Uses the solr.ASCIIFoldingFilter to convert a string to its ASCII equivalent. See solr documentation for full
149    * details.
150    * </br>
151    * When doing the conversion, this method mirrors GBIF's registry-solr schema configuration for
152    * <fieldType name="text_auto_ngram">. For example, it uses the KeywordTokenizer that treats the entire string as a
153    * single token, regardless of its content. See the solr documentation for more details.
154    * </br>
155    * This method is needed when checking if the query string matches the dataset title. For example, if the query
156    * string is "straße", it won't match the dataset title "Schulhof Gymnasium Hürth Bonnstrasse" unless "straße" gets
157    * converted to its ASCII equivalent "strasse".
158    * 
159    * @param q query string
160    * @return query string converted to ASCII equivalent
161    * @see org.gbif.portal.action.dataset.SearchAction#addMissingHighlighting(String, String)
162    * @see org.apache.lucene.analysis.miscellaneous.ASCIIFoldingFilter
163    * @see org.apache.lucene.analysis.core.KeywordTokenizer
164    */
165   protected static String foldToAscii(String q) {
166     if (!Strings.isNullOrEmpty(q)) {
167       ASCIIFoldingFilter filter = null;
168       try {
169         StringReader reader = new StringReader(q);
170         TokenStream stream = new KeywordTokenizer(reader);
171         filter = new ASCIIFoldingFilter(stream);
172         CharTermAttribute termAtt = filter.addAttribute(CharTermAttribute.class);
173         filter.reset();
174         filter.incrementToken();
175         // converted q to ASCII equivalent and return it
176         return termAtt.toString();
177       } catch (IOException e) {
178         // swallow
179       } finally {
180         if (filter != null) {
181           try {
182             filter.end();
183             filter.close();
184           } catch (IOException e) {
185             // swallow
186           }
187         }
188       }
189     }
190     return q;
191   }

193   @Override
194   public String execute() {
195     super.execute();
196     // replace organisation keys with real names
197     lookupFacetTitles(DatasetSearchParameter.HOSTING_ORG, getOrgTitle);
198     lookupFacetTitles(DatasetSearchParameter.OWNING_ORG, getOrgTitle);
199     lookupFacetTitles(DatasetSearchParameter.TYPE, getDatasetTypeTitle);
200     lookupFacetTitles(DatasetSearchParameter.PUBLISHING_COUNTRY, getCountryTitle);
201 
202     // populate counts
203     for (DatasetSearchResult dsr : getSearchResponse().getResults()) {
204       if (DatasetType.OCCURRENCE == dsr.getType()) {
205         Long count = occurrenceCube.get(new ReadBuilder().at(OccurrenceCube.DATASET_KEY, dsr.getKey()));
206         recordCounts.put(dsr.getKey(), count);
207       } else if (DatasetType.CHECKLIST == dsr.getType()) {
208         // Client response status 204 (Equal to no content) gets converted into NULL
209         // See HttpErrorResponseInterceptor.java in gbif-common-ws for more information
210         DatasetMetrics metrics = checklistMetricsService.get(dsr.getKey());
211         if (metrics != null) {
212           recordCounts.put(dsr.getKey(), Long.valueOf(metrics.getUsagesCount()));
213         }
214       }
215     }
216 
217 
218     return SUCCESS;
219   }

225   /**
226    * Initializes the getTitle functions

228   private void initGetTitleFunctions() {
229     getOrgTitle = new Function<String, String>() {
230 
231       @Override
232       public String apply(String key) {
233         if (!Strings.isNullOrEmpty(key)) {
234           try {
235             return orgService.get(UUID.fromString(key)).getTitle();
236           } catch (Exception e) {
237           }
238         }
239         return null;
240       }
241     };
242 
243     getDatasetTypeTitle = new Function<String, String>() {
244 
245       @Override
246       public String apply(String name) {
247         return getEnumTitle("datasettype", name);
248       }
249     };
250 
251     getCountryTitle = new Function<String, String>() {
252 
253       @Override
254       public String apply(String name) {
255         try {
256           Country c = (Country) VocabularyUtils.lookupEnum(name, Country.class);
257           return c.getTitle();
258 
259         } catch (IllegalArgumentException e) {
260           return name;
261         }
262       }
263     };
